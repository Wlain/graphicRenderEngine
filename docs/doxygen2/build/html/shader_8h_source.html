96 enum class StencilFunc
97 {
98 Never = GL_NEVER, // 无论模板值为何值，都不能通过模板测试
99 Less = GL_LESS, // 测试模板值是否小于ref值
100 Equal = GL_EQUAL, // 测试模板值是否等于ref值
101 LEqual = GL_LEQUAL, // 测试模板值是否小于等于ref值
102 Greater = GL_GREATER, // 测试模板值是否大于ref值
103 NotEqual = GL_NOTEQUAL, // 测试模板值是否不等于ref值
104 GEqual = GL_GEQUAL, // 测试模板值是否大于等于ref值
105 Always = GL_ALWAYS, // 无论模板值为何值，总是能通过模板测试。
106 Disabled // 关闭模板测试
107 };

109 enum class StencilOp
110 {
111 Keep = GL_KEEP, // 保持当前值
112 Zero = GL_ZERO, // 将模板值清0
113 Replace = GL_REPLACE, // 模板值将会被替换为glStencilFunc中的ref参数值。
114 Incr = GL_INCR, // 如果模板值小于最大值，模板值将会+1
115 IncrWrap = GL_INCR_WRAP, // 和GL_INCR一样，只是遇到最大值时，模板值会清0。
116 Decr = GL_DECR, // 如果模板值大于0，模板值将会-1。
117 DecrWrap = GL_DECR_WRAP, // 和GL_DECR一样，只是当模板值减到0时，会变为最大值。
118 Invert = GL_INVERT // 将模板值按位取反。
119 };

121 struct StencilDescriptor
122 {
123 StencilFunc func = StencilFunc::Disabled;
124 uint16_t ref; // 指定模板测试的ref值。此值被限制在[0，2n-1]的范围内，其中 n 是模板缓冲区中的位平面数。
125 uint16_t mask; // 指定用于在测试完成时与ref和存储模板值进行与操作的位元掩码。
126 StencilOp fail = StencilOp::Keep; // 模板测试失败后执行的操作
127 StencilOp zFail = StencilOp::Keep; // 模板测试通过但深度测试不通过时候执行的操作。
128 StencilOp zPass = StencilOp::Keep; // 模板测试和深度测试都通过后，或者深度测试未开启时执行的操作。
129 };
