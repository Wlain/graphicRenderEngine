243ParticleEmitter::ParticleEmitter(ParticleEmitter::ParticleEmitterBuilder& build)
244{
245 m_particleProp = build.m_particleProp;
246 auto particleCount = m_particleProp.particleCount;
247 m_particles.resize(particleCount);
248 m_particleProp.material = m_particleProp.material;
249 m_positions.resize(particleCount, { 0.0f, 0.0f, 0.0f });
250 m_colors.resize(particleCount, { 1.0f, 1.0f, 1.0f, 1.0f });
251 m_sizes.resize(particleCount, { 100.0f });
252 m_uvs.resize(particleCount, { 0.0f, 0.0f, 1.0f, 1.0f });
253 // preallocate arrays for mesh data
254 for (int i = 0; i < particleCount; i++)
255 {
256 m_particles[i].velocity = glm::sphericalRand(1.0f);
257 }
258 m_mesh = Mesh::create()
259 .withPositions(m_positions)
260 .withColors(m_colors)
261 .withParticleSizes(m_sizes)
262 .withUvs(m_uvs)
263 .withMeshTopology(Mesh::Topology::Points)
264 .build();
265}

269void ParticleEmitter::update(float deltaTime)
270{
271 if (m_particleProp.emitting && m_particleProp.visible)
272 {
273 int oldEmissions = (int)m_emissions;
274 m_emissions += deltaTime * m_particleProp.emissionRate;
275 int newEmissions = (int)m_emissions;
276 for (int i = oldEmissions; i < newEmissions; ++i)
277 {
278 emitOnce();
279 }
280 }
281 if (m_particleProp.started)
282 {
283 m_totalTime += deltaTime;
284 m_activeParticles = 0;
285 for (auto& p : m_particles)
286 {
287 // v = at
288 p.velocity += deltaTime * p.acceleration;
289 // deltaS = vt
290 p.position += p.velocity * deltaTime;
291 p.color = glm::mix(m_particleProp.colorStart, m_particleProp.colorEnd, p.age);
292 p.size = glm::mix(m_particleProp.sizeStart, m_particleProp.sizeEnd, p.age);
293 p.rotation += p.angularVelocity * deltaTime;
294 p.alive = p.timeOfBirth + m_particleProp.lifeSpan > m_totalTime;
295 if (p.alive)
296 m_activeParticles++;
297 p.age = (m_totalTime - p.timeOfBirth) / m_particleProp.lifeSpan;
298 }
299 }

302void ParticleEmitter::draw(RenderPass& renderPass, glm::mat4 transform)
303{
304 if (!m_particleProp.visible) return;
305 for (int i = 0; i < m_particles.size(); i++)
306 {
307 auto& p = m_particles[i];
308 m_sizes[i] = p.alive ? p.size : 0;
309 m_positions[i] = p.position;
310 m_colors[i] = p.color;
311 m_uvs[i].w = p.rotation;
312 }
313 m_mesh->update()
314 .withPositions(m_positions)
315 .withColors(m_colors)
316 .withParticleSizes(m_sizes)
317 .withUvs(m_uvs)
318 .build();
319 renderPass.draw(m_mesh, transform, m_particleProp.material);
320}

322void ParticleEmitter::emitOnce()
323{
324 auto& p = m_particles[m_index];
325 p.alive = true;
326 p.timeOfBirth = m_totalTime;
327 p.age = 0;
328 p.position = m_particleProp.position;
329 p.colorStart = generateValue(m_particleProp.colorStart, m_particleProp.colorStartVariance);
330 p.colorEnd = generateValue(m_particleProp.colorEnd, m_particleProp.colorEndVariance);
331 p.sizeStart = generateValue(m_particleProp.sizeStart, m_particleProp.sizeStartVariance);
332 p.acceleration = m_particleProp.acceleration;
333 p.velocity = generateValue(m_particleProp.velocity, m_particleProp.velocityVariance);
334 p.rotation = m_particleProp.rotation;
335 p.angularVelocity = m_particleProp.angularVelocity;
336 m_index = (m_index + 1) % m_particles.size();
337}

355T ParticleEmitter::generateValue(const T& base, const T& variance)
356{
357 return base + (variance - base) * MATH_RANDOM_0_1();
