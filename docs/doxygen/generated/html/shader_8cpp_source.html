54// 正则替换，在shader里面实现include的功能
55std::string pragmaInclude(const std::string& source, std::vector<std::string>& errors, uint32_t shaderType)
56{
57 if (source.find("#include") == -1)
58 {
59 return source;
60 }
61 std::stringstream sstream;
62 // \s: 匹配任意的空白符
63 // +: 重复一次或更多次
64 // ^": 不包含"这个字符(否定)
65 // *: 匹配前面的子表达式任意次
66 std::regex e("#include\\s+\"([^\"]*)\"");
67 int lineNumber = 0;
68 std::vector<std::string> lines = split(source, '\n');
69 int includes = 0;
70 for (auto& s : lines)
71 {
72 lineNumber++;
73 std::smatch m;
74 if (std::regex_search(s, m, e))
75 {
76 std::string match = m[1];
77 match.insert(0, "shaders/");
78 auto res = getFileContents(match);
79 if (res.empty())
80 {
81 errors.push_back(std::string("0:") + std::to_string(lineNumber) + " cannot find include file " + match + "##" + std::to_string(shaderType));
82 sstream << s << "\n";
83 }
84 else
85 {
86 includes++;
87 sstream << "#line " << (includes * 10000 + 1) << "\n"
88 << res << "\n"
89 << "#line " << lineNumber << "\n";
90 }
91 }
92 else
93 {
94 sstream << s << "\n";
95 }
96 }
97 return sstream.str();
98}

100// 获取当前shader编译信息
101void logCurrentCompileInfo(GLuint& shader, GLenum type, std::vector<std::string>& errors, std::string_view name)
102{
103 GLint logSize = 0;
104 glGetShaderiv(shader, GL_INFO_LOG_LENGTH, &logSize);
105 if (logSize > 0)
106 {
107 std::vector<char> errorLog((unsigned long)logSize);
108 glGetShaderInfoLog(shader, logSize, &logSize, errorLog.data());
109 std::string typeStr;
110 switch (type)
111 {
112 case GL_FRAGMENT_SHADER:
113 typeStr = "Fragment shader";
114 break;
115 case GL_VERTEX_SHADER:
116 typeStr = "Vertex shader";
117 break;
118 case GL_GEOMETRY_SHADER:
119 typeStr = "Geometry shader";
120 break;
121 case GL_TESS_CONTROL_SHADER:
122 typeStr = "Tessellation control shader";
123 break;
124 case GL_TESS_EVALUATION_SHADER:
125 typeStr = "Tessellation eval shader";
126 break;
127 default:
128 typeStr = std::string("Unknown error type: ") + std::to_string(type);
129 break;
130 }
131 LOG_ERROR("Shader compile error in {} {}: {}", name.data(), typeStr.c_str(), errorLog.data());
132 errors.push_back(std::string(errorLog.data()) + "##" + std::to_string(type));
133 }

136bool linkProgram(GLuint shaderProgram, std::vector<std::string>& errors)
137{
138 glBindFragDataLocation(shaderProgram, 0, "fragColor");
139 glLinkProgram(shaderProgram);
140 GLint linked;
141 glGetProgramiv(shaderProgram, GL_LINK_STATUS, &linked);
142 if (linked == GL_FALSE)
143 {
144 GLint logSize;
145 glGetProgramiv(shaderProgram, GL_INFO_LOG_LENGTH, &logSize);
146 if (logSize > 1)
147 { // log size of 1 is empty, since it includes \0
148 std::vector<char> errorLog((size_t)logSize);
149 glGetProgramInfoLog(shaderProgram, logSize, nullptr, errorLog.data());
150 errors.emplace_back(errorLog.data());
151 LOG_ERROR("{}", errorLog.data());
152 }
153 return false;
154 }
155 return true;
156}

233std::shared_ptr<Shader> Shader::ShaderBuilder::build(std::vector<std::string>& errors)
234{
235 std::shared_ptr<Shader> shader;
236 if (updateShader)
237 {
238 shader = updateShader->shared_from_this();
239 }
240 else
241 {
242 if (m_name.empty())
243 {
244 m_name = "Unnamed shader";
245 }
246 shader = std::shared_ptr<Shader>(new Shader());
247 shader->m_specializationConstants = m_specializationConstants;
248 }
249 bool compileSuccess = shader->build(m_shaderSources, errors);
250 if (!compileSuccess)
251 {
252 if (!updateShader)
253 {
254 shader.reset();
255 }
256 return shader;
257 }
258 shader->m_cullFace = m_cullFace;
259 shader->m_depthTest = m_depthTest;
260 shader->m_depthWrite = m_depthWrite;
261 shader->m_blendType = m_blendType;
262 shader->m_name = m_name;
263 shader->m_offset = m_offset;
264 shader->m_shaderSources = m_shaderSources;
265 shader->m_shaderUniqueId = ++s_globalShaderCounter;
266 shader->m_stencil = m_stencil;
267 shader->m_colorWrite = m_colorWrite;
268 return shader;
269}

286bool Shader::compileShader(const Shader::Resource& resource, GLenum type, GLuint& shader, std::vector<std::string>& errors)
287{
288 auto source = getSource(resource);
289 std::string source_ = precompile(source, errors, type);
290 shader = glCreateShader(type);
291 const auto* stringPtr = source_.c_str();
292 auto length = (GLint)source_.size();
293 glShaderSource(shader, 1, &stringPtr, &length);
294 glCompileShader(shader);
295 GLint success = 0;
296 glGetShaderiv(shader, GL_COMPILE_STATUS, &success);
297 logCurrentCompileInfo(shader, type, errors, resource.value);
298
299 return success == 1;
300}

384bool Shader::setLights(WorldLights* worldLights) const
385{
386 int maxSceneLights = Renderer::s_instance->getMaxSceneLights();
387 if (worldLights == nullptr)
388 {
389 glUniform4f(m_uniformLocationAmbientLight, 0, 0, 0, 0);
390 std::vector<float> noLight(maxSceneLights * 4, 0);
391 glUniform4fv(m_uniformLocationLightPosType, maxSceneLights, noLight.data());
392 glUniform4fv(m_uniformLocationLightColorRange, maxSceneLights, noLight.data());
393 return false;
394 }
395 if (m_uniformLocationAmbientLight != -1)
396 {
397 glUniform4fv(m_uniformLocationAmbientLight, 1, glm::value_ptr(worldLights->ambientLight));
398 }
399 if (m_uniformLocationLightPosType != -1 && m_uniformLocationLightColorRange != -1)
400 {
401 std::vector<glm::vec4> lightPosType(maxSceneLights, glm::vec4{ 0 });
402 std::vector<glm::vec4> lightColorRange(maxSceneLights, glm::vec4{ 0 });
403 for (int i = 0; i < maxSceneLights; i++)
404 {
405 auto light = worldLights->getLight(i);
406 if (light == nullptr || light->type == Light::Type::Unused)
407 {
408 lightPosType[i] = glm::vec4(0.0f, 0.0f, 0.0f, 2);
409 continue;
410 }
411 else if (light->type == Light::Type::Directional)
412 {
413 lightPosType[i] = glm::vec4(glm::normalize(light->direction), 0);
414 }
415 else if (light->type == Light::Type::Point)
416 {
417 lightPosType[i] = glm::vec4(light->position, 1);
418 }
419 // transform to eye space
420 lightPosType[i] = lightPosType[i];
421 lightColorRange[i] = glm::vec4(light->color, light->range);
422 }
423 if (m_uniformLocationLightPosType != -1)
424 {
425 glUniform4fv(m_uniformLocationLightPosType, maxSceneLights, glm::value_ptr(lightPosType[0]));
426 }
427 if (m_uniformLocationLightColorRange != -1)
428 {
429 glUniform4fv(m_uniformLocationLightColorRange, maxSceneLights, glm::value_ptr(lightColorRange[0]));
430 }
431 }
432 return true;
433}

435void Shader::bind()
436{
437 glUseProgram(m_id);
438 if (m_depthWrite)
439 {
440 glEnable(GL_DEPTH_TEST);
441 }
442 else
443 {
444 glDisable(GL_DEPTH_TEST);
445 }
446 if (m_stencil.func == StencilFunc::Disabled)
447 {
448 glDisable(GL_STENCIL_TEST);
449 glStencilMask(0);
450 }
451 else
452 {
453 glEnable(GL_STENCIL_TEST);
454 glStencilFunc(static_cast<GLenum>(m_stencil.func), (GLint)m_stencil.ref, (GLint)m_stencil.mask);
455 glStencilOp(static_cast<GLenum>(m_stencil.fail), static_cast<GLenum>(m_stencil.zFail), static_cast<GLenum>(m_stencil.zPass));
456 glStencilMask(0xFFFF);
457 }
458 if (m_cullFace == CullFace::None)
459 {
460 glDisable(GL_CULL_FACE);
461 }
462 else
463 {
464 glEnable(GL_CULL_FACE);
465 glCullFace(m_cullFace == CullFace::Back ? GL_BACK : GL_FRONT);
466 }
467 glDepthMask(m_depthWrite ? GL_TRUE : GL_FALSE);
468 glColorMask(m_colorWrite.r, m_colorWrite.g, m_colorWrite.b, m_colorWrite.a);
469 switch (m_blendType)
470 {
471 case BlendType::Disabled:
472 glDisable(GL_BLEND);
473 break;
474 case BlendType::AlphaBlending:
475 glEnable(GL_BLEND);
476 glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
477 break;
478 case BlendType::AdditiveBlending:
479 glEnable(GL_BLEND);
480 glBlendFunc(GL_SRC_ALPHA, GL_ONE);
481 break;
482 default:
483 LOG_ERROR("Invalid blend value - was {}", (int)m_blendType);
484 break;
485 }
486 // 第一个参数代表：即将绘制的片元因子，也就是源因子
487 // 第二个参数代表：framebuffer里面存储的颜色，也就是目标因子
488 // glEnable(GL_BLEND);
489 // glBlendFunc(GL_ZERO, GL_ONE); // 此时，代表全部用framebuffer里存储的颜色,反之同理
490
491 if (m_offset.x == 0 && m_offset.y == 0)
492 {
493 glDisable(GL_POLYGON_OFFSET_FILL);
494 glDisable(GL_POLYGON_OFFSET_LINE);
495 glDisable(GL_POLYGON_OFFSET_POINT);
496 }
497 else
498 {
499 glEnable(GL_POLYGON_OFFSET_FILL);
500 glEnable(GL_POLYGON_OFFSET_LINE);
501 glEnable(GL_POLYGON_OFFSET_POINT);
502 glPolygonOffset(m_offset.x, m_offset.y);
503 }
504}

516void Shader::updateUniformsAndAttributes()
517{
518 // update uniforms
519 m_uniformLocationModel = -1;
520 m_uniformLocationView = -1;
521 m_uniformLocationProjection = -1;
522 m_uniformLocationModelViewInverseTranspose = -1;
523 m_uniformLocationModelInverseTranspose = -1;
524 m_uniformLocationViewport = -1;
525 m_uniformLocationAmbientLight = -1;
526 m_uniformLocationLightPosType = -1;
527 m_uniformLocationLightColorRange = -1;
528 m_uniformLocationCameraPosition = -1;
529 m_uniforms.clear();
530 GLint uniformCount;
531 glGetProgramiv(m_id, GL_ACTIVE_UNIFORMS, &uniformCount);
532 UniformType uniformType = UniformType::Invalid;
533 for (int i = 0; i < uniformCount; i++)
534 {
535 const int nameSize = 50;
536 GLchar name[nameSize];
537 GLsizei nameLength;
538 GLint size;
539 GLenum type;
540 glGetActiveUniform(m_id, i, nameSize, &nameLength, &size, &type, name);
541 switch (type)
542 {
543 case GL_FLOAT:
544 uniformType = UniformType::Float;
545 break;
546 case GL_FLOAT_VEC4:
547 uniformType = UniformType::Vec4;
548 break;
549 case GL_INT_VEC4:
550 uniformType = UniformType::IVec4;
551 break;
552 case GL_FLOAT_VEC3:
553 uniformType = UniformType::Vec3;
554 break;
555 case GL_INT:
556 uniformType = UniformType::Int;
557 break;
558 case GL_FLOAT_MAT3:
559 uniformType = UniformType::Mat3;
560 break;
561 case GL_FLOAT_MAT4:
562 uniformType = UniformType::Mat4;
563 break;
564 case GL_SAMPLER_2D:
565 uniformType = UniformType::Texture;
566 break;
567 case GL_SAMPLER_CUBE:
568 uniformType = UniformType::TextureCube;
569 break;
570 default:
571 LOG_ERROR("Unsupported shader type {}, name {}", type, name);
572 }
573 // remove [0] if exists
574 char* bracketIndex = strchr(name, '[');
575 if (bracketIndex != nullptr)
576 {
577 *bracketIndex = '\0';
578 }
579 bool isGlobalUniform = name[0] == 'g' && name[1] == '_';
580 GLint location = glGetUniformLocation(m_id, name);
581 if (!isGlobalUniform)
582 {
583 m_uniforms.emplace_back(name, location, uniformType, size);
584 }
585 else
586 {
587 if (strcmp(name, "g_model") == 0)
588 {
589 if (uniformType == UniformType::Mat4)
590 {
591 m_uniformLocationModel = location;
592 }
593 else
594 {
595 LOG_ERROR("Invalid g_model uniform type. Expected mat4.was {}", toStr(uniformType));
596 }
597 }
598 if (strcmp(name, "g_view") == 0)
599 {
600 if (uniformType == UniformType::Mat4)
601 {
602 m_uniformLocationView = location;
603 }
604 else
605 {
606 LOG_ERROR("Invalid g_view uniform type. Expected mat4.was {}", toStr(uniformType));
607 }
608 }
609 if (strcmp(name, "g_projection") == 0)
610 {
611 if (uniformType == UniformType::Mat4)
612 {
613 m_uniformLocationProjection = location;
614 }
615 else
616 {
617 LOG_ERROR("Invalid g_projection uniform type. Expected mat4.was {}", toStr(uniformType));
618 }
619 }
620 if (strcmp(name, "g_model_it") == 0)
621 {
622 if (uniformType == UniformType::Mat3)
623 {
624 m_uniformLocationModelInverseTranspose = location;
625 }
626 else
627 {
628 LOG_ERROR("Invalid g_model_it uniform type. Expected mat3 - was {}.", toStr(uniformType));
629 }
630 }
631 if (strcmp(name, "g_model_view_it") == 0)
632 {
633 if (uniformType == UniformType::Mat3)
634 {
635 m_uniformLocationModelViewInverseTranspose = location;
636 }
637 else
638 {
639 LOG_ERROR("Invalid g_model_view_it uniform type. Expected mat3.was {}", toStr(uniformType));
640 }
641 }
642 if (strcmp(name, "g_viewport") == 0)
643 {
644 if (uniformType == UniformType::Vec4)
645 {
646 m_uniformLocationViewport = location;
647 }
648 else
649 {
650 LOG_ERROR("Invalid g_viewport uniform type. Expected vec4.was {}", toStr(uniformType));
651 }
652 }
653 if (strcmp(name, "g_ambientLight") == 0)
654 {
655 if (uniformType == UniformType::Vec4)
656 {
657 m_uniformLocationAmbientLight = location;
658 }
659 else
660 {
661 LOG_ERROR("Invalid g_ambientLight uniform type. Expected vec4.was {}", toStr(uniformType));
662 }
663 }
664 if (strcmp(name, "g_lightPosType") == 0)
665 {
666 if (uniformType == UniformType::Vec4 && size == Renderer::s_instance->getMaxSceneLights())
667 {
668 m_uniformLocationLightPosType = location;
669 }
670 else
671 {
672 LOG_ERROR("Invalid g_lightPosType uniform type. Expected vec4[Renderer::s_maxSceneLights].was {}", toStr(uniformType));
673 }
674 }
675 if (strcmp(name, "g_lightColorRange") == 0)
676 {
677 if (uniformType == UniformType::Vec4 && size == Renderer::s_instance->getMaxSceneLights())
678 {
679 m_uniformLocationLightColorRange = location;
680 }
681 else
682 {
683 LOG_ERROR("Invalid g_lightPosType uniform type. Expected vec4[Renderer::s_maxSceneLights].was {}", toStr(uniformType));
684 }
685 }
686 if (strcmp(name, "g_cameraPos") == 0)
687 {
688 if (uniformType == UniformType::Vec4)
689 {
690 m_uniformLocationCameraPosition = location;
691 }
692 else
693 {
694 LOG_ERROR("Invalid g_cameraPos uniform type. Expected vec4 - was {}[{}].", toStr(uniformType), size);
695 }
696 }
697 }
698 }
699
700 // update attributes
701 m_attributes.clear();
702 int attributeCount;
703 glGetProgramiv(m_id, GL_ACTIVE_ATTRIBUTES, &attributeCount);
704
705 for (int i = 0; i < attributeCount; ++i)
706 {
707 const int nameSize = 50;
708 GLchar name[nameSize];
709 GLsizei nameLength;
710 GLint size;
711 GLenum type;
712 glGetActiveAttrib(m_id, i, nameSize, &nameLength, &size, &type, name);
713 auto location = glGetAttribLocation(m_id, name);
714 m_attributes[std::string(name)] = { location, type, size };
715 }
716}

723bool Shader::build(const std::map<ShaderType, Resource>& shaderSources, std::vector<std::string>& errors)
724{
725 auto oldShaderProgramId = m_id;
726 m_id = glCreateProgram();
727 assert(m_id != 0);
728 std::vector<GLuint> shaders;
729 auto cleanupShaders = [&]() {
730 for (auto id : shaders)
731 {
732 glDeleteShader(id);
733 }
734 };
735 for (const auto& shaderSource : shaderSources)
736 {
737 GLuint s;
738 GLenum shader = toId(shaderSource.first);
739 auto ret = compileShader(shaderSource.second, shader, s, errors);
740 if (!ret)
741 {
742 cleanupShaders();
743 glDeleteProgram(m_id);
744 m_id = oldShaderProgramId;
745 for (const auto& e : errors)
746 {
747 LOG_ERROR("{}", e.c_str());
748 }
749 return false;
750 }
751 else
752 {
753 shaders.push_back(s);
754 }
755 glAttachShader(m_id, s);
756 }
757 bool linked = linkProgram(m_id, errors);
758 cleanupShaders();
759 if (!linked)
760 {
761 glDeleteProgram(m_id);
762 m_id = oldShaderProgramId;
763 for (const auto& e : errors)
764 {
765 LOG_ERROR("{}", e.c_str());
766 }
767 return false;
768 }
769 if (oldShaderProgramId != 0)
770 {
771 glDeleteProgram(oldShaderProgramId); // delete old shader if any
772 }
773 // setup global uniform
774 if (Renderer::s_instance->m_globalUniformBuffer)
775 {
776 glUseProgram(m_id);
777 auto index = glGetUniformBlockIndex(m_id, "g_global_uniforms");
778 if (index != GL_INVALID_VALUE)
779 {
780 constexpr int globalUniformBindingIndex = 1;
781 glUniformBlockBinding(m_id, index, globalUniformBindingIndex);
782 glBindBufferRange(GL_UNIFORM_BUFFER, globalUniformBindingIndex, Renderer::s_instance->m_globalUniformBuffer, 0, Renderer::s_instance->m_globalUniformBufferSize);
783 }
784 }
785 updateUniformsAndAttributes();
786 return true;
787}

849{

869 res.m_name = m_name;
870 res.m_offset = m_offset;
871 res.m_shaderSources = m_shaderSources;
872 res.m_specializationConstants = std::move(specializationConstants);
873 auto specializedShader = res.build();
874 if (specializedShader == nullptr)
875 {
876 LOG_WARN("Cannot create specialized shader. Using shader without specialization.");
877 return std::make_shared<Material>(shared_from_this());
878 }
879 specializedShader->m_parent = shared_from_this();
880 m_specializations.push_back(std::weak_ptr<Shader>(specializedShader));
881 return std::make_shared<Material>(specializedShader);
882 }
883 return std::make_shared<Material>(shared_from_this());
884}

1038std::string Shader::insertPreprocessorDefines(std::string source, std::map<std::string, std::string>& specializationConstants, uint32_t shaderType)
1039{
1040 LOG_INFO("before insertPreprocessorDefines:source is:\n {}", source.c_str());
1041 std::stringstream ss;
1042 ss << "#define SI_LIGHTS " << Renderer::s_instance->getMaxSceneLights() << "\n";
1043 // add shader type
1044 switch (shaderType)
1045 {
1046 case GL_FRAGMENT_SHADER:
1047 ss << "#define SI_FRAGMENT 1\n";
1048 break;
1049 case GL_VERTEX_SHADER:
1050 ss << "#define SI_VERTEX 1\n";
1051 break;
1052 case GL_GEOMETRY_SHADER:
1053 ss << "#define SI_GEOMETRY 1\n";
1054 break;
1055 case GL_TESS_CONTROL_SHADER:
1056 ss << "#define SI_TESS_CTRL 1\n";
1057 break;
1058 case GL_TESS_EVALUATION_SHADER:
1059 ss << "#define SI_TESS_EVAL 1\n";
1060 break;
1061 default:
1062 LOG_WARN("Unknown shader type");
1063 break;
1064 }
1065 for (auto& sc : specializationConstants)
1066 {
1067 ss << "#define " << sc.first << " " << sc.second << "\n";
1068 }
1069 // framebuffers is sRGB.
1070 if (Renderer::s_instance->renderInfo().useFramebufferSRGB)
1071 {
1072 ss << "#define SI_FRAMEBUFFER_SRGB 1\n";
1073 }
1074 // texture sampler supports sRGB.
1075 if (Renderer::s_instance->renderInfo().supportTextureSamplerSRGB)
1076 {
1077 ss << "#define SI_TEX_SAMPLER_SRGB 1\n";
1078 }
1079 // frame buffer support depth texture attachments.
1080 if (Renderer::s_instance->renderInfo().supportFBODepthAttachment)
1081 {
1082 ss << "#define SI_FBO_DEPTH_ATTACHMENT 1\n";
1083 }
1084 // If the shader contains any #version or #extension statements, the defines are added after them.
1085 auto version = static_cast<int>(source.rfind("#version"));
1086 auto extension = static_cast<int>(source.rfind("#extension"));
1087 auto last = std::max(version, extension);
1088 if (last == -1)
1089 {
1090 ss << "#line 1\n";
1091 return ss.str() + source;
1092 }
1093 auto insertPos = source.find('\n', last);
1094 int lines = 0;
1095 for (int i = 0; i < insertPos; i++)
1096 {
1097 if (source.at(i) == '\n')
1098 {
1099 lines++;
1100 }
1101 }
1102 ss << "#line " << (lines + 1) << "\n";
1103 LOG_INFO("after insertPreprocessorDefines:source is:\n {}", source.substr(0, insertPos + 1) + ss.str() + source.substr(insertPos + 1));
1104 return source.substr(0, insertPos + 1) + ss.str() + source.substr(insertPos + 1);
1105}
