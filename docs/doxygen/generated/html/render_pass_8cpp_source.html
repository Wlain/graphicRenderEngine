223void RenderPass::setupShader(const glm::mat4& modelTransform, Shader* shader)
224{
225 if (m_lastBoundShader != shader)
226 {
227 m_builder.m_renderStats->stateChangesShader++;
228 m_lastBoundShader = shader;
229 shader->bind();
230 }
231 // model
232 if (shader->m_uniformLocationModel != -1)
233 {
234 glUniformMatrix4fv(shader->m_uniformLocationModel, 1, GL_FALSE, glm::value_ptr(modelTransform));
235 }
236 // normal
237 if (shader->m_uniformLocationModelViewInverseTranspose != -1)
238 {
239 auto normalMatrix = transpose(inverse((glm::mat3)(m_builder.m_camera.getViewTransform() * modelTransform)));
240 glUniformMatrix3fv(shader->m_uniformLocationModelViewInverseTranspose, 1, GL_FALSE, glm::value_ptr(normalMatrix));
241 }
242 if (shader->m_uniformLocationModelInverseTranspose != -1)
243 {
244 auto normalMatrix = transpose(inverse((glm::mat3)modelTransform));
245 glUniformMatrix3fv(shader->m_uniformLocationModelInverseTranspose, 1, GL_FALSE, glm::value_ptr(normalMatrix));
246 }

249void RenderPass::finish()
250{
251 if (m_isFinished) return;
252 if (m_builder.m_framebuffer != nullptr)
253 {
254 m_builder.m_framebuffer->bind();
255 }
256 else
257 {
258 glBindFramebuffer(GL_FRAMEBUFFER, 0);
259 }
260 auto framebufferSize = (m_builder.m_framebuffer != nullptr) ? m_builder.m_framebuffer->m_size : static_cast<glm::vec2>(Renderer::s_instance->getFramebufferSize());
261 m_viewportOffset = static_cast<glm::uvec2>(m_builder.m_camera.m_viewportOffset * framebufferSize);
262 m_viewportSize = static_cast<glm::uvec2>(framebufferSize * m_builder.m_camera.m_viewportSize);
263 glEnable(GL_SCISSOR_TEST);
264 glScissor(m_viewportOffset.x, m_viewportOffset.y, m_viewportSize.x, m_viewportSize.y);
265 glViewport(m_viewportOffset.x, m_viewportOffset.y, m_viewportSize.x, m_viewportSize.y);
266 GLbitfield clear = 0;
267 if (m_builder.m_clearColor)
268 {
269 glClearColor(m_builder.m_clearColorValue.r, m_builder.m_clearColorValue.g, m_builder.m_clearColorValue.b, m_builder.m_clearColorValue.a);
270 clear |= GL_COLOR_BUFFER_BIT;
271 glColorMask(GL_TRUE, GL_TRUE, GL_TRUE, GL_TRUE);
272 }
273 if (m_builder.m_clearDepth)
274 {
275 glClearDepthf(m_builder.m_clearDepthValue);
276 glClearDepth(m_builder.m_clearDepthValue);
277 clear |= GL_DEPTH_BUFFER_BIT;
278 glDepthMask(GL_TRUE);
279 }
280 if (m_builder.m_clearStencil)
281 {
282 glClearStencil(m_builder.m_clearStencilValue);
283 clear |= GL_STENCIL_BUFFER_BIT;
284 glStencilMask(0xFFFF);
285 }
286 // 0u:表示无符号整型:0
287 if (clear != 0u)
288 {
289 glClear(clear);
290 }
291 m_projection = m_builder.m_camera.getProjectionTransform(m_viewportSize);
292 if (m_builder.m_skybox)
293 {
294 // create an infinite projection
295 glm::mat4 infMatrix = m_builder.m_camera.getInfiniteProjectionTransform(m_viewportSize);
296 // passing the inf projection as the model matrix
297 m_renderQueue[0] = { m_builder.m_skybox->m_skyboxMesh, infMatrix, m_builder.m_skybox->m_material };
298 }
299 setupGlobalShaderUniforms();
300 for (auto& rqObj : m_renderQueue)
301 {
302 drawInstance(rqObj);
303 }
304 if (m_builder.m_gui)
305 {
306 ImGui::Render();
307 ImGui_ImplOpenGL3_RenderDrawData(ImGui::GetDrawData());
308 }
309 glBindFramebuffer(GL_FRAMEBUFFER, 0);
310 if (m_builder.m_framebuffer != nullptr)
311 {
312 for (auto& tex : m_builder.m_framebuffer->m_textures)
313 {
314 if (tex->m_info.generateMipmap)
315 {
316 glBindTexture(tex->m_info.target, tex->m_info.id);
317 glGenerateMipmap(tex->m_info.target);
318 glBindTexture(tex->m_info.target, 0);
319 }
320 }
321 }
322 m_isFinished = true;
323#ifndef NDEBUG
324 checkGlError();
325#endif
326}

328std::vector<Color> RenderPass::readPixels(unsigned int x, unsigned int y, unsigned int width, unsigned int height)
329{
330 ASSERT(m_isFinished);
331 finish();
332 if (m_builder.m_framebuffer != nullptr)
333 {
334 m_builder.m_framebuffer->bind();
335 }
336 std::vector<Color> res(width * height);
337 std::vector<glm::u8vec4> resUnsigned(width * height);
338 glReadPixels(x, y, width, height, GL_RGBA, GL_UNSIGNED_BYTE, resUnsigned.data());
339 for (int i = 0; i < resUnsigned.size(); i++)
340 {
341 for (int j = 0; j < 4; j++)
342 {
343 res[i][j] = resUnsigned[i][j] / 255.0f;
344 }
345 }
346 // set default framebuffer
347 if (m_builder.m_framebuffer != nullptr)
348 {
349 glBindFramebuffer(GL_FRAMEBUFFER, 0);
350 }
351 return res;
352}

364void RenderPass::drawInstance(RenderPass::RenderQueueObj& rqObj)
365{
366 auto materialPtr = rqObj.material;
367 auto* mesh = rqObj.mesh.get();
368 auto* material = materialPtr.get();
369 auto* shader = material->getShader().get();
370 assert(mesh != nullptr);
371 m_builder.m_renderStats->drawCalls++;
372 setupShader(rqObj.modelTransform, shader);
373 if (material != m_lastBoundMaterial)
374 {
375 m_builder.m_renderStats->stateChangesMaterial++;
376 m_lastBoundMaterial = material;
377 m_lastBoundMeshId = -1; // force mesh to rebind
378 material->bind();
379 }
380 if (mesh->m_meshId != m_lastBoundMeshId)
381 {
382 m_builder.m_renderStats->stateChangesMesh++;
383 m_lastBoundMeshId = mesh->m_meshId;
384 }
385 mesh->bind(shader);
386 if (mesh->getIndexSets() == 0)
387 {
388 glDrawArrays((GLenum)mesh->getMeshTopology(), 0, mesh->getVertexCount());
389 }
390 else
391 {
392 auto offsetCount = mesh->m_elementBufferOffsetCount[rqObj.subMesh];
393 GLsizei indexCount = offsetCount.second;
394 glDrawElements((GLenum)mesh->getMeshTopology(), indexCount, GL_UNSIGNED_SHORT, (char*)nullptr + offsetCount.first);
395 }
396}

398void RenderPass::setupShaderRenderPass(Shader* shader)
399{
400 if (shader->m_uniformLocationView != -1)
401 {
402 glUniformMatrix4fv(shader->m_uniformLocationModel, 1, GL_FALSE, glm::value_ptr(m_builder.m_camera.m_viewTransform));
403 }
404 if (shader->m_uniformLocationProjection != -1)
405 {
406 glUniformMatrix4fv(shader->m_uniformLocationProjection, 1, GL_FALSE, glm::value_ptr(m_projection));
407 }
408 if (shader->m_uniformLocationCameraPosition != -1)
409 {
410 glUniformMatrix4fv(shader->m_uniformLocationCameraPosition, 1, GL_FALSE, glm::value_ptr(glm::vec4{ m_builder.m_camera.getPosition(), 1.0f }));
411 }
412 shader->setLights(m_builder.m_worldLights);
413}

415void RenderPass::setupShaderRenderPass(const RenderPass::GlobalUniforms& globalUniforms)
416{
417 *globalUniforms.g_view = m_builder.m_camera.m_viewTransform;
418 *globalUniforms.g_projection = m_projection;
419 *globalUniforms.g_viewport = glm::vec4(m_viewportOffset.x, m_viewportOffset.y, m_viewportSize.x, m_viewportSize.y);
420 *globalUniforms.g_cameraPos = glm::vec4(m_builder.m_camera.getPosition(), 1.0f);
421 int maxSceneLights = Renderer::s_instance->m_maxSceneLights;
422 size_t lightSize = sizeof(glm::vec4) * (1 + maxSceneLights * 2); // ambient + (lightPosType + lightColorRange) * maxSceneLights
423 memset(globalUniforms.g_ambientLight, 0, lightSize);
424 if (m_builder.m_worldLights)
425 {
426 *globalUniforms.g_ambientLight = glm::vec4(m_builder.m_worldLights->getAmbientLight(), 1.0);
427 for (int i = 0; i < maxSceneLights; i++)
428 {
429 auto light = m_builder.m_worldLights->getLight(i);
430 if (light == nullptr || light->type == Light::Type::Unused)
431 {
432 globalUniforms.g_lightPosType[i] = glm::vec4(0.0f, 0.0f, 0.0f, 2);
433 continue;
434 }
435 else if (light->type == Light::Type::Point)
436 {
437 globalUniforms.g_lightPosType[i] = glm::vec4(light->position, 1);
438 }
439 else if (light->type == Light::Type::Directional)
440 {
441 globalUniforms.g_lightPosType[i] = glm::vec4(glm::normalize(light->direction), 0);
442 }
443 // transform to eye space
444 globalUniforms.g_lightColorRange[i] = glm::vec4(light->color, light->range);
445 }
446 }
447 glBindBuffer(GL_UNIFORM_BUFFER, Renderer::s_instance->m_globalUniformBuffer);
448 glBufferData(GL_UNIFORM_BUFFER, Renderer::s_instance->m_globalUniformBufferSize, globalUniforms.g_view, GL_STREAM_DRAW);
449 glBindBuffer(GL_UNIFORM_BUFFER, 0);
450}

452void RenderPass::setupGlobalShaderUniforms()
453{
454 static std::vector<char> buffer;
455 static GlobalUniforms globalUniforms;
456 static bool once = [&]() {
457 // allocate buffer and setup pointers into buffer
458 buffer.resize(Renderer::s_instance->m_globalUniformBufferSize, 0);
459 globalUniforms.g_view = reinterpret_cast<glm::mat4*>(buffer.data());
460 globalUniforms.g_projection = reinterpret_cast<glm::mat4*>(buffer.data() + sizeof(glm::mat4));
461 globalUniforms.g_viewport = reinterpret_cast<glm::vec4*>(buffer.data() + sizeof(glm::mat4) * 2);
462 globalUniforms.g_cameraPos = reinterpret_cast<glm::vec4*>(buffer.data() + sizeof(glm::mat4) * 2 + sizeof(glm::vec4));
463 globalUniforms.g_ambientLight = reinterpret_cast<glm::vec4*>(buffer.data() + sizeof(glm::mat4) * 2 + sizeof(glm::vec4) * 2);
464 int lightColorRangeOffset = sizeof(glm::mat4) * 2 + sizeof(glm::vec4) * 3;
465 globalUniforms.g_lightColorRange = reinterpret_cast<glm::vec4*>(buffer.data() + lightColorRangeOffset);
466 int g_lightPosTypeOffset = lightColorRangeOffset + (int)sizeof(glm::vec4) * (Renderer::s_instance->m_maxSceneLights);
467 globalUniforms.g_lightPosType = reinterpret_cast<glm::vec4*>(buffer.data() + g_lightPosTypeOffset);
468 return true;
469 }();
470 if (Renderer::s_instance->m_globalUniformBuffer)
471 {
472 setupShaderRenderPass(globalUniforms);
473 }
474 else
475 {
476 // find list of used shaders
477 m_shaders.clear();
478 for (auto& rqObj : m_renderQueue)
479 {
480 m_shaders.insert(rqObj.material->getShader().get());
481 }
482 // update global uniforms
483 for (auto shader : m_shaders)
484 {
485 glUseProgram(shader->m_id);
486 setupShaderRenderPass(shader);
487 }
488 }
