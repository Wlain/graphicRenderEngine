128std::array<glm::vec2, 4> Sprite::getTrimmedCorners() const
129{
130 float x0 = 0 - m_spriteAnchor.x * (float)m_spriteSize.x;
131 float x1 = (float)m_spriteSize.x - m_spriteAnchor.x * (float)m_spriteSize.x;
132 float y0 = 0 - m_spriteAnchor.y * (float)m_spriteSize.y;
133 float y1 = (float)m_spriteSize.y - m_spriteAnchor.y * (float)m_spriteSize.y;
134 std::array<glm::vec2, 4> res{};
135 res[0] = { x1, y0 };
136 res[1] = { x1, y1 };
137 res[2] = { x0, y1 };
138 res[3] = { x0, y0 };
139 // compute transformation
140 glm::mat4 matrix = glm::translate(glm::vec3{ m_position, 0.0 }) * glm::rotate(glm::radians(m_rotation), glm::vec3{ 0, 0, 1 }) * glm::scale(glm::vec3{ m_scale, 1.0 });
141 for (int i = 0; i < 4; i++)
142 {
143 res[i] = (glm::vec2)(matrix * glm::vec4(res[i], 0.0, 1.0));
144 }
145 return res;
146}

148std::array<glm::vec2, 4> Sprite::getUvs()
149{
150 std::array<glm::vec2, 4> res{};
151 auto texWidth = (float)m_texture->width();
152 auto texHeight = (float)m_texture->height();
153 float x0 = (float)(m_spritePos.x) / texWidth;
154 float x1 = (float)(m_spritePos.x + m_spriteSize.x) / texWidth;
155 float y0 = (float)(m_spritePos.y) / texHeight;
156 float y1 = (float)(m_spritePos.y + m_spriteSize.y) / texHeight;
157 if (m_flip.x)
158 {
159 std::swap(x0, x1);
160 }
161 if (m_flip.y)
162 {
163 std::swap(y0, y1);
164 }
165 res[0] = { x1, y0 };
166 res[1] = { x1, y1 };
167 res[2] = { x0, y1 };
168 res[3] = { x0, y0 };
169 return res;
170}
