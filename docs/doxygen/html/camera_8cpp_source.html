70glm::mat4 Camera::getProjectionTransform(const glm::uvec2& viewportSize)
71{
72 switch (m_projectionType)
73 {
74 case ProjectionType::Custom:
75 return glm::make_mat4(m_projectionValue.customProjectionMatrix);
76 case ProjectionType::Orthographic: {
77 float aspect = (float)viewportSize.x / (float)viewportSize.y;
78 float sizeX = aspect * m_projectionValue.orthographic.orthographicSize;
79 return glm::ortho<float>(-sizeX, sizeX, -m_projectionValue.orthographic.orthographicSize, m_projectionValue.orthographic.orthographicSize, m_projectionValue.orthographic.nearPlane, m_projectionValue.orthographic.farPlane);
80 }
81 case ProjectionType::OrthographicWindow:
82 return glm::ortho<float>(0, (float)viewportSize.x, 0, (float)viewportSize.y, 1.0f, -1.0f);
83 case ProjectionType::Perspective:
84 return glm::perspectiveFov<float>(m_projectionValue.perspective.fieldOfViewY,
85 (float)viewportSize.x,
86 (float)viewportSize.y,
87 m_projectionValue.perspective.nearPlane,
88 m_projectionValue.perspective.farPlane);
89 default:
90 return glm::mat4(1);
91 }
92}

134Ray Camera::screenPointToRay(glm::vec2 position)
135{
136 glm::vec2 viewportSize = (glm::vec2)Renderer::s_instance->getFramebufferSize() * m_viewportSize;
137 // 将屏幕坐标归一化[0, 1]
138 position = (position / viewportSize) * 2.0f - glm::vec2(1.0f);
139 // 获取pv矩阵
140 auto viewProjection = getProjectionTransform(viewportSize) * m_viewTransform;
141 auto invViewProjection = glm::inverse(viewProjection);
142 // 射线在投影空间的开始和结尾,其中0.1是near，1.0是far
143 // 原始裁剪坐标
144 glm::vec4 originClipSpace{ position, 1.0f, 1 };
145 // 目标裁剪坐标
146 glm::vec4 destClipSpace{ position, 0.1f, 1 };
147 glm::vec4 originClipSpaceWS = invViewProjection * originClipSpace;
148 glm::vec4 destClipSpaceWS = invViewProjection * destClipSpace;
149 auto originClipSpaceWS3 = glm::vec3(originClipSpaceWS);
150 auto destClipSpaceWS3 = glm::vec3(destClipSpaceWS);
151 if (originClipSpaceWS.w != 0)
152 {
153 originClipSpaceWS3 /= originClipSpaceWS.w;
154 }
155 if (destClipSpaceWS.w != 0)
156 {
157 destClipSpaceWS3 /= destClipSpaceWS.w;
158 }
159 // 获取
160 auto rayDirWorldSpace = glm::normalize(destClipSpaceWS3 - originClipSpaceWS3);
161 return { originClipSpaceWS3, rayDirWorldSpace };
162}
