2// This needs to be used along with a Platform Backend (e.g. Win32)
3
4// Implemented features:
5// [X] Renderer: User texture binding. Use 'ID3D10ShaderResourceView*' as ImTextureID. Read the FAQ about ImTextureID!
6// [X] Renderer: Multi-viewport support. Enable with 'io.ConfigFlags |= ImGuiConfigFlags_ViewportsEnable'.
7// [X] Renderer: Support for large meshes (64k+ vertices) with 16-bit indices.
8
9// You can use unmodified imgui_impl_* files in your project. See examples/ folder for examples of using this.
10// Prefer including the entire imgui/ repository into your project (either as a copy or as a submodule), and only build the backends you need.
11// If you are new to Dear ImGui, read documentation from the docs/ folder + read the top of imgui.cpp.
12// Read online: https://github.com/ocornut/imgui/tree/master/docs

83// Functions
84static void ImGui_ImplDX10_SetupRenderState(ImDrawData* draw_data, ID3D10Device* ctx)
85{
86 ImGui_ImplDX10_Data* bd = ImGui_ImplDX10_GetBackendData();
87
88 // Setup viewport
89 D3D10_VIEWPORT vp;
90 memset(&vp, 0, sizeof(D3D10_VIEWPORT));
91 vp.Width = (UINT)draw_data->DisplaySize.x;
92 vp.Height = (UINT)draw_data->DisplaySize.y;
93 vp.MinDepth = 0.0f;
94 vp.MaxDepth = 1.0f;
95 vp.TopLeftX = vp.TopLeftY = 0;
96 ctx->RSSetViewports(1, &vp);
97
98 // Bind shader and vertex buffers
99 unsigned int stride = sizeof(ImDrawVert);
100 unsigned int offset = 0;
101 ctx->IASetInputLayout(bd->pInputLayout);
102 ctx->IASetVertexBuffers(0, 1, &bd->pVB, &stride, &offset);
103 ctx->IASetIndexBuffer(bd->pIB, sizeof(ImDrawIdx) == 2 ? DXGI_FORMAT_R16_UINT : DXGI_FORMAT_R32_UINT, 0);
104 ctx->IASetPrimitiveTopology(D3D10_PRIMITIVE_TOPOLOGY_TRIANGLELIST);
105 ctx->VSSetShader(bd->pVertexShader);
106 ctx->VSSetConstantBuffers(0, 1, &bd->pVertexConstantBuffer);
107 ctx->PSSetShader(bd->pPixelShader);
108 ctx->PSSetSamplers(0, 1, &bd->pFontSampler);
109 ctx->GSSetShader(NULL);
110
111 // Setup render state
112 const float blend_factor[4] = { 0.f, 0.f, 0.f, 0.f };
113 ctx->OMSetBlendState(bd->pBlendState, blend_factor, 0xffffffff);
114 ctx->OMSetDepthStencilState(bd->pDepthStencilState, 0);
115 ctx->RSSetState(bd->pRasterizerState);
116}

118// Render function
119void ImGui_ImplDX10_RenderDrawData(ImDrawData* draw_data)
120{
121 // Avoid rendering when minimized
122 if (draw_data->DisplaySize.x <= 0.0f || draw_data->DisplaySize.y <= 0.0f)
123 return;
124
125 ImGui_ImplDX10_Data* bd = ImGui_ImplDX10_GetBackendData();
126 ID3D10Device* ctx = bd->pd3dDevice;
127
128 // Create and grow vertex/index buffers if needed
129 if (!bd->pVB || bd->VertexBufferSize < draw_data->TotalVtxCount)
130 {
131 if (bd->pVB) { bd->pVB->Release(); bd->pVB = NULL; }
132 bd->VertexBufferSize = draw_data->TotalVtxCount + 5000;
133 D3D10_BUFFER_DESC desc;
134 memset(&desc, 0, sizeof(D3D10_BUFFER_DESC));
135 desc.Usage = D3D10_USAGE_DYNAMIC;
136 desc.ByteWidth = bd->VertexBufferSize * sizeof(ImDrawVert);
137 desc.BindFlags = D3D10_BIND_VERTEX_BUFFER;
138 desc.CPUAccessFlags = D3D10_CPU_ACCESS_WRITE;
139 desc.MiscFlags = 0;
140 if (ctx->CreateBuffer(&desc, NULL, &bd->pVB) < 0)
141 return;
142 }
143
144 if (!bd->pIB || bd->IndexBufferSize < draw_data->TotalIdxCount)
145 {
146 if (bd->pIB) { bd->pIB->Release(); bd->pIB = NULL; }
147 bd->IndexBufferSize = draw_data->TotalIdxCount + 10000;
148 D3D10_BUFFER_DESC desc;
149 memset(&desc, 0, sizeof(D3D10_BUFFER_DESC));
150 desc.Usage = D3D10_USAGE_DYNAMIC;
151 desc.ByteWidth = bd->IndexBufferSize * sizeof(ImDrawIdx);
152 desc.BindFlags = D3D10_BIND_INDEX_BUFFER;
153 desc.CPUAccessFlags = D3D10_CPU_ACCESS_WRITE;
154 if (ctx->CreateBuffer(&desc, NULL, &bd->pIB) < 0)
155 return;
156 }
157
158 // Copy and convert all vertices into a single contiguous buffer
159 ImDrawVert* vtx_dst = NULL;
160 ImDrawIdx* idx_dst = NULL;
161 bd->pVB->Map(D3D10_MAP_WRITE_DISCARD, 0, (void**)&vtx_dst);
162 bd->pIB->Map(D3D10_MAP_WRITE_DISCARD, 0, (void**)&idx_dst);
163 for (int n = 0; n < draw_data->CmdListsCount; n++)
164 {
165 const ImDrawList* cmd_list = draw_data->CmdLists[n];
166 memcpy(vtx_dst, cmd_list->VtxBuffer.Data, cmd_list->VtxBuffer.Size * sizeof(ImDrawVert));
167 memcpy(idx_dst, cmd_list->IdxBuffer.Data, cmd_list->IdxBuffer.Size * sizeof(ImDrawIdx));
168 vtx_dst += cmd_list->VtxBuffer.Size;
169 idx_dst += cmd_list->IdxBuffer.Size;
170 }
171 bd->pVB->Unmap();
172 bd->pIB->Unmap();
173
174 // Setup orthographic projection matrix into our constant buffer
175 // Our visible imgui space lies from draw_data->DisplayPos (top left) to draw_data->DisplayPos+data_data->DisplaySize (bottom right). DisplayPos is (0,0) for single viewport apps.
176 {
177 void* mapped_resource;
178 if (bd->pVertexConstantBuffer->Map(D3D10_MAP_WRITE_DISCARD, 0, &mapped_resource) != S_OK)
179 return;
180 VERTEX_CONSTANT_BUFFER* constant_buffer = (VERTEX_CONSTANT_BUFFER*)mapped_resource;
181 float L = draw_data->DisplayPos.x;
182 float R = draw_data->DisplayPos.x + draw_data->DisplaySize.x;
183 float T = draw_data->DisplayPos.y;
184 float B = draw_data->DisplayPos.y + draw_data->DisplaySize.y;
185 float mvp[4][4] =
186 {
187 { 2.0f/(R-L), 0.0f, 0.0f, 0.0f },
188 { 0.0f, 2.0f/(T-B), 0.0f, 0.0f },
189 { 0.0f, 0.0f, 0.5f, 0.0f },
190 { (R+L)/(L-R), (T+B)/(B-T), 0.5f, 1.0f },
191 };
192 memcpy(&constant_buffer->mvp, mvp, sizeof(mvp));
193 bd->pVertexConstantBuffer->Unmap();
194 }

196 // Backup DX state that will be modified to restore it afterwards (unfortunately this is very ugly looking and verbose. Close your eyes!)
197 struct BACKUP_DX10_STATE
198 {
199 UINT ScissorRectsCount, ViewportsCount;
200 D3D10_RECT ScissorRects[D3D10_VIEWPORT_AND_SCISSORRECT_OBJECT_COUNT_PER_PIPELINE];
201 D3D10_VIEWPORT Viewports[D3D10_VIEWPORT_AND_SCISSORRECT_OBJECT_COUNT_PER_PIPELINE];
202 ID3D10RasterizerState* RS;
203 ID3D10BlendState* BlendState;
204 FLOAT BlendFactor[4];
205 UINT SampleMask;
206 UINT StencilRef;
207 ID3D10DepthStencilState* DepthStencilState;
208 ID3D10ShaderResourceView* PSShaderResource;
209 ID3D10SamplerState* PSSampler;
210 ID3D10PixelShader* PS;
211 ID3D10VertexShader* VS;
212 ID3D10GeometryShader* GS;
213 D3D10_PRIMITIVE_TOPOLOGY PrimitiveTopology;
214 ID3D10Buffer* IndexBuffer, *VertexBuffer, *VSConstantBuffer;
215 UINT IndexBufferOffset, VertexBufferStride, VertexBufferOffset;
216 DXGI_FORMAT IndexBufferFormat;
217 ID3D10InputLayout* InputLayout;
218 };
219 BACKUP_DX10_STATE old = {};
220 old.ScissorRectsCount = old.ViewportsCount = D3D10_VIEWPORT_AND_SCISSORRECT_OBJECT_COUNT_PER_PIPELINE;
221 ctx->RSGetScissorRects(&old.ScissorRectsCount, old.ScissorRects);
222 ctx->RSGetViewports(&old.ViewportsCount, old.Viewports);
223 ctx->RSGetState(&old.RS);
224 ctx->OMGetBlendState(&old.BlendState, old.BlendFactor, &old.SampleMask);
225 ctx->OMGetDepthStencilState(&old.DepthStencilState, &old.StencilRef);
226 ctx->PSGetShaderResources(0, 1, &old.PSShaderResource);
227 ctx->PSGetSamplers(0, 1, &old.PSSampler);
228 ctx->PSGetShader(&old.PS);
229 ctx->VSGetShader(&old.VS);
230 ctx->VSGetConstantBuffers(0, 1, &old.VSConstantBuffer);
231 ctx->GSGetShader(&old.GS);
232 ctx->IAGetPrimitiveTopology(&old.PrimitiveTopology);
233 ctx->IAGetIndexBuffer(&old.IndexBuffer, &old.IndexBufferFormat, &old.IndexBufferOffset);
234 ctx->IAGetVertexBuffers(0, 1, &old.VertexBuffer, &old.VertexBufferStride, &old.VertexBufferOffset);
235 ctx->IAGetInputLayout(&old.InputLayout);
236
237 // Setup desired DX state
238 ImGui_ImplDX10_SetupRenderState(draw_data, ctx);
239
240 // Render command lists
241 // (Because we merged all buffers into a single one, we maintain our own offset into them)
242 int global_vtx_offset = 0;
243 int global_idx_offset = 0;
244 ImVec2 clip_off = draw_data->DisplayPos;
245 for (int n = 0; n < draw_data->CmdListsCount; n++)
246 {
247 const ImDrawList* cmd_list = draw_data->CmdLists[n];
248 for (int cmd_i = 0; cmd_i < cmd_list->CmdBuffer.Size; cmd_i++)
249 {
250 const ImDrawCmd* pcmd = &cmd_list->CmdBuffer[cmd_i];
251 if (pcmd->UserCallback)
252 {
253 // User callback, registered via ImDrawList::AddCallback()
254 // (ImDrawCallback_ResetRenderState is a special callback value used by the user to request the renderer to reset render state.)
255 if (pcmd->UserCallback == ImDrawCallback_ResetRenderState)
256 ImGui_ImplDX10_SetupRenderState(draw_data, ctx);
257 else
258 pcmd->UserCallback(cmd_list, pcmd);
259 }
260 else
261 {
262 // Project scissor/clipping rectangles into framebuffer space
263 ImVec2 clip_min(pcmd->ClipRect.x - clip_off.x, pcmd->ClipRect.y - clip_off.y);
264 ImVec2 clip_max(pcmd->ClipRect.z - clip_off.x, pcmd->ClipRect.w - clip_off.y);
265 if (clip_max.x <= clip_min.x || clip_max.y <= clip_min.y)
266 continue;
267
268 // Apply scissor/clipping rectangle
269 const D3D10_RECT r = { (LONG)clip_min.x, (LONG)clip_min.y, (LONG)clip_max.x, (LONG)clip_max.y };
270 ctx->RSSetScissorRects(1, &r);
271
272 // Bind texture, Draw
273 ID3D10ShaderResourceView* texture_srv = (ID3D10ShaderResourceView*)pcmd->GetTexID();
274 ctx->PSSetShaderResources(0, 1, &texture_srv);
275 ctx->DrawIndexed(pcmd->ElemCount, pcmd->IdxOffset + global_idx_offset, pcmd->VtxOffset + global_vtx_offset);
276 }
277 }
278 global_idx_offset += cmd_list->IdxBuffer.Size;
279 global_vtx_offset += cmd_list->VtxBuffer.Size;
280 }
281
282 // Restore modified DX state
283 ctx->RSSetScissorRects(old.ScissorRectsCount, old.ScissorRects);
284 ctx->RSSetViewports(old.ViewportsCount, old.Viewports);
285 ctx->RSSetState(old.RS); if (old.RS) old.RS->Release();
286 ctx->OMSetBlendState(old.BlendState, old.BlendFactor, old.SampleMask); if (old.BlendState) old.BlendState->Release();
287 ctx->OMSetDepthStencilState(old.DepthStencilState, old.StencilRef); if (old.DepthStencilState) old.DepthStencilState->Release();
288 ctx->PSSetShaderResources(0, 1, &old.PSShaderResource); if (old.PSShaderResource) old.PSShaderResource->Release();
289 ctx->PSSetSamplers(0, 1, &old.PSSampler); if (old.PSSampler) old.PSSampler->Release();
290 ctx->PSSetShader(old.PS); if (old.PS) old.PS->Release();
291 ctx->VSSetShader(old.VS); if (old.VS) old.VS->Release();
292 ctx->GSSetShader(old.GS); if (old.GS) old.GS->Release();
293 ctx->VSSetConstantBuffers(0, 1, &old.VSConstantBuffer); if (old.VSConstantBuffer) old.VSConstantBuffer->Release();
294 ctx->IASetPrimitiveTopology(old.PrimitiveTopology);
295 ctx->IASetIndexBuffer(old.IndexBuffer, old.IndexBufferFormat, old.IndexBufferOffset); if (old.IndexBuffer) old.IndexBuffer->Release();
296 ctx->IASetVertexBuffers(0, 1, &old.VertexBuffer, &old.VertexBufferStride, &old.VertexBufferOffset); if (old.VertexBuffer) old.VertexBuffer->Release();
297 ctx->IASetInputLayout(old.InputLayout); if (old.InputLayout) old.InputLayout->Release();
298}

300static void ImGui_ImplDX10_CreateFontsTexture()
301{
302 // Build texture atlas
303 ImGui_ImplDX10_Data* bd = ImGui_ImplDX10_GetBackendData();
304 ImGuiIO& io = ImGui::GetIO();
305 unsigned char* pixels;
306 int width, height;
307 io.Fonts->GetTexDataAsRGBA32(&pixels, &width, &height);
308
309 // Upload texture to graphics system
310 {
311 D3D10_TEXTURE2D_DESC desc;
312 ZeroMemory(&desc, sizeof(desc));
313 desc.Width = width;
314 desc.Height = height;
315 desc.MipLevels = 1;
316 desc.ArraySize = 1;
317 desc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
318 desc.SampleDesc.Count = 1;
319 desc.Usage = D3D10_USAGE_DEFAULT;
320 desc.BindFlags = D3D10_BIND_SHADER_RESOURCE;
321 desc.CPUAccessFlags = 0;
322
323 ID3D10Texture2D* pTexture = NULL;
324 D3D10_SUBRESOURCE_DATA subResource;
325 subResource.pSysMem = pixels;
326 subResource.SysMemPitch = desc.Width * 4;
327 subResource.SysMemSlicePitch = 0;
328 bd->pd3dDevice->CreateTexture2D(&desc, &subResource, &pTexture);
329 IM_ASSERT(pTexture != NULL);
330
331 // Create texture view
332 D3D10_SHADER_RESOURCE_VIEW_DESC srv_desc;
333 ZeroMemory(&srv_desc, sizeof(srv_desc));
334 srv_desc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
335 srv_desc.ViewDimension = D3D10_SRV_DIMENSION_TEXTURE2D;
336 srv_desc.Texture2D.MipLevels = desc.MipLevels;
337 srv_desc.Texture2D.MostDetailedMip = 0;
338 bd->pd3dDevice->CreateShaderResourceView(pTexture, &srv_desc, &bd->pFontTextureView);
339 pTexture->Release();
340 }
341
342 // Store our identifier
343 io.Fonts->SetTexID((ImTextureID)bd->pFontTextureView);
344
345 // Create texture sampler
346 {
347 D3D10_SAMPLER_DESC desc;
348 ZeroMemory(&desc, sizeof(desc));
349 desc.Filter = D3D10_FILTER_MIN_MAG_MIP_LINEAR;
350 desc.AddressU = D3D10_TEXTURE_ADDRESS_WRAP;
351 desc.AddressV = D3D10_TEXTURE_ADDRESS_WRAP;
352 desc.AddressW = D3D10_TEXTURE_ADDRESS_WRAP;
353 desc.MipLODBias = 0.f;
354 desc.ComparisonFunc = D3D10_COMPARISON_ALWAYS;
355 desc.MinLOD = 0.f;
356 desc.MaxLOD = 0.f;
357 bd->pd3dDevice->CreateSamplerState(&desc, &bd->pFontSampler);
358 }
359}

361bool ImGui_ImplDX10_CreateDeviceObjects()
362{
363 ImGui_ImplDX10_Data* bd = ImGui_ImplDX10_GetBackendData();
364 if (!bd->pd3dDevice)
365 return false;
366 if (bd->pFontSampler)
367 ImGui_ImplDX10_InvalidateDeviceObjects();
368
369 // By using D3DCompile() from <d3dcompiler.h> / d3dcompiler.lib, we introduce a dependency to a given version of d3dcompiler_XX.dll (see D3DCOMPILER_DLL_A)
370 // If you would like to use this DX10 sample code but remove this dependency you can:
371 // 1) compile once, save the compiled shader blobs into a file or source code and pass them to CreateVertexShader()/CreatePixelShader() [preferred solution]
372 // 2) use code to detect any version of the DLL and grab a pointer to D3DCompile from the DLL.
373 // See https://github.com/ocornut/imgui/pull/638 for sources and details.
374
375 // Create the vertex shader
376 {
377 static const char* vertexShader =
378 "cbuffer vertexBuffer : register(b0) \
379 {\
380 float4x4 ProjectionMatrix; \
381 };\
382 struct VS_INPUT\
383 {\
384 float2 pos : POSITION;\
385 float4 col : COLOR0;\
386 float2 uv : TEXCOORD0;\
387 };\
388 \
389 struct PS_INPUT\
390 {\
391 float4 pos : SV_POSITION;\
392 float4 col : COLOR0;\
393 float2 uv : TEXCOORD0;\
394 };\
395 \
396 PS_INPUT main(VS_INPUT input)\
397 {\
398 PS_INPUT output;\
399 output.pos = mul( ProjectionMatrix, float4(input.pos.xy, 0.f, 1.f));\
400 output.col = input.col;\
401 output.uv = input.uv;\
402 return output;\
403 }";
404
405 ID3DBlob* vertexShaderBlob;
406 if (FAILED(D3DCompile(vertexShader, strlen(vertexShader), NULL, NULL, NULL, "main", "vs_4_0", 0, 0, &vertexShaderBlob, NULL)))
407 return false; // NB: Pass ID3DBlob* pErrorBlob to D3DCompile() to get error showing in (const char*)pErrorBlob->GetBufferPointer(). Make sure to Release() the blob!
408 if (bd->pd3dDevice->CreateVertexShader(vertexShaderBlob->GetBufferPointer(), vertexShaderBlob->GetBufferSize(), &bd->pVertexShader) != S_OK)
409 {
410 vertexShaderBlob->Release();
411 return false;
412 }
413
414 // Create the input layout
415 D3D10_INPUT_ELEMENT_DESC local_layout[] =
416 {
417 { "POSITION", 0, DXGI_FORMAT_R32G32_FLOAT, 0, (UINT)IM_OFFSETOF(ImDrawVert, pos), D3D10_INPUT_PER_VERTEX_DATA, 0 },
418 { "TEXCOORD", 0, DXGI_FORMAT_R32G32_FLOAT, 0, (UINT)IM_OFFSETOF(ImDrawVert, uv), D3D10_INPUT_PER_VERTEX_DATA, 0 },
419 { "COLOR", 0, DXGI_FORMAT_R8G8B8A8_UNORM, 0, (UINT)IM_OFFSETOF(ImDrawVert, col), D3D10_INPUT_PER_VERTEX_DATA, 0 },
420 };
421 if (bd->pd3dDevice->CreateInputLayout(local_layout, 3, vertexShaderBlob->GetBufferPointer(), vertexShaderBlob->GetBufferSize(), &bd->pInputLayout) != S_OK)
422 {
423 vertexShaderBlob->Release();
424 return false;
425 }
426 vertexShaderBlob->Release();
427
428 // Create the constant buffer
429 {
430 D3D10_BUFFER_DESC desc;
431 desc.ByteWidth = sizeof(VERTEX_CONSTANT_BUFFER);
432 desc.Usage = D3D10_USAGE_DYNAMIC;
433 desc.BindFlags = D3D10_BIND_CONSTANT_BUFFER;
434 desc.CPUAccessFlags = D3D10_CPU_ACCESS_WRITE;
435 desc.MiscFlags = 0;
436 bd->pd3dDevice->CreateBuffer(&desc, NULL, &bd->pVertexConstantBuffer);
437 }
438 }
439
440 // Create the pixel shader
441 {
442 static const char* pixelShader =
443 "struct PS_INPUT\
444 {\
445 float4 pos : SV_POSITION;\
446 float4 col : COLOR0;\
447 float2 uv : TEXCOORD0;\
448 };\
449 sampler sampler0;\
450 Texture2D texture0;\
451 \
452 float4 main(PS_INPUT input) : SV_Target\
453 {\
454 float4 out_col = input.col * texture0.Sample(sampler0, input.uv); \
455 return out_col; \
456 }";
457
458 ID3DBlob* pixelShaderBlob;
459 if (FAILED(D3DCompile(pixelShader, strlen(pixelShader), NULL, NULL, NULL, "main", "ps_4_0", 0, 0, &pixelShaderBlob, NULL)))
460 return false; // NB: Pass ID3DBlob* pErrorBlob to D3DCompile() to get error showing in (const char*)pErrorBlob->GetBufferPointer(). Make sure to Release() the blob!
461 if (bd->pd3dDevice->CreatePixelShader(pixelShaderBlob->GetBufferPointer(), pixelShaderBlob->GetBufferSize(), &bd->pPixelShader) != S_OK)
462 {
463 pixelShaderBlob->Release();
464 return false;
465 }
466 pixelShaderBlob->Release();
467 }
468
469 // Create the blending setup
470 {
471 D3D10_BLEND_DESC desc;
472 ZeroMemory(&desc, sizeof(desc));
473 desc.AlphaToCoverageEnable = false;
474 desc.BlendEnable[0] = true;
475 desc.SrcBlend = D3D10_BLEND_SRC_ALPHA;
476 desc.DestBlend = D3D10_BLEND_INV_SRC_ALPHA;
477 desc.BlendOp = D3D10_BLEND_OP_ADD;
478 desc.SrcBlendAlpha = D3D10_BLEND_ONE;
479 desc.DestBlendAlpha = D3D10_BLEND_INV_SRC_ALPHA;
480 desc.BlendOpAlpha = D3D10_BLEND_OP_ADD;
481 desc.RenderTargetWriteMask[0] = D3D10_COLOR_WRITE_ENABLE_ALL;
482 bd->pd3dDevice->CreateBlendState(&desc, &bd->pBlendState);
483 }
484
485 // Create the rasterizer state
486 {
487 D3D10_RASTERIZER_DESC desc;
488 ZeroMemory(&desc, sizeof(desc));
489 desc.FillMode = D3D10_FILL_SOLID;
490 desc.CullMode = D3D10_CULL_NONE;
491 desc.ScissorEnable = true;
492 desc.DepthClipEnable = true;
493 bd->pd3dDevice->CreateRasterizerState(&desc, &bd->pRasterizerState);
494 }
495
496 // Create depth-stencil State
497 {
498 D3D10_DEPTH_STENCIL_DESC desc;
499 ZeroMemory(&desc, sizeof(desc));
500 desc.DepthEnable = false;
501 desc.DepthWriteMask = D3D10_DEPTH_WRITE_MASK_ALL;
502 desc.DepthFunc = D3D10_COMPARISON_ALWAYS;
503 desc.StencilEnable = false;
504 desc.FrontFace.StencilFailOp = desc.FrontFace.StencilDepthFailOp = desc.FrontFace.StencilPassOp = D3D10_STENCIL_OP_KEEP;
505 desc.FrontFace.StencilFunc = D3D10_COMPARISON_ALWAYS;
506 desc.BackFace = desc.FrontFace;
507 bd->pd3dDevice->CreateDepthStencilState(&desc, &bd->pDepthStencilState);
508 }

510 ImGui_ImplDX10_CreateFontsTexture();
511
512 return true;
513}

534bool ImGui_ImplDX10_Init(ID3D10Device* device)
535{
536 ImGuiIO& io = ImGui::GetIO();
537 IM_ASSERT(io.BackendRendererUserData == NULL && "Already initialized a renderer backend!");
538
539 // Setup backend capabilities flags
540 ImGui_ImplDX10_Data* bd = IM_NEW(ImGui_ImplDX10_Data)();
541 io.BackendRendererUserData = (void*)bd;
542 io.BackendRendererName = "imgui_impl_dx10";
543 io.BackendFlags |= ImGuiBackendFlags_RendererHasVtxOffset; // We can honor the ImDrawCmd::VtxOffset field, allowing for large meshes.
544 io.BackendFlags |= ImGuiBackendFlags_RendererHasViewports; // We can create multi-viewports on the Renderer side (optional)
545
546 // Get factory from device
547 IDXGIDevice* pDXGIDevice = NULL;
548 IDXGIAdapter* pDXGIAdapter = NULL;
549 IDXGIFactory* pFactory = NULL;
550 if (device->QueryInterface(IID_PPV_ARGS(&pDXGIDevice)) == S_OK)
551 if (pDXGIDevice->GetParent(IID_PPV_ARGS(&pDXGIAdapter)) == S_OK)
552 if (pDXGIAdapter->GetParent(IID_PPV_ARGS(&pFactory)) == S_OK)
553 {
554 bd->pd3dDevice = device;
555 bd->pFactory = pFactory;
556 }
557 if (pDXGIDevice) pDXGIDevice->Release();
558 if (pDXGIAdapter) pDXGIAdapter->Release();
559 bd->pd3dDevice->AddRef();
560
561 if (io.ConfigFlags & ImGuiConfigFlags_ViewportsEnable)
562 ImGui_ImplDX10_InitPlatformInterface();
563 return true;
564}

590//--------------------------------------------------------------------------------------------------------
591// MULTI-VIEWPORT / PLATFORM INTERFACE SUPPORT
592// This is an _advanced_ and _optional_ feature, allowing the backend to create and handle multiple viewports simultaneously.
593// If you are new to dear imgui or creating a new binding for dear imgui, it is recommended that you completely ignore this section first..
594//--------------------------------------------------------------------------------------------------------
595
596// Helper structure we store in the void* RenderUserData field of each ImGuiViewport to easily retrieve our backend data.
597struct ImGui_ImplDX10_ViewportData
598{
599 IDXGISwapChain* SwapChain;
600 ID3D10RenderTargetView* RTView;
601
602 ImGui_ImplDX10_ViewportData() { SwapChain = NULL; RTView = NULL; }
603 ~ImGui_ImplDX10_ViewportData() { IM_ASSERT(SwapChain == NULL && RTView == NULL); }
604};
605
606static void ImGui_ImplDX10_CreateWindow(ImGuiViewport* viewport)
607{
608 ImGui_ImplDX10_Data* bd = ImGui_ImplDX10_GetBackendData();
609 ImGui_ImplDX10_ViewportData* vd = IM_NEW(ImGui_ImplDX10_ViewportData)();
610 viewport->RendererUserData = vd;
611
612 // PlatformHandleRaw should always be a HWND, whereas PlatformHandle might be a higher-level handle (e.g. GLFWWindow*, SDL_Window*).
613 // Some backends will leave PlatformHandleRaw NULL, in which case we assume PlatformHandle will contain the HWND.
614 HWND hwnd = viewport->PlatformHandleRaw ? (HWND)viewport->PlatformHandleRaw : (HWND)viewport->PlatformHandle;
615 IM_ASSERT(hwnd != 0);
616
617 // Create swap chain
618 DXGI_SWAP_CHAIN_DESC sd;
619 ZeroMemory(&sd, sizeof(sd));
620 sd.BufferDesc.Width = (UINT)viewport->Size.x;
621 sd.BufferDesc.Height = (UINT)viewport->Size.y;
622 sd.BufferDesc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
623 sd.SampleDesc.Count = 1;
624 sd.SampleDesc.Quality = 0;
625 sd.BufferUsage = DXGI_USAGE_RENDER_TARGET_OUTPUT;
626 sd.BufferCount = 1;
627 sd.OutputWindow = hwnd;
628 sd.Windowed = TRUE;
629 sd.SwapEffect = DXGI_SWAP_EFFECT_DISCARD;
630 sd.Flags = 0;
631
632 IM_ASSERT(vd->SwapChain == NULL && vd->RTView == NULL);
633 bd->pFactory->CreateSwapChain(bd->pd3dDevice, &sd, &vd->SwapChain);
634
635 // Create the render target
636 if (vd->SwapChain)
637 {
638 ID3D10Texture2D* pBackBuffer;
639 vd->SwapChain->GetBuffer(0, IID_PPV_ARGS(&pBackBuffer));
640 bd->pd3dDevice->CreateRenderTargetView(pBackBuffer, NULL, &vd->RTView);
641 pBackBuffer->Release();
642 }
643}

645static void ImGui_ImplDX10_DestroyWindow(ImGuiViewport* viewport)
646{
647 // The main viewport (owned by the application) will always have RendererUserData == NULL here since we didn't create the data for it.
648 if (ImGui_ImplDX10_ViewportData* vd = (ImGui_ImplDX10_ViewportData*)viewport->RendererUserData)
649 {
650 if (vd->SwapChain)
651 vd->SwapChain->Release();
652 vd->SwapChain = NULL;
653 if (vd->RTView)
654 vd->RTView->Release();
655 vd->RTView = NULL;
656 IM_DELETE(vd);
657 }
658 viewport->RendererUserData = NULL;
659}

662{
663 ImGui_ImplDX10_Data* bd = ImGui_ImplDX10_GetBackendData();
