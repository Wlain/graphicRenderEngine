1// dear imgui: Platform Backend for SDL2
2// This needs to be used along with a Renderer (e.g. DirectX11, OpenGL3, Vulkan..)
3// (Info: SDL2 is a cross-platform general purpose library for handling windows, inputs, graphics context creation, etc.)
4// (Prefer SDL 2.0.5+ for full feature support.)
5
6// Implemented features:
7// [X] Platform: Clipboard support.
8// [X] Platform: Keyboard support. Since 1.87 we are using the io.AddKeyEvent() function. Pass ImGuiKey values to all key functions e.g. ImGui::IsKeyPressed(ImGuiKey_Space). [Legacy SDL_SCANCODE_* values will also be supported unless IMGUI_DISABLE_OBSOLETE_KEYIO is set]
9// [X] Platform: Gamepad support. Enabled with 'io.ConfigFlags |= ImGuiConfigFlags_NavEnableGamepad'.
10// [X] Platform: Mouse cursor shape and visibility. Disable with 'io.ConfigFlags |= ImGuiConfigFlags_NoMouseCursorChange'.
11// [X] Platform: Multi-viewport support (multiple windows). Enable with 'io.ConfigFlags |= ImGuiConfigFlags_ViewportsEnable'.
12// Missing features:
13// [ ] Platform: SDL2 handling of IME under Windows appears to be broken and it explicitly disable the regular Windows IME. You can restore Windows IME by compiling SDL with SDL_DISABLE_WINDOWS_IME.
14// [ ] Platform: Multi-viewport + Minimized windows seems to break mouse wheel events (at least under Windows).
15
16// You can use unmodified imgui_impl_* files in your project. See examples/ folder for examples of using this.
17// Prefer including the entire imgui/ repository into your project (either as a copy or as a submodule), and only build the backends you need.
18// If you are new to Dear ImGui, read documentation from the docs/ folder + read the top of imgui.cpp.
19// Read online: https://github.com/ocornut/imgui/tree/master/docs

101// Backend data stored in io.BackendPlatformUserData to allow support for multiple Dear ImGui contexts
102// It is STRONGLY preferred that you use docking branch with multi-viewports (== single Dear ImGui context + multiple windows) instead of multiple Dear ImGui contexts.
103// FIXME: multi-context support is not well tested and probably dysfunctional in this backend.
104// FIXME: some shared resources (mouse cursor shape, gamepad) are mishandled when using multi-context.
105static ImGui_ImplSDL2_Data* ImGui_ImplSDL2_GetBackendData()
106{
107 return ImGui::GetCurrentContext() ? (ImGui_ImplSDL2_Data*)ImGui::GetIO().BackendPlatformUserData : NULL;
108}

243static void ImGui_ImplSDL2_UpdateKeyModifiers(SDL_Keymod sdl_key_mods)
244{
245 ImGuiIO& io = ImGui::GetIO();
246 ImGuiKeyModFlags key_mods =
247 ((sdl_key_mods & KMOD_CTRL) ? ImGuiKeyModFlags_Ctrl : 0) |
248 ((sdl_key_mods & KMOD_SHIFT) ? ImGuiKeyModFlags_Shift : 0) |
249 ((sdl_key_mods & KMOD_ALT) ? ImGuiKeyModFlags_Alt : 0) |
250 ((sdl_key_mods & KMOD_GUI) ? ImGuiKeyModFlags_Super : 0);
251 io.AddKeyModsEvent(key_mods);
252}

254// You can read the io.WantCaptureMouse, io.WantCaptureKeyboard flags to tell if dear imgui wants to use your inputs.
255// - When io.WantCaptureMouse is true, do not dispatch mouse input data to your main application.
256// - When io.WantCaptureKeyboard is true, do not dispatch keyboard input data to your main application.
257// Generally you may always pass all inputs to dear imgui, and hide them from your application based on those two flags.
258// If you have multiple SDL events and some of them are not meant to be used by dear imgui, you may need to filter events based on their windowID field.
259bool ImGui_ImplSDL2_ProcessEvent(const SDL_Event* event)
260{
261 ImGuiIO& io = ImGui::GetIO();
262 ImGui_ImplSDL2_Data* bd = ImGui_ImplSDL2_GetBackendData();
263
264 switch (event->type)
265 {
266 case SDL_MOUSEMOTION:
267 {
268 ImVec2 mouse_pos((float)event->motion.x, (float)event->motion.y);
269 if (io.ConfigFlags & ImGuiConfigFlags_ViewportsEnable)
270 {
271 int window_x, window_y;
272 SDL_GetWindowPosition(SDL_GetWindowFromID(event->motion.windowID), &window_x, &window_y);
273 mouse_pos.x += window_x;
274 mouse_pos.y += window_y;
275 }
276 io.AddMousePosEvent(mouse_pos.x, mouse_pos.y);
277 return true;
278 }
279 case SDL_MOUSEWHEEL:
280 {
281 float wheel_x = (event->wheel.x > 0) ? 1.0f : (event->wheel.x < 0) ? -1.0f : 0.0f;
282 float wheel_y = (event->wheel.y > 0) ? 1.0f : (event->wheel.y < 0) ? -1.0f : 0.0f;
283 io.AddMouseWheelEvent(wheel_x, wheel_y);
284 return true;
285 }
286 case SDL_MOUSEBUTTONDOWN:
287 case SDL_MOUSEBUTTONUP:
288 {
289 int mouse_button = -1;
290 if (event->button.button == SDL_BUTTON_LEFT) { mouse_button = 0; }
291 if (event->button.button == SDL_BUTTON_RIGHT) { mouse_button = 1; }
292 if (event->button.button == SDL_BUTTON_MIDDLE) { mouse_button = 2; }
293 if (mouse_button == -1)
294 break;
295 io.AddMouseButtonEvent(mouse_button, (event->type == SDL_MOUSEBUTTONDOWN));
296 bd->MouseButtonsDown = (event->type == SDL_MOUSEBUTTONDOWN) ? (bd->MouseButtonsDown | (1 << mouse_button)) : (bd->MouseButtonsDown & ~(1 << mouse_button));
297 return true;
298 }
299 case SDL_TEXTINPUT:
300 {
301 io.AddInputCharactersUTF8(event->text.text);
302 return true;
303 }
304 case SDL_KEYDOWN:
305 case SDL_KEYUP:
306 {
307 ImGui_ImplSDL2_UpdateKeyModifiers((SDL_Keymod)event->key.keysym.mod);
308 ImGuiKey key = ImGui_ImplSDL2_KeycodeToImGuiKey(event->key.keysym.sym);
309 io.AddKeyEvent(key, (event->type == SDL_KEYDOWN));
310 io.SetKeyEventNativeData(key, event->key.keysym.sym, event->key.keysym.scancode, event->key.keysym.scancode); // To support legacy indexing (<1.87 user code). Legacy backend uses SDLK_*** as indices to IsKeyXXX() functions.
311 return true;
312 }
313 case SDL_WINDOWEVENT:
314 {
315 Uint8 window_event = event->window.event;
316 if (window_event == SDL_WINDOWEVENT_LEAVE)
317 io.AddMousePosEvent(-FLT_MAX, -FLT_MAX);
318 if (window_event == SDL_WINDOWEVENT_FOCUS_GAINED)
319 io.AddFocusEvent(true);
320 else if (window_event == SDL_WINDOWEVENT_FOCUS_LOST)
321 io.AddFocusEvent(false);
322 if (window_event == SDL_WINDOWEVENT_CLOSE || window_event == SDL_WINDOWEVENT_MOVED || window_event == SDL_WINDOWEVENT_RESIZED)
323 if (ImGuiViewport* viewport = ImGui::FindViewportByPlatformHandle((void*)SDL_GetWindowFromID(event->window.windowID)))
324 {
325 if (window_event == SDL_WINDOWEVENT_CLOSE)
326 viewport->PlatformRequestClose = true;
327 if (window_event == SDL_WINDOWEVENT_MOVED)
328 viewport->PlatformRequestMove = true;
329 if (window_event == SDL_WINDOWEVENT_RESIZED)
330 viewport->PlatformRequestResize = true;
331 return true;
332 }
333 return true;
334 }
335 }
336 return false;
337}

339static bool ImGui_ImplSDL2_Init(SDL_Window* window, void* sdl_gl_context)
340{
341 ImGuiIO& io = ImGui::GetIO();
342 IM_ASSERT(io.BackendPlatformUserData == NULL && "Already initialized a platform backend!");
343
344 // Check and store if we are on a SDL backend that supports global mouse position
345 // ("wayland" and "rpi" don't support it, but we chose to use a white-list instead of a black-list)
346 bool mouse_can_use_global_state = false;
347#if SDL_HAS_CAPTURE_AND_GLOBAL_MOUSE
348 const char* sdl_backend = SDL_GetCurrentVideoDriver();
349 const char* global_mouse_whitelist[] = { "windows", "cocoa", "x11", "DIVE", "VMAN" };
350 for (int n = 0; n < IM_ARRAYSIZE(global_mouse_whitelist); n++)
351 if (strncmp(sdl_backend, global_mouse_whitelist[n], strlen(global_mouse_whitelist[n])) == 0)
352 mouse_can_use_global_state = true;
353#endif
354
355 // Setup backend capabilities flags
356 ImGui_ImplSDL2_Data* bd = IM_NEW(ImGui_ImplSDL2_Data)();
357 io.BackendPlatformUserData = (void*)bd;
358 io.BackendPlatformName = "imgui_impl_sdl";
359 io.BackendFlags |= ImGuiBackendFlags_HasMouseCursors; // We can honor GetMouseCursor() values (optional)
360 io.BackendFlags |= ImGuiBackendFlags_HasSetMousePos; // We can honor io.WantSetMousePos requests (optional, rarely used)
361 if (mouse_can_use_global_state)
362 io.BackendFlags |= ImGuiBackendFlags_PlatformHasViewports; // We can create multi-viewports on the Platform side (optional)
363
364 bd->Window = window;
365 bd->MouseCanUseGlobalState = mouse_can_use_global_state;
366
367 io.SetClipboardTextFn = ImGui_ImplSDL2_SetClipboardText;
368 io.GetClipboardTextFn = ImGui_ImplSDL2_GetClipboardText;
369 io.ClipboardUserData = NULL;
370
371 // Load mouse cursors
372 bd->MouseCursors[ImGuiMouseCursor_Arrow] = SDL_CreateSystemCursor(SDL_SYSTEM_CURSOR_ARROW);
373 bd->MouseCursors[ImGuiMouseCursor_TextInput] = SDL_CreateSystemCursor(SDL_SYSTEM_CURSOR_IBEAM);
374 bd->MouseCursors[ImGuiMouseCursor_ResizeAll] = SDL_CreateSystemCursor(SDL_SYSTEM_CURSOR_SIZEALL);
375 bd->MouseCursors[ImGuiMouseCursor_ResizeNS] = SDL_CreateSystemCursor(SDL_SYSTEM_CURSOR_SIZENS);
376 bd->MouseCursors[ImGuiMouseCursor_ResizeEW] = SDL_CreateSystemCursor(SDL_SYSTEM_CURSOR_SIZEWE);
377 bd->MouseCursors[ImGuiMouseCursor_ResizeNESW] = SDL_CreateSystemCursor(SDL_SYSTEM_CURSOR_SIZENESW);
378 bd->MouseCursors[ImGuiMouseCursor_ResizeNWSE] = SDL_CreateSystemCursor(SDL_SYSTEM_CURSOR_SIZENWSE);
379 bd->MouseCursors[ImGuiMouseCursor_Hand] = SDL_CreateSystemCursor(SDL_SYSTEM_CURSOR_HAND);
380 bd->MouseCursors[ImGuiMouseCursor_NotAllowed] = SDL_CreateSystemCursor(SDL_SYSTEM_CURSOR_NO);
381
382 // Set platform dependent data in viewport
383 // Our mouse update function expect PlatformHandle to be filled for the main viewport
384 ImGuiViewport* main_viewport = ImGui::GetMainViewport();
385 main_viewport->PlatformHandle = (void*)window;
386#ifdef _WIN32
387 SDL_SysWMinfo info;
388 SDL_VERSION(&info.version);
389 if (SDL_GetWindowWMInfo(window, &info))
390 main_viewport->PlatformHandleRaw = (void*)info.info.win.window;
391#endif
392
393 // Set SDL hint to receive mouse click events on window focus, otherwise SDL doesn't emit the event.
394 // Without this, when clicking to gain focus, our widgets wouldn't activate even though they showed as hovered.
395 // (This is unfortunately a global SDL setting, so enabling it might have a side-effect on your application.
396 // It is unlikely to make a difference, but if your app absolutely needs to ignore the initial on-focus click:
397 // you can ignore SDL_MOUSEBUTTONDOWN events coming right after a SDL_WINDOWEVENT_FOCUS_GAINED)
398#if SDL_HAS_MOUSE_FOCUS_CLICKTHROUGH
399 SDL_SetHint(SDL_HINT_MOUSE_FOCUS_CLICKTHROUGH, "1");
400#endif
401
402 // Update monitors
403 ImGui_ImplSDL2_UpdateMonitors();
404
405 // We need SDL_CaptureMouse(), SDL_GetGlobalMouseState() from SDL 2.0.4+ to support multiple viewports.
406 // We left the call to ImGui_ImplSDL2_InitPlatformInterface() outside of #ifdef to avoid unused-function warnings.
407 if ((io.ConfigFlags & ImGuiConfigFlags_ViewportsEnable) && (io.BackendFlags & ImGuiBackendFlags_PlatformHasViewports))
408 ImGui_ImplSDL2_InitPlatformInterface(window, sdl_gl_context);
409
410 return true;
411}

466// This code is incredibly messy because some of the functions we need for full viewport support are not available in SDL < 2.0.4.
467static void ImGui_ImplSDL2_UpdateMouseData()
468{
469 ImGui_ImplSDL2_Data* bd = ImGui_ImplSDL2_GetBackendData();
470 ImGuiIO& io = ImGui::GetIO();
471
472 // We forward mouse input when hovered or captured (via SDL_MOUSEMOTION) or when focused (below)
473#if SDL_HAS_CAPTURE_AND_GLOBAL_MOUSE
474 // SDL_CaptureMouse() let the OS know e.g. that our imgui drag outside the SDL window boundaries shouldn't e.g. trigger other operations outside
475 SDL_CaptureMouse(bd->MouseButtonsDown != 0 ? SDL_TRUE : SDL_FALSE);
476 SDL_Window* focused_window = SDL_GetKeyboardFocus();
477 const bool is_app_focused = (focused_window && (bd->Window == focused_window || ImGui::FindViewportByPlatformHandle((void*)focused_window)));
478#else
479 SDL_Window* focused_window = bd->Window;
480 const bool is_app_focused = (SDL_GetWindowFlags(bd->Window) & SDL_WINDOW_INPUT_FOCUS) != 0; // SDL 2.0.3 and non-windowed systems: single-viewport only
481#endif
482
483 if (is_app_focused)
484 {
485 // (Optional) Set OS mouse position from Dear ImGui if requested (rarely used, only when ImGuiConfigFlags_NavEnableSetMousePos is enabled by user)
486 if (io.WantSetMousePos)
487 {
488#if SDL_HAS_CAPTURE_AND_GLOBAL_MOUSE
489 if (io.ConfigFlags & ImGuiConfigFlags_ViewportsEnable)
490 SDL_WarpMouseGlobal((int)io.MousePos.x, (int)io.MousePos.y);
491 else
492#endif
493 SDL_WarpMouseInWindow(bd->Window, (int)io.MousePos.x, (int)io.MousePos.y);
494 }
495
496 // (Optional) Fallback to provide mouse position when focused (SDL_MOUSEMOTION already provides this when hovered or captured)
497 if (bd->MouseCanUseGlobalState && bd->MouseButtonsDown == 0)
498 {
499 // Single-viewport mode: mouse position in client window coordinates (io.MousePos is (0,0) when the mouse is on the upper-left corner of the app window)
500 // Multi-viewport mode: mouse position in OS absolute coordinates (io.MousePos is (0,0) when the mouse is on the upper-left of the primary monitor)
501 int mouse_x, mouse_y, window_x, window_y;
502 SDL_GetGlobalMouseState(&mouse_x, &mouse_y);
503 if (!(io.ConfigFlags & ImGuiConfigFlags_ViewportsEnable))
504 {
505 SDL_GetWindowPosition(focused_window, &window_x, &window_y);
506 mouse_x -= window_x;
507 mouse_y -= window_y;
508 }
509 io.AddMousePosEvent((float)mouse_x, (float)mouse_y);
510 }
511 }
512
513 // We don't support ImGuiBackendFlags_HasMouseHoveredViewport
514 io.MouseHoveredViewport = 0;
515}

538static void ImGui_ImplSDL2_UpdateGamepads()
539{
540 ImGuiIO& io = ImGui::GetIO();
541 memset(io.NavInputs, 0, sizeof(io.NavInputs));
542 if ((io.ConfigFlags & ImGuiConfigFlags_NavEnableGamepad) == 0)
543 return;
544
545 // Get gamepad
546 SDL_GameController* game_controller = SDL_GameControllerOpen(0);
547 if (!game_controller)
548 {
549 io.BackendFlags &= ~ImGuiBackendFlags_HasGamepad;
550 return;
551 }
552
553 // Update gamepad inputs
554 #define MAP_BUTTON(NAV_NO, BUTTON_NO) { io.NavInputs[NAV_NO] = (SDL_GameControllerGetButton(game_controller, BUTTON_NO) != 0) ? 1.0f : 0.0f; }
555 #define MAP_ANALOG(NAV_NO, AXIS_NO, V0, V1) { float vn = (float)(SDL_GameControllerGetAxis(game_controller, AXIS_NO) - V0) / (float)(V1 - V0); if (vn > 1.0f) vn = 1.0f; if (vn > 0.0f && io.NavInputs[NAV_NO] < vn) io.NavInputs[NAV_NO] = vn; }
556 const int thumb_dead_zone = 8000; // SDL_gamecontroller.h suggests using this value.
557 MAP_BUTTON(ImGuiNavInput_Activate, SDL_CONTROLLER_BUTTON_A); // Cross / A
558 MAP_BUTTON(ImGuiNavInput_Cancel, SDL_CONTROLLER_BUTTON_B); // Circle / B
559 MAP_BUTTON(ImGuiNavInput_Menu, SDL_CONTROLLER_BUTTON_X); // Square / X
560 MAP_BUTTON(ImGuiNavInput_Input, SDL_CONTROLLER_BUTTON_Y); // Triangle / Y
561 MAP_BUTTON(ImGuiNavInput_DpadLeft, SDL_CONTROLLER_BUTTON_DPAD_LEFT); // D-Pad Left

563 MAP_BUTTON(ImGuiNavInput_DpadUp, SDL_CONTROLLER_BUTTON_DPAD_UP); // D-Pad Up
564 MAP_BUTTON(ImGuiNavInput_DpadDown, SDL_CONTROLLER_BUTTON_DPAD_DOWN); // D-Pad Down
565 MAP_BUTTON(ImGuiNavInput_FocusPrev, SDL_CONTROLLER_BUTTON_LEFTSHOULDER); // L1 / LB
566 MAP_BUTTON(ImGuiNavInput_FocusNext, SDL_CONTROLLER_BUTTON_RIGHTSHOULDER); // R1 / RB
567 MAP_BUTTON(ImGuiNavInput_TweakSlow, SDL_CONTROLLER_BUTTON_LEFTSHOULDER); // L1 / LB
568 MAP_BUTTON(ImGuiNavInput_TweakFast, SDL_CONTROLLER_BUTTON_RIGHTSHOULDER); // R1 / RB
569 MAP_ANALOG(ImGuiNavInput_LStickLeft, SDL_CONTROLLER_AXIS_LEFTX, -thumb_dead_zone, -32768);
570 MAP_ANALOG(ImGuiNavInput_LStickRight, SDL_CONTROLLER_AXIS_LEFTX, +thumb_dead_zone, +32767);
571 MAP_ANALOG(ImGuiNavInput_LStickUp, SDL_CONTROLLER_AXIS_LEFTY, -thumb_dead_zone, -32767);
572 MAP_ANALOG(ImGuiNavInput_LStickDown, SDL_CONTROLLER_AXIS_LEFTY, +thumb_dead_zone, +32767);
573
574 io.BackendFlags |= ImGuiBackendFlags_HasGamepad;
575 #undef MAP_BUTTON
576 #undef MAP_ANALOG
577}

581{

607void ImGui_ImplSDL2_NewFrame()
608{
609 ImGui_ImplSDL2_Data* bd = ImGui_ImplSDL2_GetBackendData();
610 IM_ASSERT(bd != NULL && "Did you call ImGui_ImplSDL2_Init()?");
611 ImGuiIO& io = ImGui::GetIO();
612
613 // Setup display size (every frame to accommodate for window resizing)
614 int w, h;
615 int display_w, display_h;
616 SDL_GetWindowSize(bd->Window, &w, &h);
617 if (SDL_GetWindowFlags(bd->Window) & SDL_WINDOW_MINIMIZED)
618 w = h = 0;
619 SDL_GL_GetDrawableSize(bd->Window, &display_w, &display_h);
620 io.DisplaySize = ImVec2((float)w, (float)h);
621 if (w > 0 && h > 0)
622 io.DisplayFramebufferScale = ImVec2((float)display_w / w, (float)display_h / h);
623
624 // Setup time step (we don't use SDL_GetTicks() because it is using millisecond resolution)
625 static Uint64 frequency = SDL_GetPerformanceFrequency();
626 Uint64 current_time = SDL_GetPerformanceCounter();
627 io.DeltaTime = bd->Time > 0 ? (float)((double)(current_time - bd->Time) / frequency) : (float)(1.0f / 60.0f);
628 bd->Time = current_time;
629
630 ImGui_ImplSDL2_UpdateMouseData();
631 ImGui_ImplSDL2_UpdateMouseCursor();
632
633 // Update game controllers (if enabled and available)
634 ImGui_ImplSDL2_UpdateGamepads();
635}

655static void ImGui_ImplSDL2_CreateWindow(ImGuiViewport* viewport)
656{
657 ImGui_ImplSDL2_Data* bd = ImGui_ImplSDL2_GetBackendData();
658 ImGui_ImplSDL2_ViewportData* vd = IM_NEW(ImGui_ImplSDL2_ViewportData)();
659 viewport->PlatformUserData = vd;
660
661 ImGuiViewport* main_viewport = ImGui::GetMainViewport();
662 ImGui_ImplSDL2_ViewportData* main_viewport_data = (ImGui_ImplSDL2_ViewportData*)main_viewport->PlatformUserData;
663
664 // Share GL resources with main context
665 bool use_opengl = (main_viewport_data->GLContext != NULL);
666 SDL_GLContext backup_context = NULL;
667 if (use_opengl)
668 {
669 backup_context = SDL_GL_GetCurrentContext();
670 SDL_GL_SetAttribute(SDL_GL_SHARE_WITH_CURRENT_CONTEXT, 1);
671 SDL_GL_MakeCurrent(main_viewport_data->Window, main_viewport_data->GLContext);
672 }
673
674 Uint32 sdl_flags = 0;
675 sdl_flags |= use_opengl ? SDL_WINDOW_OPENGL : (bd->UseVulkan ? SDL_WINDOW_VULKAN : 0);
676 sdl_flags |= SDL_GetWindowFlags(bd->Window) & SDL_WINDOW_ALLOW_HIGHDPI;
677 sdl_flags |= SDL_WINDOW_HIDDEN;
678 sdl_flags |= (viewport->Flags & ImGuiViewportFlags_NoDecoration) ? SDL_WINDOW_BORDERLESS : 0;
679 sdl_flags |= (viewport->Flags & ImGuiViewportFlags_NoDecoration) ? 0 : SDL_WINDOW_RESIZABLE;
680#if !defined(_WIN32)
681 // See SDL hack in ImGui_ImplSDL2_ShowWindow().
682 sdl_flags |= (viewport->Flags & ImGuiViewportFlags_NoTaskBarIcon) ? SDL_WINDOW_SKIP_TASKBAR : 0;
683#endif
684#if SDL_HAS_ALWAYS_ON_TOP
685 sdl_flags |= (viewport->Flags & ImGuiViewportFlags_TopMost) ? SDL_WINDOW_ALWAYS_ON_TOP : 0;
686#endif
687 vd->Window = SDL_CreateWindow("No Title Yet", (int)viewport->Pos.x, (int)viewport->Pos.y, (int)viewport->Size.x, (int)viewport->Size.y, sdl_flags);
688 vd->WindowOwned = true;
689 if (use_opengl)
690 {
691 vd->GLContext = SDL_GL_CreateContext(vd->Window);
692 SDL_GL_SetSwapInterval(0);
693 }
694 if (use_opengl && backup_context)
695 SDL_GL_MakeCurrent(vd->Window, backup_context);
696
697 viewport->PlatformHandle = (void*)vd->Window;
698#if defined(_WIN32)
699 SDL_SysWMinfo info;
700 SDL_VERSION(&info.version);
701 if (SDL_GetWindowWMInfo(vd->Window, &info))
702 viewport->PlatformHandleRaw = info.info.win.window;
703#endif
704}
