1// dear imgui: Platform Backend for GLFW
2// This needs to be used along with a Renderer (e.g. OpenGL3, Vulkan, WebGPU..)
3// (Info: GLFW is a cross-platform general purpose library for handling windows, inputs, OpenGL/Vulkan graphics context creation, etc.)
4// (Requires: GLFW 3.1+. Prefer GLFW 3.3+ for full feature support.)
5
6// Implemented features:
7// [X] Platform: Clipboard support.
8// [X] Platform: Keyboard support. Since 1.87 we are using the io.AddKeyEvent() function. Pass ImGuiKey values to all key functions e.g. ImGui::IsKeyPressed(ImGuiKey_Space). [Legacy GLFW_KEY_* values will also be supported unless IMGUI_DISABLE_OBSOLETE_KEYIO is set]
9// [X] Platform: Gamepad support. Enable with 'io.ConfigFlags |= ImGuiConfigFlags_NavEnableGamepad'.
10// [X] Platform: Mouse cursor shape and visibility. Disable with 'io.ConfigFlags |= ImGuiConfigFlags_NoMouseCursorChange' (note: the resizing cursors requires GLFW 3.4+).
11// [X] Platform: Multi-viewport support (multiple windows). Enable with 'io.ConfigFlags |= ImGuiConfigFlags_ViewportsEnable'.
12
13// Issues:
14// [ ] Platform: Multi-viewport support: ParentViewportID not honored, and so io.ConfigViewportsNoDefaultParent has no effect (minor).

127// Backend data stored in io.BackendPlatformUserData to allow support for multiple Dear ImGui contexts
128// It is STRONGLY preferred that you use docking branch with multi-viewports (== single Dear ImGui context + multiple windows) instead of multiple Dear ImGui contexts.
129// FIXME: multi-context support is not well tested and probably dysfunctional in this backend.
130// - Because glfwPollEvents() process all windows and some events may be called outside of it, you will need to register your own callbacks
131// (passing install_callbacks=false in ImGui_ImplGlfw_InitXXX functions), set the current dear imgui context and then call our callbacks.
132// - Otherwise we may need to store a GLFWWindow* -> ImGuiContext* map and handle this in the backend, adding a little bit of extra complexity to it.
133// FIXME: some shared resources (mouse cursor shape, gamepad) are mishandled when using multi-context.
134static ImGui_ImplGlfw_Data* ImGui_ImplGlfw_GetBackendData()
135{
136 return ImGui::GetCurrentContext() ? (ImGui_ImplGlfw_Data*)ImGui::GetIO().BackendPlatformUserData : NULL;
137}

302static int ImGui_ImplGlfw_TranslateUntranslatedKey(int key, int scancode)
303{
304#if GLFW_HAS_GET_KEY_NAME && !defined(__EMSCRIPTEN__)
305 // GLFW 3.1+ attempts to "untranslate" keys, which goes the opposite of what every other framework does, making using lettered shortcuts difficult.
306 // (It had reasons to do so: namely GLFW is/was more likely to be used for WASD-type game controls rather than lettered shortcuts, but IHMO the 3.1 change could have been done differently)
307 // See https://github.com/glfw/glfw/issues/1502 for details.
308 // Adding a workaround to undo this (so our keys are translated->untranslated->translated, likely a lossy process).
309 // This won't cover edge cases but this is at least going to cover common cases.
310 const char* key_name = glfwGetKeyName(key, scancode);
311 if (key_name && key_name[0] != 0 && key_name[1] == 0)
312 {
313 const char char_names[] = "'-=[]\\,;\'./";
314 const int char_keys[] = { GLFW_KEY_GRAVE_ACCENT, GLFW_KEY_MINUS, GLFW_KEY_EQUAL, GLFW_KEY_LEFT_BRACKET, GLFW_KEY_RIGHT_BRACKET, GLFW_KEY_BACKSLASH, GLFW_KEY_COMMA, GLFW_KEY_SEMICOLON, GLFW_KEY_APOSTROPHE, GLFW_KEY_PERIOD, GLFW_KEY_SLASH, 0 };
315 IM_ASSERT(IM_ARRAYSIZE(char_names) == IM_ARRAYSIZE(char_keys));
316 if (key_name[0] >= '0' && key_name[0] <= '9') { key = GLFW_KEY_0 + (key_name[0] - '0'); }
317 else if (key_name[0] >= 'A' && key_name[0] <= 'Z') { key = GLFW_KEY_A + (key_name[0] - 'A'); }
318 else if (const char* p = strchr(char_names, key_name[0])) { key = char_keys[p - char_names]; }
319 }
320 // if (action == GLFW_PRESS) printf("key %d scancode %d name '%s'\n", key, scancode, key_name);
321#else
322 IM_UNUSED(scancode);
323#endif
324 return key;
325}

327void ImGui_ImplGlfw_KeyCallback(GLFWwindow* window, int keycode, int scancode, int action, int mods)
328{
329 ImGui_ImplGlfw_Data* bd = ImGui_ImplGlfw_GetBackendData();
330 if (bd->PrevUserCallbackKey != NULL && window == bd->Window)
331 bd->PrevUserCallbackKey(window, keycode, scancode, action, mods);
332
333 if (action != GLFW_PRESS && action != GLFW_RELEASE)
334 return;
335
336 ImGui_ImplGlfw_UpdateKeyModifiers(mods);
337
338 if (keycode >= 0 && keycode < IM_ARRAYSIZE(bd->KeyOwnerWindows))
339 bd->KeyOwnerWindows[keycode] = (action == GLFW_PRESS) ? window : NULL;
340
341 keycode = ImGui_ImplGlfw_TranslateUntranslatedKey(keycode, scancode);
342
343 ImGuiIO& io = ImGui::GetIO();
344 ImGuiKey imgui_key = ImGui_ImplGlfw_KeyToImGuiKey(keycode);
345 io.AddKeyEvent(imgui_key, (action == GLFW_PRESS));
346 io.SetKeyEventNativeData(imgui_key, keycode, scancode); // To support legacy indexing (<1.87 user code)
347}

536static void ImGui_ImplGlfw_UpdateMouseData()
537{
538 ImGui_ImplGlfw_Data* bd = ImGui_ImplGlfw_GetBackendData();
539 ImGuiIO& io = ImGui::GetIO();
540 ImGuiPlatformIO& platform_io = ImGui::GetPlatformIO();
541
542 io.MouseHoveredViewport = 0;
543
544 const ImVec2 mouse_pos_prev = io.MousePos;
545 for (int n = 0; n < platform_io.Viewports.Size; n++)
546 {
547 ImGuiViewport* viewport = platform_io.Viewports[n];
548 GLFWwindow* window = (GLFWwindow*)viewport->PlatformHandle;
549
550#ifdef __EMSCRIPTEN__
551 const bool is_window_focused = true;
552#else
553 const bool is_window_focused = glfwGetWindowAttrib(window, GLFW_FOCUSED) != 0;
554#endif
555 if (is_window_focused)
556 {
557 // (Optional) Set OS mouse position from Dear ImGui if requested (rarely used, only when ImGuiConfigFlags_NavEnableSetMousePos is enabled by user)
558 // When multi-viewports are enabled, all Dear ImGui positions are same as OS positions.
559 if (io.WantSetMousePos)
560 glfwSetCursorPos(window, (double)(mouse_pos_prev.x - viewport->Pos.x), (double)(mouse_pos_prev.y - viewport->Pos.y));
561
562 // (Optional) Fallback to provide mouse position when focused (ImGui_ImplGlfw_CursorPosCallback already provides this when hovered or captured)
563 if (bd->MouseWindow == NULL)
564 {
565 double mouse_x, mouse_y;
566 glfwGetCursorPos(window, &mouse_x, &mouse_y);
567 if (io.ConfigFlags & ImGuiConfigFlags_ViewportsEnable)
568 {
569 // Single viewport mode: mouse position in client window coordinates (io.MousePos is (0,0) when the mouse is on the upper-left corner of the app window)
570 // Multi-viewport mode: mouse position in OS absolute coordinates (io.MousePos is (0,0) when the mouse is on the upper-left of the primary monitor)
571 int window_x, window_y;
572 glfwGetWindowPos(window, &window_x, &window_y);
573 mouse_x += window_x;
574 mouse_y += window_y;
575 }
576 io.AddMousePosEvent((float)mouse_x, (float)mouse_y);
577 }
578 }
579
580 // (Optional) When using multiple viewports: set io.MouseHoveredViewport to the viewport the OS mouse cursor is hovering.
581 // Important: this information is not easy to provide and many high-level windowing library won't be able to provide it correctly, because
582 // - This is _ignoring_ viewports with the ImGuiViewportFlags_NoInputs flag (pass-through windows).
583 // - This is _regardless_ of whether another viewport is focused or being dragged from.
584 // If ImGuiBackendFlags_HasMouseHoveredViewport is not set by the backend, imgui will ignore this field and infer the information by relying on the
585 // rectangles and last focused time of every viewports it knows about. It will be unaware of other windows that may be sitting between or over your windows.
586 // [GLFW] FIXME: This is currently only correct on Win32. See what we do below with the WM_NCHITTEST, missing an equivalent for other systems.
587 // See https://github.com/glfw/glfw/issues/1236 if you want to help in making this a GLFW feature.
588#if GLFW_HAS_MOUSE_PASSTHROUGH || (GLFW_HAS_WINDOW_HOVERED && defined(_WIN32))
589 const bool window_no_input = (viewport->Flags & ImGuiViewportFlags_NoInputs) != 0;
590#if GLFW_HAS_MOUSE_PASSTHROUGH
591 glfwSetWindowAttrib(window, GLFW_MOUSE_PASSTHROUGH, window_no_input);
592#endif
593 if (glfwGetWindowAttrib(window, GLFW_HOVERED) && !window_no_input)
594 io.MouseHoveredViewport = viewport->ID;
595#endif
596 }
597}

600{
601 ImGuiIO& io = ImGui::GetIO();
602 ImGui_ImplGlfw_Data* bd = ImGui_ImplGlfw_GetBackendData();

619 // FIXME-PLATFORM: Unfocused windows seems to fail changing the mouse cursor with GLFW 3.2, but 3.3 works here.
620 glfwSetCursor(window, bd->MouseCursors[imgui_cursor] ? bd->MouseCursors[imgui_cursor] : bd->MouseCursors[ImGuiMouseCursor_Arrow]);
621 glfwSetInputMode(window, GLFW_CURSOR, GLFW_CURSOR_NORMAL);
622 }

627{
628 ImGuiIO& io = ImGui::GetIO();
629 memset(io.NavInputs, 0, sizeof(io.NavInputs));
630 if ((io.ConfigFlags & ImGuiConfigFlags_NavEnableGamepad) == 0)

663static void ImGui_ImplGlfw_UpdateMonitors()
664{
665 ImGui_ImplGlfw_Data* bd = ImGui_ImplGlfw_GetBackendData();
666 ImGuiPlatformIO& platform_io = ImGui::GetPlatformIO();
667 int monitors_count = 0;
668 GLFWmonitor** glfw_monitors = glfwGetMonitors(&monitors_count);
669 platform_io.Monitors.resize(0);
670 for (int n = 0; n < monitors_count; n++)
671 {
672 ImGuiPlatformMonitor monitor;
673 int x, y;
674 glfwGetMonitorPos(glfw_monitors[n], &x, &y);
675 const GLFWvidmode* vid_mode = glfwGetVideoMode(glfw_monitors[n]);
676 monitor.MainPos = monitor.WorkPos = ImVec2((float)x, (float)y);
677 monitor.MainSize = monitor.WorkSize = ImVec2((float)vid_mode->width, (float)vid_mode->height);
678#if GLFW_HAS_MONITOR_WORK_AREA
679 int w, h;
680 glfwGetMonitorWorkarea(glfw_monitors[n], &x, &y, &w, &h);
681 if (w > 0 && h > 0) // Workaround a small GLFW issue reporting zero on monitor changes: https://github.com/glfw/glfw/pull/1761
682 {
683 monitor.WorkPos = ImVec2((float)x, (float)y);
684 monitor.WorkSize = ImVec2((float)w, (float)h);
685 }
686#endif
687#if GLFW_HAS_PER_MONITOR_DPI
688 // Warning: the validity of monitor DPI information on Windows depends on the application DPI awareness settings, which generally needs to be set in the manifest or at runtime.
689 float x_scale, y_scale;
690 glfwGetMonitorContentScale(glfw_monitors[n], &x_scale, &y_scale);
691 monitor.DpiScale = x_scale;
692#endif
693 platform_io.Monitors.push_back(monitor);
694 }
695 bd->WantUpdateMonitors = false;

698void ImGui_ImplGlfw_NewFrame()
699{
700 ImGuiIO& io = ImGui::GetIO();
701 ImGui_ImplGlfw_Data* bd = ImGui_ImplGlfw_GetBackendData();
702 IM_ASSERT(bd != NULL && "Did you call ImGui_ImplGlfw_InitForXXX()?");
703
704 // Setup display size (every frame to accommodate for window resizing)
705 int w, h;
706 int display_w, display_h;
707 glfwGetWindowSize(bd->Window, &w, &h);
708 glfwGetFramebufferSize(bd->Window, &display_w, &display_h);
709 io.DisplaySize = ImVec2((float)w, (float)h);
710 if (w > 0 && h > 0)
711 io.DisplayFramebufferScale = ImVec2((float)display_w / (float)w, (float)display_h / (float)h);
712 if (bd->WantUpdateMonitors)
713 ImGui_ImplGlfw_UpdateMonitors();
714

716 double current_time = glfwGetTime();

718 bd->Time = current_time;
719
720 ImGui_ImplGlfw_UpdateMouseData();
721 ImGui_ImplGlfw_UpdateMouseCursor();
722
723 // Update game controllers (if enabled and available)
724 ImGui_ImplGlfw_UpdateGamepads();
725}

857// We have submitted https://github.com/glfw/glfw/pull/1568 to allow GLFW to support "transparent inputs".
858// In the meanwhile we implement custom per-platform workarounds here (FIXME-VIEWPORT: Implement same work-around for Linux/OSX!)
