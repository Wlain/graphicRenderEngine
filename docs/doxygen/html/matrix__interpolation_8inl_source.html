117 GLM_FUNC_QUALIFIER mat<4, 4, T, Q> interpolate(mat<4, 4, T, Q> const& m1, mat<4, 4, T, Q> const& m2, T const delta)
118 {
119 mat<4, 4, T, Q> m1rot = extractMatrixRotation(m1);
120 mat<4, 4, T, Q> dltRotation = m2 * transpose(m1rot);

123 axisAngle(dltRotation, dltAxis, dltAngle);
124 mat<4, 4, T, Q> out = axisAngleMatrix(dltAxis, dltAngle * delta) * m1rot;

127 out[3][2] = m1[3][2] + delta * (m2[3][2] - m1[3][2]);
