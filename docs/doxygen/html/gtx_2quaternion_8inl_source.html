53 {
54 vec<3, T, Q> u(q.x, q.y, q.z);
55 T const Angle = glm::length(u);
56 if (Angle < epsilon<T>())
57 return tquat<T, Q>();
58
59 vec<3, T, Q> const v(u / Angle);
60 return tquat<T, Q>(cos(Angle), sin(Angle) * v);

63 template<typename T, qualifier Q>
64 GLM_FUNC_QUALIFIER tquat<T, Q> log(tquat<T, Q> const& q)
65 {
66 vec<3, T, Q> u(q.x, q.y, q.z);
67 T Vec3Len = length(u);
68
69 if (Vec3Len < epsilon<T>())
70 {
71 if(q.w > static_cast<T>(0))
72 return tquat<T, Q>(log(q.w), static_cast<T>(0), static_cast<T>(0), static_cast<T>(0));
73 else if(q.w < static_cast<T>(0))
74 return tquat<T, Q>(log(-q.w), pi<T>(), static_cast<T>(0), static_cast<T>(0));
75 else
76 return tquat<T, Q>(std::numeric_limits<T>::infinity(), std::numeric_limits<T>::infinity(), std::numeric_limits<T>::infinity(), std::numeric_limits<T>::infinity());
77 }
78 else
79 {
80 T t = atan(Vec3Len, T(q.w)) / Vec3Len;
81 T QuatLen2 = Vec3Len * Vec3Len + q.w * q.w;
82 return tquat<T, Q>(static_cast<T>(0.5) * log(QuatLen2), t * q.x, t * q.y, t * q.z);
83 }
84 }

86 template<typename T, qualifier Q>
87 GLM_FUNC_QUALIFIER tquat<T, Q> pow(tquat<T, Q> const& x, T const& y)
88 {
89 //Raising to the power of 0 should yield 1
90 //Needed to prevent a division by 0 error later on
91 if(y > -epsilon<T>() && y < epsilon<T>())
92 return tquat<T, Q>(1,0,0,0);
93
94 //To deal with non-unit quaternions
95 T magnitude = sqrt(x.x * x.x + x.y * x.y + x.z * x.z + x.w *x.w);
96
97 //Equivalent to raising a real number to a power
98 //Needed to prevent a division by 0 error later on
99 if(abs(x.w / magnitude) > static_cast<T>(1) - epsilon<T>() && abs(x.w / magnitude) < static_cast<T>(1) + epsilon<T>())
100 return tquat<T, Q>(pow(x.w, y),0,0,0);
101
102 T Angle = acos(x.w / magnitude);
103 T NewAngle = Angle * y;
104 T Div = sin(NewAngle) / sin(Angle);
105 T Mag = pow(magnitude, y - static_cast<T>(1));
106
107 return tquat<T, Q>(cos(NewAngle) * magnitude * Mag, x.x * Div * Mag, x.y * Div * Mag, x.z * Div * Mag);
108 }

138 template<typename T, qualifier Q>
139 GLM_FUNC_QUALIFIER tquat<T, Q> shortMix(tquat<T, Q> const& x, tquat<T, Q> const& y, T const& a)
140 {
141 if(a <= static_cast<T>(0)) return x;
142 if(a >= static_cast<T>(1)) return y;
143
144 T fCos = dot(x, y);
145 tquat<T, Q> y2(y); //BUG!!! tquat<T> y2;
146 if(fCos < static_cast<T>(0))
147 {
148 y2 = -y;
149 fCos = -fCos;
150 }
151
152 //if(fCos > 1.0f) // problem
153 T k0, k1;
154 if(fCos > (static_cast<T>(1) - epsilon<T>()))
155 {
156 k0 = static_cast<T>(1) - a;
157 k1 = static_cast<T>(0) + a; //BUG!!! 1.0f + a;
158 }
159 else
160 {
161 T fSin = sqrt(T(1) - fCos * fCos);
162 T fAngle = atan(fSin, fCos);
163 T fOneOverSin = static_cast<T>(1) / fSin;
164 k0 = sin((static_cast<T>(1) - a) * fAngle) * fOneOverSin;
165 k1 = sin((static_cast<T>(0) + a) * fAngle) * fOneOverSin;
166 }
167
168 return tquat<T, Q>(
169 k0 * x.w + k1 * y2.w,
170 k0 * x.x + k1 * y2.x,
171 k0 * x.y + k1 * y2.y,
172 k0 * x.z + k1 * y2.z);
173 }

181 template<typename T, qualifier Q>
182 GLM_FUNC_QUALIFIER tquat<T, Q> rotation(vec<3, T, Q> const& orig, vec<3, T, Q> const& dest)
183 {
184 T cosTheta = dot(orig, dest);
185 vec<3, T, Q> rotationAxis;
186
187 if(cosTheta >= static_cast<T>(1) - epsilon<T>()) {
188 // orig and dest point in the same direction
189 return quat_identity<T,Q>();
190 }
191
192 if(cosTheta < static_cast<T>(-1) + epsilon<T>())
193 {
194 // special case when vectors in opposite directions :
195 // there is no "ideal" rotation axis
196 // So guess one; any will do as long as it's perpendicular to start
197 // This implementation favors a rotation around the Up axis (Y),
198 // since it's often what you want to do.
199 rotationAxis = cross(vec<3, T, Q>(0, 0, 1), orig);
200 if(length2(rotationAxis) < epsilon<T>()) // bad luck, they were parallel, try again!
201 rotationAxis = cross(vec<3, T, Q>(1, 0, 0), orig);
202
203 rotationAxis = normalize(rotationAxis);
204 return angleAxis(pi<T>(), rotationAxis);
205 }
206
207 // Implementation from Stan Melax's Game Programming Gems 1 article
208 rotationAxis = cross(orig, dest);
209
210 T s = sqrt((T(1) + cosTheta) * static_cast<T>(2));
211 T invs = static_cast<T>(1) / s;
212
213 return tquat<T, Q>(
214 s * static_cast<T>(0.5f),
215 rotationAxis.x * invs,
216 rotationAxis.y * invs,
217 rotationAxis.z * invs);
218 }

231 GLM_FUNC_QUALIFIER tquat<T, Q> quatLookAtRH(vec<3, T, Q> const& direction, vec<3, T, Q> const& up)
232 {
233 mat<3, 3, T, Q> Result;
234
235 Result[2] = -direction;
236 Result[0] = normalize(cross(up, Result[2]));
237 Result[1] = cross(Result[2], Result[0]);
238
239 return quat_cast(Result);
