15// You can use unmodified imgui_impl_* files in your project. See examples/ folder for examples of using this.

166int32_t ImGui_ImplAndroid_HandleInputEvent(AInputEvent* input_event)
167{
168 ImGuiIO& io = ImGui::GetIO();
169 int32_t event_type = AInputEvent_getType(input_event);
170 switch (event_type)
171 {
172 case AINPUT_EVENT_TYPE_KEY:
173 {
174 int32_t event_key_code = AKeyEvent_getKeyCode(input_event);
175 int32_t event_scan_code = AKeyEvent_getScanCode(input_event);
176 int32_t event_action = AKeyEvent_getAction(input_event);
177 int32_t event_meta_state = AKeyEvent_getMetaState(input_event);
178
179 g_KeyModFlags = ImGuiKeyModFlags_None;
180 if ((event_meta_state & AMETA_CTRL_ON) != 0)
181 g_KeyModFlags |= ImGuiKeyModFlags_Ctrl;
182 if ((event_meta_state & AMETA_SHIFT_ON) != 0)
183 g_KeyModFlags |= ImGuiKeyModFlags_Shift;
184 if ((event_meta_state & AMETA_ALT_ON) != 0)
185 g_KeyModFlags |= ImGuiKeyModFlags_Alt;
186 if ((event_meta_state & AMETA_META_ON) != 0)
187 g_KeyModFlags |= ImGuiKeyModFlags_Super;
188
189 switch (event_action)
190 {
191 // FIXME: AKEY_EVENT_ACTION_DOWN and AKEY_EVENT_ACTION_UP occur at once as soon as a touch pointer
192 // goes up from a key. We use a simple key event queue/ and process one event per key per frame in
193 // ImGui_ImplAndroid_NewFrame()...or consider using IO queue, if suitable: https://github.com/ocornut/imgui/issues/2787
194 case AKEY_EVENT_ACTION_DOWN:
195 case AKEY_EVENT_ACTION_UP:
196 {
197 ImGuiKey key = ImGui_ImplAndroid_KeyCodeToImGuiKey(event_key_code);
198 if (key != ImGuiKey_None && (event_action == AKEY_EVENT_ACTION_DOWN || event_action == AKEY_EVENT_ACTION_UP))
199 {
200 KeyEvent io_event;
201 io_event.Key = key;
202 io_event.Down = event_action == AKEY_EVENT_ACTION_DOWN;
203 io_event.NativeKeycode = event_key_code;
204 io_event.NativeScancode = event_scan_code;
205
206 g_KeyEventQueues[key].push(io_event);
207 }
208
209 break;
210 }
211 default:
212 break;
213 }
214 break;
215 }
216 case AINPUT_EVENT_TYPE_MOTION:
217 {
218 int32_t event_action = AMotionEvent_getAction(input_event);
219 int32_t event_pointer_index = (event_action & AMOTION_EVENT_ACTION_POINTER_INDEX_MASK) >> AMOTION_EVENT_ACTION_POINTER_INDEX_SHIFT;
220 event_action &= AMOTION_EVENT_ACTION_MASK;
221 switch (event_action)
222 {
223 case AMOTION_EVENT_ACTION_DOWN:
224 case AMOTION_EVENT_ACTION_UP:
225 // Physical mouse buttons (and probably other physical devices) also invoke the actions AMOTION_EVENT_ACTION_DOWN/_UP,
226 // but we have to process them separately to identify the actual button pressed. This is done below via
227 // AMOTION_EVENT_ACTION_BUTTON_PRESS/_RELEASE. Here, we only process "FINGER" input (and "UNKNOWN", as a fallback).
228 if((AMotionEvent_getToolType(input_event, event_pointer_index) == AMOTION_EVENT_TOOL_TYPE_FINGER)
229 || (AMotionEvent_getToolType(input_event, event_pointer_index) == AMOTION_EVENT_TOOL_TYPE_UNKNOWN))
230 {
231 io.AddMousePosEvent(AMotionEvent_getX(input_event, event_pointer_index), AMotionEvent_getY(input_event, event_pointer_index));
232 io.AddMouseButtonEvent(0, event_action == AMOTION_EVENT_ACTION_DOWN);
233 }
234 break;
235 case AMOTION_EVENT_ACTION_BUTTON_PRESS:
236 case AMOTION_EVENT_ACTION_BUTTON_RELEASE:
237 {
238 int32_t button_state = AMotionEvent_getButtonState(input_event);
239 io.AddMouseButtonEvent(0, (button_state & AMOTION_EVENT_BUTTON_PRIMARY) != 0);
240 io.AddMouseButtonEvent(1, (button_state & AMOTION_EVENT_BUTTON_SECONDARY) != 0);
241 io.AddMouseButtonEvent(2, (button_state & AMOTION_EVENT_BUTTON_TERTIARY) != 0);
242 }
243 break;
244 case AMOTION_EVENT_ACTION_HOVER_MOVE: // Hovering: Tool moves while NOT pressed (such as a physical mouse)
245 case AMOTION_EVENT_ACTION_MOVE: // Touch pointer moves while DOWN
246 io.AddMousePosEvent(AMotionEvent_getX(input_event, event_pointer_index), AMotionEvent_getY(input_event, event_pointer_index));
247 break;
248 case AMOTION_EVENT_ACTION_SCROLL:
249 io.AddMouseWheelEvent(AMotionEvent_getAxisValue(input_event, AMOTION_EVENT_AXIS_HSCROLL, event_pointer_index), AMotionEvent_getAxisValue(input_event, AMOTION_EVENT_AXIS_VSCROLL, event_pointer_index));
250 break;
251 default:
252 break;
253 }
254 }
255 return 1;
256 default:
257 break;
258 }
259
260 return 0;
261}

279void ImGui_ImplAndroid_NewFrame()
280{
281 ImGuiIO& io = ImGui::GetIO();
282
283 // Process queued key events
284 // FIXME: This is a workaround for multiple key event actions occurring at once (see above) and can be removed once we use upcoming input queue.
285 for (auto& key_queue : g_KeyEventQueues)
286 {
287 if (key_queue.second.empty())
288 continue;
289
290 auto& key_event = key_queue.second.front();
291 io.AddKeyEvent(key_event.Key, key_event.Down);
292 io.SetKeyEventNativeData(key_event.Key, key_event.NativeKeycode, key_event.NativeScancode); // To support legacy indexing (<1.87 user code)
293 key_queue.second.pop();
294 }
295 io.AddKeyModsEvent(g_KeyModFlags);
296
297 // Setup display size (every frame to accommodate for window resizing)
298 int32_t window_width = ANativeWindow_getWidth(g_Window);
299 int32_t window_height = ANativeWindow_getHeight(g_Window);
300 int display_width = window_width;
301 int display_height = window_height;
302
303 io.DisplaySize = ImVec2((float)window_width, (float)window_height);
304 if (window_width > 0 && window_height > 0)
305 io.DisplayFramebufferScale = ImVec2((float)display_width / window_width, (float)display_height / window_height);
306
307 // Setup time step
308 struct timespec current_timespec;
309 clock_gettime(CLOCK_MONOTONIC, &current_timespec);
310 double current_time = (double)(current_timespec.tv_sec) + (current_timespec.tv_nsec / 1000000000.0);
311 io.DeltaTime = g_Time > 0.0 ? (float)(current_time - g_Time) : (float)(1.0f / 60.0f);
312 g_Time = current_time;
313}
