164static int stb__Mul8Bit(int a, int b)
165{
166 int t = a*b + 128;
167 return (t + (t >> 8)) >> 8;
168}

170static void stb__From16Bit(unsigned char *out, unsigned short v)
171{
172 int rv = (v & 0xf800) >> 11;
173 int gv = (v & 0x07e0) >> 5;
174 int bv = (v & 0x001f) >> 0;
175
176 // expand to 8 bits via bit replication
177 out[0] = (rv * 33) >> 2;
178 out[1] = (gv * 65) >> 4;
179 out[2] = (bv * 33) >> 2;
180 out[3] = 0;
181}

183static unsigned short stb__As16Bit(int r, int g, int b)
184{
185 return (unsigned short)((stb__Mul8Bit(r,31) << 11) + (stb__Mul8Bit(g,63) << 5) + stb__Mul8Bit(b,31));
186}

188// linear interpolation at 1/3 point between a and b, using desired rounding type
189static int stb__Lerp13(int a, int b)
190{
191#ifdef STB_DXT_USE_ROUNDING_BIAS
192 // with rounding bias
193 return a + stb__Mul8Bit(b-a, 0x55);
194#else
195 // without rounding bias
196 // replace "/ 3" by "* 0xaaab) >> 17" if your compiler sucks or you really need every ounce of speed.
197 return (2*a + b) / 3;
198#endif
199}

219// The color matching function
220static unsigned int stb__MatchColorsBlock(unsigned char *block, unsigned char *color)
221{
222 unsigned int mask = 0;
223 int dirr = color[0*4+0] - color[1*4+0];
224 int dirg = color[0*4+1] - color[1*4+1];
225 int dirb = color[0*4+2] - color[1*4+2];
226 int dots[16];
227 int stops[4];
228 int i;
229 int c0Point, halfPoint, c3Point;
230
231 for(i=0;i<16;i++)
232 dots[i] = block[i*4+0]*dirr + block[i*4+1]*dirg + block[i*4+2]*dirb;
233
234 for(i=0;i<4;i++)
235 stops[i] = color[i*4+0]*dirr + color[i*4+1]*dirg + color[i*4+2]*dirb;
236
237 // think of the colors as arranged on a line; project point onto that line, then choose
238 // next color out of available ones. we compute the crossover points for "best color in top
239 // half"/"best in bottom half" and then the same inside that subinterval.
240 //
241 // relying on this 1d approximation isn't always optimal in terms of euclidean distance,
242 // but it's very close and a lot faster.
243 // http://cbloomrants.blogspot.com/2008/12/12-08-08-dxtc-summary.html
244
245 c0Point = (stops[1] + stops[3]);
246 halfPoint = (stops[3] + stops[2]);
247 c3Point = (stops[2] + stops[0]);
248
249 for (i=15;i>=0;i--) {
250 int dot = dots[i]*2;
251 mask <<= 2;
252
253 if(dot < halfPoint)
254 mask |= (dot < c0Point) ? 1 : 3;
255 else
256 mask |= (dot < c3Point) ? 2 : 0;
257 }
258
259 return mask;
260}

262// The color optimization function. (Clever code, part 1)
263static void stb__OptimizeColorsBlock(unsigned char *block, unsigned short *pmax16, unsigned short *pmin16)
264{
265 int mind,maxd;
266 unsigned char *minp, *maxp;
267 double magn;
268 int v_r,v_g,v_b;
269 static const int nIterPower = 4;
270 float covf[6],vfr,vfg,vfb;
271
272 // determine color distribution
273 int cov[6];
274 int mu[3],min[3],max[3];
275 int ch,i,iter;
276
277 for(ch=0;ch<3;ch++)
278 {
279 const unsigned char *bp = ((const unsigned char *) block) + ch;
280 int muv,minv,maxv;
281
282 muv = minv = maxv = bp[0];
283 for(i=4;i<64;i+=4)
284 {
285 muv += bp[i];
286 if (bp[i] < minv) minv = bp[i];
287 else if (bp[i] > maxv) maxv = bp[i];
288 }
289
290 mu[ch] = (muv + 8) >> 4;
291 min[ch] = minv;
292 max[ch] = maxv;
293 }
294
295 // determine covariance matrix
296 for (i=0;i<6;i++)
297 cov[i] = 0;
298
299 for (i=0;i<16;i++)
300 {
301 int r = block[i*4+0] - mu[0];
302 int g = block[i*4+1] - mu[1];
303 int b = block[i*4+2] - mu[2];
304
305 cov[0] += r*r;
306 cov[1] += r*g;
307 cov[2] += r*b;
308 cov[3] += g*g;
309 cov[4] += g*b;
310 cov[5] += b*b;
311 }
312
313 // convert covariance matrix to float, find principal axis via power iter
314 for(i=0;i<6;i++)
315 covf[i] = cov[i] / 255.0f;
316
317 vfr = (float) (max[0] - min[0]);
318 vfg = (float) (max[1] - min[1]);
319 vfb = (float) (max[2] - min[2]);
320
321 for(iter=0;iter<nIterPower;iter++)
322 {
323 float r = vfr*covf[0] + vfg*covf[1] + vfb*covf[2];
324 float g = vfr*covf[1] + vfg*covf[3] + vfb*covf[4];
325 float b = vfr*covf[2] + vfg*covf[4] + vfb*covf[5];
326
327 vfr = r;
328 vfg = g;
329 vfb = b;
330 }
331
332 magn = STBD_FABS(vfr);
333 if (STBD_FABS(vfg) > magn) magn = STBD_FABS(vfg);
334 if (STBD_FABS(vfb) > magn) magn = STBD_FABS(vfb);
335
336 if(magn < 4.0f) { // too small, default to luminance
337 v_r = 299; // JPEG YCbCr luma coefs, scaled by 1000.
338 v_g = 587;
339 v_b = 114;
340 } else {
341 magn = 512.0 / magn;
342 v_r = (int) (vfr * magn);
343 v_g = (int) (vfg * magn);
344 v_b = (int) (vfb * magn);
345 }
346
347 minp = maxp = block;
348 mind = maxd = block[0]*v_r + block[1]*v_g + block[2]*v_b;
349 // Pick colors at extreme points
350 for(i=1;i<16;i++)
351 {
352 int dot = block[i*4+0]*v_r + block[i*4+1]*v_g + block[i*4+2]*v_b;
353
354 if (dot < mind) {
355 mind = dot;
356 minp = block+i*4;
357 }
358
359 if (dot > maxd) {
360 maxd = dot;
361 maxp = block+i*4;
362 }
363 }
364
365 *pmax16 = stb__As16Bit(maxp[0],maxp[1],maxp[2]);
366 *pmin16 = stb__As16Bit(minp[0],minp[1],minp[2]);
367}

399// The refinement function. (Clever code, part 2)
400// Tries to optimize colors to suit block contents better.
401// (By solving a least squares system via normal equations+Cramer's rule)
402static int stb__RefineBlock(unsigned char *block, unsigned short *pmax16, unsigned short *pmin16, unsigned int mask)
403{
404 static const int w1Tab[4] = { 3,0,2,1 };
405 static const int prods[4] = { 0x090000,0x000900,0x040102,0x010402 };
406 // ^some magic to save a lot of multiplies in the accumulating loop...
407 // (precomputed products of weights for least squares system, accumulated inside one 32-bit register)
408
409 float f;
410 unsigned short oldMin, oldMax, min16, max16;
411 int i, akku = 0, xx,xy,yy;
412 int At1_r,At1_g,At1_b;
413 int At2_r,At2_g,At2_b;
414 unsigned int cm = mask;
415
416 oldMin = *pmin16;
417 oldMax = *pmax16;
418
419 if((mask ^ (mask<<2)) < 4) // all pixels have the same index?
420 {
421 // yes, linear system would be singular; solve using optimal
422 // single-color match on average color
423 int r = 8, g = 8, b = 8;
424 for (i=0;i<16;++i) {
425 r += block[i*4+0];
426 g += block[i*4+1];
427 b += block[i*4+2];
428 }
429
430 r >>= 4; g >>= 4; b >>= 4;
431
432 max16 = (stb__OMatch5[r][0]<<11) | (stb__OMatch6[g][0]<<5) | stb__OMatch5[b][0];
433 min16 = (stb__OMatch5[r][1]<<11) | (stb__OMatch6[g][1]<<5) | stb__OMatch5[b][1];
434 } else {
435 At1_r = At1_g = At1_b = 0;
436 At2_r = At2_g = At2_b = 0;
437 for (i=0;i<16;++i,cm>>=2) {
438 int step = cm&3;
439 int w1 = w1Tab[step];
440 int r = block[i*4+0];
441 int g = block[i*4+1];
442 int b = block[i*4+2];
443
444 akku += prods[step];
445 At1_r += w1*r;
446 At1_g += w1*g;
447 At1_b += w1*b;
448 At2_r += r;
449 At2_g += g;
450 At2_b += b;
451 }
452
453 At2_r = 3*At2_r - At1_r;
454 At2_g = 3*At2_g - At1_g;
455 At2_b = 3*At2_b - At1_b;
456
457 // extract solutions and decide solvability
458 xx = akku >> 16;
459 yy = (akku >> 8) & 0xff;
460 xy = (akku >> 0) & 0xff;
461
462 f = 3.0f / 255.0f / (xx*yy - xy*xy);
463
464 max16 = stb__Quantize5((At1_r*yy - At2_r * xy) * f) << 11;
465 max16 |= stb__Quantize6((At1_g*yy - At2_g * xy) * f) << 5;
466 max16 |= stb__Quantize5((At1_b*yy - At2_b * xy) * f) << 0;
467
468 min16 = stb__Quantize5((At2_r*xx - At1_r * xy) * f) << 11;
469 min16 |= stb__Quantize6((At2_g*xx - At1_g * xy) * f) << 5;
470 min16 |= stb__Quantize5((At2_b*xx - At1_b * xy) * f) << 0;
471 }
472
473 *pmin16 = min16;
474 *pmax16 = max16;
475 return oldMin != min16 || oldMax != max16;
476}

478// Color block compression
479static void stb__CompressColorBlock(unsigned char *dest, unsigned char *block, int mode)
480{
481 unsigned int mask;
482 int i;
483 int refinecount;
484 unsigned short max16, min16;
485 unsigned char color[4*4];
486
487 refinecount = (mode & STB_DXT_HIGHQUAL) ? 2 : 1;
488
489 // check if block is constant
490 for (i=1;i<16;i++)
491 if (((unsigned int *) block)[i] != ((unsigned int *) block)[0])
492 break;
493
494 if(i == 16) { // constant color
495 int r = block[0], g = block[1], b = block[2];
496 mask = 0xaaaaaaaa;
497 max16 = (stb__OMatch5[r][0]<<11) | (stb__OMatch6[g][0]<<5) | stb__OMatch5[b][0];
498 min16 = (stb__OMatch5[r][1]<<11) | (stb__OMatch6[g][1]<<5) | stb__OMatch5[b][1];
499 } else {
500 // first step: PCA+map along principal axis
501 stb__OptimizeColorsBlock(block,&max16,&min16);
502 if (max16 != min16) {
503 stb__EvalColors(color,max16,min16);
504 mask = stb__MatchColorsBlock(block,color);
505 } else
506 mask = 0;
507
508 // third step: refine (multiple times if requested)
509 for (i=0;i<refinecount;i++) {
510 unsigned int lastmask = mask;
511
512 if (stb__RefineBlock(block,&max16,&min16,mask)) {
513 if (max16 != min16) {
514 stb__EvalColors(color,max16,min16);
515 mask = stb__MatchColorsBlock(block,color);
516 } else {
517 mask = 0;
518 break;
519 }
520 }
521
522 if(mask == lastmask)
523 break;
524 }
525 }
526
527 // write the color block
528 if(max16 < min16)
529 {
530 unsigned short t = min16;
531 min16 = max16;
532 max16 = t;
533 mask ^= 0x55555555;
534 }
535
536 dest[0] = (unsigned char) (max16);
537 dest[1] = (unsigned char) (max16 >> 8);
538 dest[2] = (unsigned char) (min16);
539 dest[3] = (unsigned char) (min16 >> 8);
540 dest[4] = (unsigned char) (mask);
541 dest[5] = (unsigned char) (mask >> 8);
542 dest[6] = (unsigned char) (mask >> 16);
543 dest[7] = (unsigned char) (mask >> 24);
544}

546// Alpha block compression (this is easy for a change)
547static void stb__CompressAlphaBlock(unsigned char *dest,unsigned char *src, int stride)
548{
549 int i,dist,bias,dist4,dist2,bits,mask;
550
551 // find min/max color
552 int mn,mx;
553 mn = mx = src[0];
554
555 for (i=1;i<16;i++)
556 {
557 if (src[i*stride] < mn) mn = src[i*stride];
558 else if (src[i*stride] > mx) mx = src[i*stride];
559 }
560
561 // encode them
562 dest[0] = (unsigned char)mx;
563 dest[1] = (unsigned char)mn;
564 dest += 2;
565
566 // determine bias and emit color indices
567 // given the choice of mx/mn, these indices are optimal:
568 // http://fgiesen.wordpress.com/2009/12/15/dxt5-alpha-block-index-determination/
569 dist = mx-mn;
570 dist4 = dist*4;
571 dist2 = dist*2;
572 bias = (dist < 8) ? (dist - 1) : (dist/2 + 2);
573 bias -= mn * 7;
574 bits = 0,mask=0;
575
576 for (i=0;i<16;i++) {
577 int a = src[i*stride]*7 + bias;
578 int ind,t;
579
580 // select index. this is a "linear scale" lerp factor between 0 (val=min) and 7 (val=max).
581 t = (a >= dist4) ? -1 : 0; ind = t & 4; a -= dist4 & t;
582 t = (a >= dist2) ? -1 : 0; ind += t & 2; a -= dist2 & t;
583 ind += (a >= dist);
584
585 // turn linear scale into DXT index (0/1 are extremal pts)
586 ind = -ind & 7;
587 ind ^= (2 > ind);
588
589 // write index
590 mask |= ind << bits;
591 if((bits += 3) >= 8) {
592 *dest++ = (unsigned char)mask;
593 mask >>= 8;
594 bits -= 8;
595 }
596 }
597}
