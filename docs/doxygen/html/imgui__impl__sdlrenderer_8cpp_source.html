57bool ImGui_ImplSDLRenderer_Init(SDL_Renderer* renderer)
58{
59 ImGuiIO& io = ImGui::GetIO();
60 IM_ASSERT(io.BackendRendererUserData == NULL && "Already initialized a renderer backend!");
61 IM_ASSERT(renderer != NULL && "SDL_Renderer not initialized!");
62
63 // Setup backend capabilities flags
64 ImGui_ImplSDLRenderer_Data* bd = IM_NEW(ImGui_ImplSDLRenderer_Data)();
65 io.BackendRendererUserData = (void*)bd;
66 io.BackendRendererName = "imgui_impl_sdlrenderer";
67 io.BackendFlags |= ImGuiBackendFlags_RendererHasVtxOffset; // We can honor the ImDrawCmd::VtxOffset field, allowing for large meshes.
68
69 bd->SDLRenderer = renderer;
70
71 return true;
72}

106void ImGui_ImplSDLRenderer_RenderDrawData(ImDrawData* draw_data)
107{
108 ImGui_ImplSDLRenderer_Data* bd = ImGui_ImplSDLRenderer_GetBackendData();
109
110 // If there's a scale factor set by the user, use that instead
111 // If the user has specified a scale factor to SDL_Renderer already via SDL_RenderSetScale(), SDL will scale whatever we pass
112 // to SDL_RenderGeometryRaw() by that scale factor. In that case we don't want to be also scaling it ourselves here.
113 float rsx = 1.0f;
114 float rsy = 1.0f;
115 SDL_RenderGetScale(bd->SDLRenderer, &rsx, &rsy);
116 ImVec2 render_scale;
117 render_scale.x = (rsx == 1.0f) ? draw_data->FramebufferScale.x : 1.0f;
118 render_scale.y = (rsy == 1.0f) ? draw_data->FramebufferScale.y : 1.0f;
119
120 // Avoid rendering when minimized, scale coordinates for retina displays (screen coordinates != framebuffer coordinates)
121 int fb_width = (int)(draw_data->DisplaySize.x * render_scale.x);
122 int fb_height = (int)(draw_data->DisplaySize.y * render_scale.y);
123 if (fb_width == 0 || fb_height == 0)
124 return;
125
126 // Backup SDL_Renderer state that will be modified to restore it afterwards
127 struct BackupSDLRendererState
128 {
129 SDL_Rect Viewport;
130 bool ClipEnabled;
131 SDL_Rect ClipRect;
132 };
133 BackupSDLRendererState old = {};
134 old.ClipEnabled = SDL_RenderIsClipEnabled(bd->SDLRenderer) == SDL_TRUE;
135 SDL_RenderGetViewport(bd->SDLRenderer, &old.Viewport);
136 SDL_RenderGetClipRect(bd->SDLRenderer, &old.ClipRect);
137
138 // Will project scissor/clipping rectangles into framebuffer space
139 ImVec2 clip_off = draw_data->DisplayPos; // (0,0) unless using multi-viewports
140 ImVec2 clip_scale = render_scale;
141
142 // Render command lists
143 ImGui_ImplSDLRenderer_SetupRenderState();
144 for (int n = 0; n < draw_data->CmdListsCount; n++)
145 {
146 const ImDrawList* cmd_list = draw_data->CmdLists[n];
147 const ImDrawVert* vtx_buffer = cmd_list->VtxBuffer.Data;
148 const ImDrawIdx* idx_buffer = cmd_list->IdxBuffer.Data;
149
150 for (int cmd_i = 0; cmd_i < cmd_list->CmdBuffer.Size; cmd_i++)
151 {
152 const ImDrawCmd* pcmd = &cmd_list->CmdBuffer[cmd_i];
153 if (pcmd->UserCallback)
154 {
155 // User callback, registered via ImDrawList::AddCallback()
156 // (ImDrawCallback_ResetRenderState is a special callback value used by the user to request the renderer to reset render state.)
157 if (pcmd->UserCallback == ImDrawCallback_ResetRenderState)
158 ImGui_ImplSDLRenderer_SetupRenderState();
159 else
160 pcmd->UserCallback(cmd_list, pcmd);
161 }
162 else
163 {
164 // Project scissor/clipping rectangles into framebuffer space
165 ImVec2 clip_min((pcmd->ClipRect.x - clip_off.x) * clip_scale.x, (pcmd->ClipRect.y - clip_off.y) * clip_scale.y);
166 ImVec2 clip_max((pcmd->ClipRect.z - clip_off.x) * clip_scale.x, (pcmd->ClipRect.w - clip_off.y) * clip_scale.y);
167 if (clip_min.x < 0.0f) { clip_min.x = 0.0f; }
168 if (clip_min.y < 0.0f) { clip_min.y = 0.0f; }
169 if (clip_max.x > fb_width) { clip_max.x = (float)fb_width; }
170 if (clip_max.y > fb_height) { clip_max.y = (float)fb_height; }
171 if (clip_max.x <= clip_min.x || clip_max.y <= clip_min.y)
172 continue;
173
174 SDL_Rect r = { (int)(clip_min.x), (int)(clip_min.y), (int)(clip_max.x - clip_min.x), (int)(clip_max.y - clip_min.y) };
175 SDL_RenderSetClipRect(bd->SDLRenderer, &r);
176
177 const float* xy = (const float*)((const char*)(vtx_buffer + pcmd->VtxOffset) + IM_OFFSETOF(ImDrawVert, pos));
178 const float* uv = (const float*)((const char*)(vtx_buffer + pcmd->VtxOffset) + IM_OFFSETOF(ImDrawVert, uv));
179#if SDL_VERSION_ATLEAST(2,0,19)
180 const SDL_Color* color = (const SDL_Color*)((const char*)(vtx_buffer + pcmd->VtxOffset) + IM_OFFSETOF(ImDrawVert, col)); // SDL 2.0.19+
181#else
182 const int* color = (const int*)((const char*)(vtx_buffer + pcmd->VtxOffset) + IM_OFFSETOF(ImDrawVert, col)); // SDL 2.0.17 and 2.0.18
183#endif
184
185 // Bind texture, Draw
186 SDL_Texture* tex = (SDL_Texture*)pcmd->GetTexID();
187 SDL_RenderGeometryRaw(bd->SDLRenderer, tex,
188 xy, (int)sizeof(ImDrawVert),
189 color, (int)sizeof(ImDrawVert),
190 uv, (int)sizeof(ImDrawVert),
191 cmd_list->VtxBuffer.Size - pcmd->VtxOffset,
192 idx_buffer + pcmd->IdxOffset, pcmd->ElemCount, sizeof(ImDrawIdx));
193 }
194 }
195 }
196
197 // Restore modified SDL_Renderer state
198 SDL_RenderSetViewport(bd->SDLRenderer, &old.Viewport);
199 SDL_RenderSetClipRect(bd->SDLRenderer, old.ClipEnabled ? &old.ClipRect : NULL);
200}

203bool ImGui_ImplSDLRenderer_CreateFontsTexture()
204{
205 ImGuiIO& io = ImGui::GetIO();
206 ImGui_ImplSDLRenderer_Data* bd = ImGui_ImplSDLRenderer_GetBackendData();
207
208 // Build texture atlas
209 unsigned char* pixels;
210 int width, height;
211 io.Fonts->GetTexDataAsRGBA32(&pixels, &width, &height); // Load as RGBA 32-bit (75% of the memory is wasted, but default font is so small) because it is more likely to be compatible with user's existing shaders. If your ImTextureId represent a higher-level concept than just a GL texture id, consider calling GetTexDataAsAlpha8() instead to save on GPU memory.
212
213 // Upload texture to graphics system
214 bd->FontTexture = SDL_CreateTexture(bd->SDLRenderer, SDL_PIXELFORMAT_RGBA32, SDL_TEXTUREACCESS_STATIC, width, height);
215 if (bd->FontTexture == NULL)
216 {
217 SDL_Log("error creating texture");
218 return false;
219 }
220 SDL_UpdateTexture(bd->FontTexture, NULL, pixels, 4 * width);
221 SDL_SetTextureBlendMode(bd->FontTexture, SDL_BLENDMODE_BLEND);
222
223 // Store our identifier
224 io.Fonts->SetTexID((ImTextureID)(intptr_t)bd->FontTexture);
225
226 return true;
227}
