1// dear imgui: Renderer + Platform Backend for Allegro 5
2// (Info: Allegro 5 is a cross-platform general purpose library for handling windows, inputs, graphics, etc.)
3
4// Implemented features:
5// [X] Renderer: User texture binding. Use 'ALLEGRO_BITMAP*' as ImTextureID. Read the FAQ about ImTextureID!
6// [X] Platform: Keyboard support. Since 1.87 we are using the io.AddKeyEvent() function. Pass ImGuiKey values to all key functions e.g. ImGui::IsKeyPressed(ImGuiKey_Space). [Legacy ALLEGRO_KEY_* values will also be supported unless IMGUI_DISABLE_OBSOLETE_KEYIO is set]
7// [X] Platform: Clipboard support (from Allegro 5.1.12)
8// [X] Platform: Mouse cursor shape and visibility. Disable with 'io.ConfigFlags |= ImGuiConfigFlags_NoMouseCursorChange'.
9// Issues:
10// [ ] Renderer: The renderer is suboptimal as we need to unindex our buffers and convert vertices manually.
11// [ ] Platform: Missing gamepad support.
12
13// You can use unmodified imgui_impl_* files in your project. See examples/ folder for examples of using this.
14// Prefer including the entire imgui/ repository into your project (either as a copy or as a submodule), and only build the backends you need.
15// If you are new to Dear ImGui, read documentation from the docs/ folder + read the top of imgui.cpp.
16// Read online: https://github.com/ocornut/imgui/tree/master/docs

88static void ImGui_ImplAllegro5_SetupRenderState(ImDrawData* draw_data)
89{
90 // Setup blending
91 al_set_separate_blender(ALLEGRO_ADD, ALLEGRO_ALPHA, ALLEGRO_INVERSE_ALPHA, ALLEGRO_ADD, ALLEGRO_ONE, ALLEGRO_INVERSE_ALPHA);
92
93 // Setup orthographic projection matrix
94 // Our visible imgui space lies from draw_data->DisplayPos (top left) to draw_data->DisplayPos+data_data->DisplaySize (bottom right).
95 {
96 float L = draw_data->DisplayPos.x;
97 float R = draw_data->DisplayPos.x + draw_data->DisplaySize.x;
98 float T = draw_data->DisplayPos.y;
99 float B = draw_data->DisplayPos.y + draw_data->DisplaySize.y;
100 ALLEGRO_TRANSFORM transform;
101 al_identity_transform(&transform);
102 al_use_transform(&transform);
103 al_orthographic_transform(&transform, L, T, 1.0f, R, B, -1.0f);
104 al_use_projection_transform(&transform);
105 }
106}

108// Render function.
109void ImGui_ImplAllegro5_RenderDrawData(ImDrawData* draw_data)
110{
111 // Avoid rendering when minimized
112 if (draw_data->DisplaySize.x <= 0.0f || draw_data->DisplaySize.y <= 0.0f)
113 return;
114
115 // Backup Allegro state that will be modified
116 ImGui_ImplAllegro5_Data* bd = ImGui_ImplAllegro5_GetBackendData();
117 ALLEGRO_TRANSFORM last_transform = *al_get_current_transform();
118 ALLEGRO_TRANSFORM last_projection_transform = *al_get_current_projection_transform();
119 int last_clip_x, last_clip_y, last_clip_w, last_clip_h;
120 al_get_clipping_rectangle(&last_clip_x, &last_clip_y, &last_clip_w, &last_clip_h);
121 int last_blender_op, last_blender_src, last_blender_dst;
122 al_get_blender(&last_blender_op, &last_blender_src, &last_blender_dst);
123
124 // Setup desired render state
125 ImGui_ImplAllegro5_SetupRenderState(draw_data);
126
127 // Render command lists
128 for (int n = 0; n < draw_data->CmdListsCount; n++)
129 {
130 const ImDrawList* cmd_list = draw_data->CmdLists[n];
131
132 // Allegro's implementation of al_draw_indexed_prim() for DX9 is completely broken. Unindex our buffers ourselves.
133 // FIXME-OPT: Unfortunately Allegro doesn't support 32-bit packed colors so we have to convert them to 4 float as well..
134 static ImVector<ImDrawVertAllegro> vertices;
135 vertices.resize(cmd_list->IdxBuffer.Size);
136 for (int i = 0; i < cmd_list->IdxBuffer.Size; i++)
137 {
138 const ImDrawVert* src_v = &cmd_list->VtxBuffer[cmd_list->IdxBuffer[i]];
139 ImDrawVertAllegro* dst_v = &vertices[i];
140 dst_v->pos = src_v->pos;
141 dst_v->uv = src_v->uv;
142 unsigned char* c = (unsigned char*)&src_v->col;
143 dst_v->col = al_map_rgba(c[0], c[1], c[2], c[3]);
144 }
145
146 const int* indices = NULL;
147 if (sizeof(ImDrawIdx) == 2)
148 {
149 // FIXME-OPT: Unfortunately Allegro doesn't support 16-bit indices.. You can '#define ImDrawIdx int' in imconfig.h to request Dear ImGui to output 32-bit indices.
150 // Otherwise, we convert them from 16-bit to 32-bit at runtime here, which works perfectly but is a little wasteful.
151 static ImVector<int> indices_converted;
152 indices_converted.resize(cmd_list->IdxBuffer.Size);
153 for (int i = 0; i < cmd_list->IdxBuffer.Size; ++i)
154 indices_converted[i] = (int)cmd_list->IdxBuffer.Data[i];
155 indices = indices_converted.Data;
156 }
157 else if (sizeof(ImDrawIdx) == 4)
158 {
159 indices = (const int*)cmd_list->IdxBuffer.Data;
160 }
161
162 // Render command lists
163 ImVec2 clip_off = draw_data->DisplayPos;
164 for (int cmd_i = 0; cmd_i < cmd_list->CmdBuffer.Size; cmd_i++)
165 {
166 const ImDrawCmd* pcmd = &cmd_list->CmdBuffer[cmd_i];
167 if (pcmd->UserCallback)
168 {
169 // User callback, registered via ImDrawList::AddCallback()
170 // (ImDrawCallback_ResetRenderState is a special callback value used by the user to request the renderer to reset render state.)
171 if (pcmd->UserCallback == ImDrawCallback_ResetRenderState)
172 ImGui_ImplAllegro5_SetupRenderState(draw_data);
173 else
174 pcmd->UserCallback(cmd_list, pcmd);
175 }
176 else
177 {
178 // Project scissor/clipping rectangles into framebuffer space
179 ImVec2 clip_min(pcmd->ClipRect.x - clip_off.x, pcmd->ClipRect.y - clip_off.y);
180 ImVec2 clip_max(pcmd->ClipRect.z - clip_off.x, pcmd->ClipRect.w - clip_off.y);
181 if (clip_max.x <= clip_min.x || clip_max.y <= clip_min.y)
182 continue;
183
184 // Apply scissor/clipping rectangle, Draw
185 ALLEGRO_BITMAP* texture = (ALLEGRO_BITMAP*)pcmd->GetTexID();
186 al_set_clipping_rectangle(clip_min.x, clip_min.y, clip_max.x - clip_min.x, clip_max.y - clip_min.y);
187 al_draw_prim(&vertices[0], bd->VertexDecl, texture, pcmd->IdxOffset, pcmd->IdxOffset + pcmd->ElemCount, ALLEGRO_PRIM_TRIANGLE_LIST);
188 }
189 }
190 }
191
192 // Restore modified Allegro state
193 al_set_blender(last_blender_op, last_blender_src, last_blender_dst);
194 al_set_clipping_rectangle(last_clip_x, last_clip_y, last_clip_w, last_clip_h);
195 al_use_transform(&last_transform);
196 al_use_projection_transform(&last_projection_transform);
197}

199bool ImGui_ImplAllegro5_CreateDeviceObjects()
200{
201 // Build texture atlas
202 ImGui_ImplAllegro5_Data* bd = ImGui_ImplAllegro5_GetBackendData();
203 ImGuiIO& io = ImGui::GetIO();
204 unsigned char* pixels;
205 int width, height;
206 io.Fonts->GetTexDataAsRGBA32(&pixels, &width, &height);
207
208 // Create texture
209 int flags = al_get_new_bitmap_flags();
210 int fmt = al_get_new_bitmap_format();
211 al_set_new_bitmap_flags(ALLEGRO_MEMORY_BITMAP | ALLEGRO_MIN_LINEAR | ALLEGRO_MAG_LINEAR);
212 al_set_new_bitmap_format(ALLEGRO_PIXEL_FORMAT_ABGR_8888_LE);
213 ALLEGRO_BITMAP* img = al_create_bitmap(width, height);
214 al_set_new_bitmap_flags(flags);
215 al_set_new_bitmap_format(fmt);
216 if (!img)
217 return false;
218
219 ALLEGRO_LOCKED_REGION* locked_img = al_lock_bitmap(img, al_get_bitmap_format(img), ALLEGRO_LOCK_WRITEONLY);
220 if (!locked_img)
221 {
222 al_destroy_bitmap(img);
223 return false;
224 }
225 memcpy(locked_img->data, pixels, sizeof(int) * width * height);
226 al_unlock_bitmap(img);
227
228 // Convert software texture to hardware texture.
229 ALLEGRO_BITMAP* cloned_img = al_clone_bitmap(img);
230 al_destroy_bitmap(img);
231 if (!cloned_img)
232 return false;
233
234 // Store our identifier
235 io.Fonts->SetTexID((void*)cloned_img);
236 bd->Texture = cloned_img;
237
238 // Create an invisible mouse cursor
239 // Because al_hide_mouse_cursor() seems to mess up with the actual inputs..
240 ALLEGRO_BITMAP* mouse_cursor = al_create_bitmap(8, 8);
241 bd->MouseCursorInvisible = al_create_mouse_cursor(mouse_cursor, 0, 0);
242 al_destroy_bitmap(mouse_cursor);
243
244 return true;
245}

248{
249 ImGuiIO& io = ImGui::GetIO();
250 ImGui_ImplAllegro5_Data* bd = ImGui_ImplAllegro5_GetBackendData();

260 bd->MouseCursorInvisible = NULL;

394bool ImGui_ImplAllegro5_Init(ALLEGRO_DISPLAY* display)
395{
396 ImGuiIO& io = ImGui::GetIO();
397 IM_ASSERT(io.BackendPlatformUserData == NULL && "Already initialized a platform backend!");
398
399 // Setup backend capabilities flags
400 ImGui_ImplAllegro5_Data* bd = IM_NEW(ImGui_ImplAllegro5_Data)();
401 io.BackendPlatformUserData = (void*)bd;
402 io.BackendPlatformName = io.BackendRendererName = "imgui_impl_allegro5";
403 io.BackendFlags |= ImGuiBackendFlags_HasMouseCursors; // We can honor GetMouseCursor() values (optional)
404
405 bd->Display = display;
406
407 // Create custom vertex declaration.
408 // Unfortunately Allegro doesn't support 32-bit packed colors so we have to convert them to 4 floats.
409 // We still use a custom declaration to use 'ALLEGRO_PRIM_TEX_COORD' instead of 'ALLEGRO_PRIM_TEX_COORD_PIXEL' else we can't do a reliable conversion.
410 ALLEGRO_VERTEX_ELEMENT elems[] =
411 {
412 { ALLEGRO_PRIM_POSITION, ALLEGRO_PRIM_FLOAT_2, IM_OFFSETOF(ImDrawVertAllegro, pos) },
413 { ALLEGRO_PRIM_TEX_COORD, ALLEGRO_PRIM_FLOAT_2, IM_OFFSETOF(ImDrawVertAllegro, uv) },
414 { ALLEGRO_PRIM_COLOR_ATTR, 0, IM_OFFSETOF(ImDrawVertAllegro, col) },
415 { 0, 0, 0 }
416 };
417 bd->VertexDecl = al_create_vertex_decl(elems, sizeof(ImDrawVertAllegro));
418
419#if ALLEGRO_HAS_CLIPBOARD
420 io.SetClipboardTextFn = ImGui_ImplAllegro5_SetClipboardText;
421 io.GetClipboardTextFn = ImGui_ImplAllegro5_GetClipboardText;
422 io.ClipboardUserData = NULL;
423#endif
424
425 return true;
426}

428void ImGui_ImplAllegro5_Shutdown()
429{
430 ImGui_ImplAllegro5_Data* bd = ImGui_ImplAllegro5_GetBackendData();
431 IM_ASSERT(bd != NULL && "No platform backend to shutdown, or already shutdown?");
432 ImGuiIO& io = ImGui::GetIO();
433
434 ImGui_ImplAllegro5_InvalidateDeviceObjects();
435 if (bd->VertexDecl)
436 al_destroy_vertex_decl(bd->VertexDecl);
437 if (bd->ClipboardTextData)
438 al_free(bd->ClipboardTextData);
439
440 io.BackendPlatformUserData = NULL;
441 io.BackendPlatformName = io.BackendRendererName = NULL;
442 IM_DELETE(bd);
443}

445// ev->keyboard.modifiers seems always zero so using that...
446static void ImGui_ImplAllegro5_UpdateKeyModifiers()
447{
448 ImGuiIO& io = ImGui::GetIO();
449 ALLEGRO_KEYBOARD_STATE keys;
450 al_get_keyboard_state(&keys);
451 ImGuiKeyModFlags key_mods =

459// You can read the io.WantCaptureMouse, io.WantCaptureKeyboard flags to tell if dear imgui wants to use your inputs.
460// - When io.WantCaptureMouse is true, do not dispatch mouse input data to your main application.
461// - When io.WantCaptureKeyboard is true, do not dispatch keyboard input data to your main application.
462// Generally you may always pass all inputs to dear imgui, and hide them from your application based on those two flags.
463bool ImGui_ImplAllegro5_ProcessEvent(ALLEGRO_EVENT* ev)
464{
465 ImGuiIO& io = ImGui::GetIO();
466 ImGui_ImplAllegro5_Data* bd = ImGui_ImplAllegro5_GetBackendData();
467
468 switch (ev->type)
469 {
470 case ALLEGRO_EVENT_MOUSE_AXES:
471 if (ev->mouse.display == bd->Display)
472 {
473 io.AddMousePosEvent(ev->mouse.x, ev->mouse.y);
474 io.AddMouseWheelEvent(-ev->mouse.dw, ev->mouse.dz);
475 }
476 return true;
477 case ALLEGRO_EVENT_MOUSE_BUTTON_DOWN:
478 case ALLEGRO_EVENT_MOUSE_BUTTON_UP:
479 if (ev->mouse.display == bd->Display && ev->mouse.button > 0 && ev->mouse.button <= 5)
480 io.AddMouseButtonEvent(ev->mouse.button - 1, ev->type == ALLEGRO_EVENT_MOUSE_BUTTON_DOWN);
481 return true;
482 case ALLEGRO_EVENT_TOUCH_MOVE:
483 if (ev->touch.display == bd->Display)
484 io.AddMousePosEvent(ev->touch.x, ev->touch.y);
485 return true;
486 case ALLEGRO_EVENT_TOUCH_BEGIN:
487 case ALLEGRO_EVENT_TOUCH_END:
488 case ALLEGRO_EVENT_TOUCH_CANCEL:
489 if (ev->touch.display == bd->Display && ev->touch.primary)
490 io.AddMouseButtonEvent(0, ev->type == ALLEGRO_EVENT_TOUCH_BEGIN);
491 return true;
492 case ALLEGRO_EVENT_MOUSE_LEAVE_DISPLAY:
493 if (ev->mouse.display == bd->Display)
494 io.AddMousePosEvent(-FLT_MAX, -FLT_MAX);
495 return true;
496 case ALLEGRO_EVENT_KEY_CHAR:
497 if (ev->keyboard.display == bd->Display)
498 if (ev->keyboard.unichar != 0)
499 io.AddInputCharacter((unsigned int)ev->keyboard.unichar);
500 return true;
501 case ALLEGRO_EVENT_KEY_DOWN:
502 case ALLEGRO_EVENT_KEY_UP:
503 if (ev->keyboard.display == bd->Display)
504 {
505 ImGui_ImplAllegro5_UpdateKeyModifiers();
506 ImGuiKey key = ImGui_ImplAllegro5_KeyCodeToImGuiKey(ev->keyboard.keycode);
507 io.AddKeyEvent(key, (ev->type == ALLEGRO_EVENT_KEY_DOWN));
508 io.SetKeyEventNativeData(key, ev->keyboard.keycode, -1); // To support legacy indexing (<1.87 user code)
509 }
510 return true;
511 case ALLEGRO_EVENT_DISPLAY_SWITCH_OUT:
512 if (ev->display.source == bd->Display)
513 io.AddFocusEvent(false);
514 return true;
515 case ALLEGRO_EVENT_DISPLAY_SWITCH_IN:
516 if (ev->display.source == bd->Display)
517 {
518 io.AddFocusEvent(true);
519#if defined(ALLEGRO_UNSTABLE)
520 al_clear_keyboard_state(bd->Display);
521#endif
522 }
523 return true;
524 }
525 return false;
526}
