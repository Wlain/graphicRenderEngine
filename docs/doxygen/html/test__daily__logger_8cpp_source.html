166static void test_rotate(int days_to_run, uint16_t max_days, uint16_t expected_n_files)
167{
168 using spdlog::log_clock;
169 using spdlog::details::log_msg;
170 using spdlog::sinks::daily_file_sink_st;
171
172 prepare_logdir();
173
174 spdlog::filename_t basename = SPDLOG_FILENAME_T("test_logs/daily_rotate.txt");
175 daily_file_sink_st sink{basename, 2, 30, true, max_days};
176
177 // simulate messages with 24 intervals
178
179 for (int i = 0; i < days_to_run; i++)
180 {
181 auto offset = std::chrono::seconds{24 * 3600 * i};
182 sink.log(create_msg(offset));
183 }
184
185 REQUIRE(count_files("test_logs") == static_cast<size_t>(expected_n_files));
186}
