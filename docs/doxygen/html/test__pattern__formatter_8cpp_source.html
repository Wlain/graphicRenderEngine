13 oss_logger.set_level(spdlog::level::info);
14

16
17 oss_logger.info(msg);

341class custom_test_flag : public spdlog::custom_flag_formatter
342{
343public:
344 explicit custom_test_flag(std::string txt)
345 : some_txt{std::move(txt)}
346 {}
347
348 void format(const spdlog::details::log_msg &, const std::tm &, spdlog::memory_buf_t &dest) override
349 {
350 if (some_txt == "throw_me")
351 {
352 throw spdlog::spdlog_ex("custom_flag_exception_test");
353 }
354 some_txt = std::string(padinfo_.width_, ' ') + some_txt;
355 dest.append(some_txt.data(), some_txt.data() + some_txt.size());
356 }
357 spdlog::details::padding_info get_padding_info()
358 {
359 return padinfo_;
360 }
361
362 std::string some_txt;
363
364 std::unique_ptr<custom_flag_formatter> clone() const override
365 {
366 return spdlog::details::make_unique<custom_test_flag>(some_txt);
367 }
368};
