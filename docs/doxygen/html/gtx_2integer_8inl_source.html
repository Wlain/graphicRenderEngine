154 // Hackers Delight: http://www.hackersdelight.org/HDcode/nlz.c.txt
155 GLM_FUNC_QUALIFIER unsigned int nlz(unsigned int x)
156 {
157 int y, m, n;
158
159 y = -int(x >> 16); // If left half of x is 0,
160 m = (y >> 16) & 16; // set n = 16. If left half
161 n = 16 - m; // is nonzero, set n = 0 and
162 x = x >> m; // shift x right 16.
163 // Now x is of the form 0000xxxx.
164 y = x - 0x100; // If positions 8-15 are 0,
165 m = (y >> 16) & 8; // add 8 to n and shift x left 8.
166 n = n + m;
167 x = x << m;
168
169 y = x - 0x1000; // If positions 12-15 are 0,
170 m = (y >> 16) & 4; // add 4 to n and shift x left 4.
171 n = n + m;
172 x = x << m;
173
174 y = x - 0x4000; // If positions 14-15 are 0,
175 m = (y >> 16) & 2; // add 2 to n and shift x left 2.
176 n = n + m;
177 x = x << m;
178
179 y = x >> 14; // Set y = 0, 1, 2, or 3.
180 m = y & ~(y >> 1); // Set m = 0, 1, 2, or 2 resp.
181 return unsigned(n + 2 - m);
182 }
