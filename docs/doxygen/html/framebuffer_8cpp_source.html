12void checkStatus()
13{
14 GLenum frameBufferStatus = glCheckFramebufferStatus(GL_FRAMEBUFFER);
15 if (frameBufferStatus != GL_FRAMEBUFFER_COMPLETE)
16 {
17 switch (frameBufferStatus)
18 {
19 case GL_FRAMEBUFFER_INCOMPLETE_ATTACHMENT:
20 LOG_ERROR("GL_FRAMEBUFFER_UNDEFINED");
21 break;
22 case GL_FRAMEBUFFER_INCOMPLETE_MISSING_ATTACHMENT:
23 LOG_ERROR("GL_FRAMEBUFFER_INCOMPLETE_MISSING_ATTACHMENT");
24 break;
25 case GL_FRAMEBUFFER_UNSUPPORTED:
26 LOG_ERROR("FRAMEBUFFER_UNSUPPORTED");
27 break;
28 case GL_FRAMEBUFFER_INCOMPLETE_DRAW_BUFFER:
29 LOG_ERROR("GL_FRAMEBUFFER_INCOMPLETE_DRAW_BUFFER");
30 break;
31 case GL_FRAMEBUFFER_INCOMPLETE_READ_BUFFER:
32 LOG_ERROR("GL_FRAMEBUFFER_INCOMPLETE_READ_BUFFER");
33 break;
34 case GL_FRAMEBUFFER_UNDEFINED:
35 LOG_ERROR("FRAMEBUFFER_INCOMPLETE_ATTACHMENT");
36 break;
37 case GL_FRAMEBUFFER_INCOMPLETE_MULTISAMPLE:
38 LOG_ERROR("GL_FRAMEBUFFER_INCOMPLETE_MULTISAMPLE");
39 break;
40 case GL_FRAMEBUFFER_INCOMPLETE_LAYER_TARGETS:
41 LOG_ERROR("GL_FRAMEBUFFER_INCOMPLETE_LAYER_TARGETS");
42 break;
43 default:
44 LOG_ERROR("Invalid framebuffer {}", frameBufferStatus);
45 break;
46 }
47 }

80std::shared_ptr<FrameBuffer> FrameBuffer::FrameBufferBuilder::build()
81{
82 if (m_name.empty())
83 {
84 m_name = "Unnamed Framebuffer";
85 }
86 auto framebuffer = new FrameBuffer(m_name);
87 framebuffer->m_size = m_size;
88
89 glGenFramebuffers(1, &framebuffer->m_fbo);
90 glBindFramebuffer(GL_FRAMEBUFFER, framebuffer->m_fbo);
91 std::vector<GLenum> drawBuffers;
92 for (unsigned i = 0; i < m_textures.size(); i++)
93 {
94 glFramebufferTexture2D(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0 + i, GL_TEXTURE_2D, m_textures[i]->m_info.id, 0);
95 drawBuffers.push_back(GL_COLOR_ATTACHMENT0 + i);
96 }
97 if (m_depthTexture)
98 {
99 glFramebufferTexture2D(GL_FRAMEBUFFER, GL_DEPTH_ATTACHMENT, GL_TEXTURE_2D, m_depthTexture->m_info.id, 0);
100 }
101 else
102 {
103 glGenRenderbuffers(1, &framebuffer->m_rbo);
104 glBindRenderbuffer(GL_RENDERBUFFER, framebuffer->m_rbo);
105 glRenderbufferStorage(GL_RENDERBUFFER, GL_DEPTH_COMPONENT32, m_size.x, m_size.y);
106 glRenderbufferStorage(GL_RENDERBUFFER, GL_DEPTH_COMPONENT, m_size.x, m_size.y);
107 glBindRenderbuffer(GL_RENDERBUFFER, 0);
108 // attach the renderbuffer to depth attachment point
109 glFramebufferRenderbuffer(GL_FRAMEBUFFER, GL_DEPTH_ATTACHMENT, GL_RENDERBUFFER, framebuffer->m_rbo);
110 }
111 // check FBO
112 checkStatus();
113 framebuffer->m_textures = m_textures;
114 framebuffer->m_depthTexture = m_depthTexture;
115 glBindFramebuffer(GL_FRAMEBUFFER, 0);
116 return std::shared_ptr<FrameBuffer>(framebuffer);
117}

173void FrameBuffer::bind()
174{
175 glBindFramebuffer(GL_FRAMEBUFFER, m_fbo);
176 if (m_dirty)
177 {
178 for (int i = 0; i < m_textures.size(); i++)
179 {
180 glFramebufferTexture2D(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0 + i, GL_TEXTURE_2D, m_textures[i]->m_info.id, 0);
181 }
182 if (m_depthTexture)
183 {
184 glFramebufferTexture2D(GL_FRAMEBUFFER, GL_DEPTH_ATTACHMENT, GL_TEXTURE_2D, m_depthTexture->m_info.id, 0);
185 }
186 m_dirty = false;
