200bool CreateDeviceD3D(HWND hWnd)
201{
202 // Setup swap chain
203 DXGI_SWAP_CHAIN_DESC sd;
204 ZeroMemory(&sd, sizeof(sd));
205 sd.BufferCount = 2;
206 sd.BufferDesc.Width = 0;
207 sd.BufferDesc.Height = 0;
208 sd.BufferDesc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
209 sd.BufferDesc.RefreshRate.Numerator = 60;
210 sd.BufferDesc.RefreshRate.Denominator = 1;
211 sd.Flags = DXGI_SWAP_CHAIN_FLAG_ALLOW_MODE_SWITCH;
212 sd.BufferUsage = DXGI_USAGE_RENDER_TARGET_OUTPUT;
213 sd.OutputWindow = hWnd;
214 sd.SampleDesc.Count = 1;
215 sd.SampleDesc.Quality = 0;
216 sd.Windowed = TRUE;
217 sd.SwapEffect = DXGI_SWAP_EFFECT_DISCARD;
218
219 UINT createDeviceFlags = 0;
220 //createDeviceFlags |= D3D11_CREATE_DEVICE_DEBUG;
221 D3D_FEATURE_LEVEL featureLevel;
222 const D3D_FEATURE_LEVEL featureLevelArray[2] = { D3D_FEATURE_LEVEL_11_0, D3D_FEATURE_LEVEL_10_0, };
223 if (D3D11CreateDeviceAndSwapChain(NULL, D3D_DRIVER_TYPE_HARDWARE, NULL, createDeviceFlags, featureLevelArray, 2, D3D11_SDK_VERSION, &sd, &g_pSwapChain, &g_pd3dDevice, &featureLevel, &g_pd3dDeviceContext) != S_OK)
224 return false;
225
226 CreateRenderTarget();
227 return true;
228}

240 ID3D11Texture2D* pBackBuffer;
