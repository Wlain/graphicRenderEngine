191void ImGui_ImplGLUT_NewFrame()
192{
193 // Setup time step
194 ImGuiIO& io = ImGui::GetIO();
195 int current_time = glutGet(GLUT_ELAPSED_TIME);
196 int delta_time_ms = (current_time - g_Time);
197 if (delta_time_ms <= 0)
198 delta_time_ms = 1;
199 io.DeltaTime = delta_time_ms / 1000.0f;
200 g_Time = current_time;
201
202 // Start the frame
203 ImGui::NewFrame();
204}

206static void ImGui_ImplGLUT_UpdateKeyModifiers()
207{
208 ImGuiIO& io = ImGui::GetIO();
209 int glut_key_mods = glutGetModifiers();
210 ImGuiKeyModFlags key_mods =
211 ((glut_key_mods & GLUT_ACTIVE_CTRL) ? ImGuiKeyModFlags_Ctrl : 0) |
212 ((glut_key_mods & GLUT_ACTIVE_SHIFT) ? ImGuiKeyModFlags_Shift : 0) |
213 ((glut_key_mods & GLUT_ACTIVE_ALT) ? ImGuiKeyModFlags_Alt : 0);
214 io.AddKeyModsEvent(key_mods);
215}

225{
226 // Send character to imgui
227 //printf("char_down_func %d '%c'\n", c, c);
228 ImGuiIO& io = ImGui::GetIO();
229 if (c >= 32)

265void ImGui_ImplGLUT_MouseFunc(int glut_button, int state, int x, int y)
266{
267 ImGuiIO& io = ImGui::GetIO();
268 io.AddMousePosEvent((float)x, (float)y);
269 int button = -1;
270 if (glut_button == GLUT_LEFT_BUTTON) button = 0;
271 if (glut_button == GLUT_RIGHT_BUTTON) button = 1;
272 if (glut_button == GLUT_MIDDLE_BUTTON) button = 2;
273 if (button != -1 && state == GLUT_DOWN || state == GLUT_UP)
274 io.AddMouseButtonEvent(button, state == GLUT_DOWN);
275}
