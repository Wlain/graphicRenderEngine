40void bench_threaded_logging(size_t threads, int iters)
41{

113int main(int argc, char *argv[])
114{
115 spdlog::set_automatic_registration(false);
116 spdlog::default_logger()->set_pattern("[%^%l%$] %v");
117 int iters = 250000;
118 size_t threads = 4;
119 try
120 {
121
122 if (argc > 1)
123 {
124 iters = std::stoi(argv[1]);
125 }
126 if (argc > 2)
127 {
128 threads = std::stoul(argv[2]);
129 }
130
131 if (threads > max_threads)
132 {
133 throw std::runtime_error(spdlog::fmt_lib::format("Number of threads exceeds maximum({})", max_threads));
134 }
135
136 bench_single_threaded(iters);
137 bench_threaded_logging(1, iters);
138 bench_threaded_logging(threads, iters);
139 }
140 catch (std::exception &ex)
141 {
142 spdlog::error(ex.what());
143 return EXIT_FAILURE;
144 }
145 return EXIT_SUCCESS;
146}

148void bench(int howmany, std::shared_ptr<spdlog::logger> log)
149{
150 using std::chrono::duration;
151 using std::chrono::duration_cast;
152 using std::chrono::high_resolution_clock;
153
154 auto start = high_resolution_clock::now();
155 for (auto i = 0; i < howmany; ++i)
156 {
157 log->info("Hello logger: msg number {}", i);
158 }
159
160 auto delta = high_resolution_clock::now() - start;
161 auto delta_d = duration_cast<duration<double>>(delta).count();
162
163 spdlog::info(spdlog::fmt_lib::format(
164 std::locale("en_US.UTF-8"), "{:<30} Elapsed: {:0.2f} secs {:>16L}/sec", log->name(), delta_d, int(howmany / delta_d)));
165 spdlog::drop(log->name());
166}

168void bench_mt(int howmany, std::shared_ptr<spdlog::logger> log, size_t thread_count)
169{
170 using std::chrono::duration;
171 using std::chrono::duration_cast;
172 using std::chrono::high_resolution_clock;
173
174 std::vector<std::thread> threads;
175 threads.reserve(thread_count);
176 auto start = high_resolution_clock::now();
177 for (size_t t = 0; t < thread_count; ++t)
178 {
179 threads.emplace_back([&]() {
180 for (int j = 0; j < howmany / static_cast<int>(thread_count); j++)
181 {
182 log->info("Hello logger: msg number {}", j);
183 }
184 });
185 }
186
187 for (auto &t : threads)
188 {
189 t.join();
190 };
191
192 auto delta = high_resolution_clock::now() - start;
193 auto delta_d = duration_cast<duration<double>>(delta).count();
194 spdlog::info(spdlog::fmt_lib::format(
195 std::locale("en_US.UTF-8"), "{:<30} Elapsed: {:0.2f} secs {:>16L}/sec", log->name(), delta_d, int(howmany / delta_d)));
196 spdlog::drop(log->name());
197}
