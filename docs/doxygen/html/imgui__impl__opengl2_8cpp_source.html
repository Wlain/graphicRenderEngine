1// dear imgui: Renderer Backend for OpenGL2 (legacy OpenGL, fixed pipeline)
2// This needs to be used along with a Platform Backend (e.g. GLFW, SDL, Win32, custom..)
3
4// Implemented features:
5// [X] Renderer: User texture binding. Use 'GLuint' OpenGL texture identifier as void*/ImTextureID. Read the FAQ about ImTextureID!
6// [X] Renderer: Multi-viewport support. Enable with 'io.ConfigFlags |= ImGuiConfigFlags_ViewportsEnable'.
7
8// You can use unmodified imgui_impl_* files in your project. See examples/ folder for examples of using this.
9// Prefer including the entire imgui/ repository into your project (either as a copy or as a submodule), and only build the backends you need.
10// If you are new to Dear ImGui, read documentation from the docs/ folder + read the top of imgui.cpp.
11// Read online: https://github.com/ocornut/imgui/tree/master/docs
12
13// **DO NOT USE THIS CODE IF YOUR CODE/ENGINE IS USING MODERN OPENGL (SHADERS, VBO, VAO, etc.)**
14// **Prefer using the code in imgui_impl_opengl3.cpp**
15// This code is mostly provided as a reference to learn how ImGui integration works, because it is shorter to read.
16// If your code is using GL3+ context or any semi modern OpenGL calls, using this is likely to make everything more
17// complicated, will require your code to reset every single OpenGL attributes to their initial state, and might
18// confuse your GPU driver.
19// The GL2 code is unable to reset attributes or even call e.g. "glUseProgram(0)" because they don't exist in that API.
20

88 io.BackendRendererUserData = (void*)bd;

90 io.BackendFlags |= ImGuiBackendFlags_RendererHasViewports; // We can create multi-viewports on the Renderer side (optional)

120static void ImGui_ImplOpenGL2_SetupRenderState(ImDrawData* draw_data, int fb_width, int fb_height)
121{
122 // Setup render state: alpha-blending enabled, no face culling, no depth testing, scissor enabled, vertex/texcoord/color pointers, polygon fill.
123 glEnable(GL_BLEND);
124 glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
125 //glBlendFuncSeparate(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA, GL_ONE, GL_ONE_MINUS_SRC_ALPHA); // In order to composite our output buffer we need to preserve alpha
126 glDisable(GL_CULL_FACE);
127 glDisable(GL_DEPTH_TEST);
128 glDisable(GL_STENCIL_TEST);
129 glDisable(GL_LIGHTING);
130 glDisable(GL_COLOR_MATERIAL);
131 glEnable(GL_SCISSOR_TEST);
132 glEnableClientState(GL_VERTEX_ARRAY);
133 glEnableClientState(GL_TEXTURE_COORD_ARRAY);
134 glEnableClientState(GL_COLOR_ARRAY);
135 glDisableClientState(GL_NORMAL_ARRAY);
136 glEnable(GL_TEXTURE_2D);
137 glPolygonMode(GL_FRONT_AND_BACK, GL_FILL);
138 glShadeModel(GL_SMOOTH);
139 glTexEnvi(GL_TEXTURE_ENV, GL_TEXTURE_ENV_MODE, GL_MODULATE);
140
141 // If you are using this code with non-legacy OpenGL header/contexts (which you should not, prefer using imgui_impl_opengl3.cpp!!),
142 // you may need to backup/reset/restore other state, e.g. for current shader using the commented lines below.
143 // (DO NOT MODIFY THIS FILE! Add the code in your calling function)
144 // GLint last_program;
145 // glGetIntegerv(GL_CURRENT_PROGRAM, &last_program);
146 // glUseProgram(0);
147 // ImGui_ImplOpenGL2_RenderDrawData(...);
148 // glUseProgram(last_program)
149 // There are potentially many more states you could need to clear/setup that we can't access from default headers.
150 // e.g. glBindBuffer(GL_ARRAY_BUFFER, 0), glDisable(GL_TEXTURE_CUBE_MAP).
151
152 // Setup viewport, orthographic projection matrix
153 // Our visible imgui space lies from draw_data->DisplayPos (top left) to draw_data->DisplayPos+data_data->DisplaySize (bottom right). DisplayPos is (0,0) for single viewport apps.
154 glViewport(0, 0, (GLsizei)fb_width, (GLsizei)fb_height);
155 glMatrixMode(GL_PROJECTION);
156 glPushMatrix();
157 glLoadIdentity();
158 glOrtho(draw_data->DisplayPos.x, draw_data->DisplayPos.x + draw_data->DisplaySize.x, draw_data->DisplayPos.y + draw_data->DisplaySize.y, draw_data->DisplayPos.y, -1.0f, +1.0f);
159 glMatrixMode(GL_MODELVIEW);
160 glPushMatrix();
161 glLoadIdentity();
162}

164// OpenGL2 Render function.
165// Note that this implementation is little overcomplicated because we are saving/setting up/restoring every OpenGL state explicitly.
166// This is in order to be able to run within an OpenGL engine that doesn't do so.
167void ImGui_ImplOpenGL2_RenderDrawData(ImDrawData* draw_data)
168{
169 // Avoid rendering when minimized, scale coordinates for retina displays (screen coordinates != framebuffer coordinates)
170 int fb_width = (int)(draw_data->DisplaySize.x * draw_data->FramebufferScale.x);
171 int fb_height = (int)(draw_data->DisplaySize.y * draw_data->FramebufferScale.y);
172 if (fb_width == 0 || fb_height == 0)
173 return;
174
175 // Backup GL state
176 GLint last_texture; glGetIntegerv(GL_TEXTURE_BINDING_2D, &last_texture);
177 GLint last_polygon_mode[2]; glGetIntegerv(GL_POLYGON_MODE, last_polygon_mode);
178 GLint last_viewport[4]; glGetIntegerv(GL_VIEWPORT, last_viewport);
179 GLint last_scissor_box[4]; glGetIntegerv(GL_SCISSOR_BOX, last_scissor_box);
180 GLint last_shade_model; glGetIntegerv(GL_SHADE_MODEL, &last_shade_model);
181 GLint last_tex_env_mode; glGetTexEnviv(GL_TEXTURE_ENV, GL_TEXTURE_ENV_MODE, &last_tex_env_mode);
182 glPushAttrib(GL_ENABLE_BIT | GL_COLOR_BUFFER_BIT | GL_TRANSFORM_BIT);
183
184 // Setup desired GL state
185 ImGui_ImplOpenGL2_SetupRenderState(draw_data, fb_width, fb_height);
186
187 // Will project scissor/clipping rectangles into framebuffer space
188 ImVec2 clip_off = draw_data->DisplayPos; // (0,0) unless using multi-viewports
189 ImVec2 clip_scale = draw_data->FramebufferScale; // (1,1) unless using retina display which are often (2,2)
190
191 // Render command lists
192 for (int n = 0; n < draw_data->CmdListsCount; n++)
193 {
194 const ImDrawList* cmd_list = draw_data->CmdLists[n];
195 const ImDrawVert* vtx_buffer = cmd_list->VtxBuffer.Data;
196 const ImDrawIdx* idx_buffer = cmd_list->IdxBuffer.Data;
197 glVertexPointer(2, GL_FLOAT, sizeof(ImDrawVert), (const GLvoid*)((const char*)vtx_buffer + IM_OFFSETOF(ImDrawVert, pos)));
198 glTexCoordPointer(2, GL_FLOAT, sizeof(ImDrawVert), (const GLvoid*)((const char*)vtx_buffer + IM_OFFSETOF(ImDrawVert, uv)));
199 glColorPointer(4, GL_UNSIGNED_BYTE, sizeof(ImDrawVert), (const GLvoid*)((const char*)vtx_buffer + IM_OFFSETOF(ImDrawVert, col)));
200
201 for (int cmd_i = 0; cmd_i < cmd_list->CmdBuffer.Size; cmd_i++)
202 {
203 const ImDrawCmd* pcmd = &cmd_list->CmdBuffer[cmd_i];
204 if (pcmd->UserCallback)
205 {
206 // User callback, registered via ImDrawList::AddCallback()
207 // (ImDrawCallback_ResetRenderState is a special callback value used by the user to request the renderer to reset render state.)
208 if (pcmd->UserCallback == ImDrawCallback_ResetRenderState)
209 ImGui_ImplOpenGL2_SetupRenderState(draw_data, fb_width, fb_height);
210 else
211 pcmd->UserCallback(cmd_list, pcmd);
212 }
213 else
214 {
215 // Project scissor/clipping rectangles into framebuffer space
216 ImVec2 clip_min((pcmd->ClipRect.x - clip_off.x) * clip_scale.x, (pcmd->ClipRect.y - clip_off.y) * clip_scale.y);
217 ImVec2 clip_max((pcmd->ClipRect.z - clip_off.x) * clip_scale.x, (pcmd->ClipRect.w - clip_off.y) * clip_scale.y);
218 if (clip_max.x <= clip_min.x || clip_max.y <= clip_min.y)
219 continue;
220
221 // Apply scissor/clipping rectangle (Y is inverted in OpenGL)
222 glScissor((int)clip_min.x, (int)(fb_height - clip_max.y), (int)(clip_max.x - clip_min.x), (int)(clip_max.y - clip_min.y));
223
224 // Bind texture, Draw
225 glBindTexture(GL_TEXTURE_2D, (GLuint)(intptr_t)pcmd->GetTexID());
226 glDrawElements(GL_TRIANGLES, (GLsizei)pcmd->ElemCount, sizeof(ImDrawIdx) == 2 ? GL_UNSIGNED_SHORT : GL_UNSIGNED_INT, idx_buffer + pcmd->IdxOffset);
227 }
228 }
229 }
230
231 // Restore modified GL state
232 glDisableClientState(GL_COLOR_ARRAY);
233 glDisableClientState(GL_TEXTURE_COORD_ARRAY);
234 glDisableClientState(GL_VERTEX_ARRAY);
235 glBindTexture(GL_TEXTURE_2D, (GLuint)last_texture);
236 glMatrixMode(GL_MODELVIEW);
237 glPopMatrix();
238 glMatrixMode(GL_PROJECTION);
239 glPopMatrix();
240 glPopAttrib();
241 glPolygonMode(GL_FRONT, (GLenum)last_polygon_mode[0]); glPolygonMode(GL_BACK, (GLenum)last_polygon_mode[1]);
242 glViewport(last_viewport[0], last_viewport[1], (GLsizei)last_viewport[2], (GLsizei)last_viewport[3]);
243 glScissor(last_scissor_box[0], last_scissor_box[1], (GLsizei)last_scissor_box[2], (GLsizei)last_scissor_box[3]);
244 glShadeModel(last_shade_model);
245 glTexEnvi(GL_TEXTURE_ENV, GL_TEXTURE_ENV_MODE, last_tex_env_mode);
246}

248bool ImGui_ImplOpenGL2_CreateFontsTexture()
249{
250 // Build texture atlas
251 ImGuiIO& io = ImGui::GetIO();
252 ImGui_ImplOpenGL2_Data* bd = ImGui_ImplOpenGL2_GetBackendData();
253 unsigned char* pixels;
254 int width, height;
255 io.Fonts->GetTexDataAsRGBA32(&pixels, &width, &height); // Load as RGBA 32-bit (75% of the memory is wasted, but default font is so small) because it is more likely to be compatible with user's existing shaders. If your ImTextureId represent a higher-level concept than just a GL texture id, consider calling GetTexDataAsAlpha8() instead to save on GPU memory.
256
257 // Upload texture to graphics system
258 GLint last_texture;
259 glGetIntegerv(GL_TEXTURE_BINDING_2D, &last_texture);
260 glGenTextures(1, &bd->FontTexture);
261 glBindTexture(GL_TEXTURE_2D, bd->FontTexture);
262 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
263 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
264 glPixelStorei(GL_UNPACK_ROW_LENGTH, 0);
265 glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA, width, height, 0, GL_RGBA, GL_UNSIGNED_BYTE, pixels);
266
267 // Store our identifier
268 io.Fonts->SetTexID((ImTextureID)(intptr_t)bd->FontTexture);
269
270 // Restore state
271 glBindTexture(GL_TEXTURE_2D, last_texture);
272
273 return true;
274}
