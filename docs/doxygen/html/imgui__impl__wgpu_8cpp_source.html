1// dear imgui: Renderer for WebGPU
2// This needs to be used along with a Platform Binding (e.g. GLFW)
3// (Please note that WebGPU is currently experimental, will not run on non-beta browsers, and may break.)
4
5// Implemented features:
6// [X] Renderer: User texture binding. Use 'WGPUTextureView' as ImTextureID. Read the FAQ about ImTextureID!
7// [X] Renderer: Support for large meshes (64k+ vertices) with 16-bit indices.
8
9// You can use unmodified imgui_impl_* files in your project. See examples/ folder for examples of using this.
10// Prefer including the entire imgui/ repository into your project (either as a copy or as a submodule), and only build the backends you need.
11// If you are new to Dear ImGui, read documentation from the docs/ folder + read the top of imgui.cpp.
12// Read online: https://github.com/ocornut/imgui/tree/master/docs
13

267static WGPUProgrammableStageDescriptor ImGui_ImplWGPU_CreateShaderModule(uint32_t* binary_data, uint32_t binary_data_size)
268{
269 WGPUShaderModuleSPIRVDescriptor spirv_desc = {};
270 spirv_desc.chain.sType = WGPUSType_ShaderModuleSPIRVDescriptor;
271 spirv_desc.codeSize = binary_data_size;
272 spirv_desc.code = binary_data;
273
274 WGPUShaderModuleDescriptor desc = {};
275 desc.nextInChain = reinterpret_cast<WGPUChainedStruct*>(&spirv_desc);
276
277 WGPUProgrammableStageDescriptor stage_desc = {};
278 stage_desc.module = wgpuDeviceCreateShaderModule(g_wgpuDevice, &desc);
279 stage_desc.entryPoint = "main";
280 return stage_desc;
281}

294static void ImGui_ImplWGPU_SetupRenderState(ImDrawData* draw_data, WGPURenderPassEncoder ctx, FrameResources* fr)
295{
296 // Setup orthographic projection matrix into our constant buffer
297 // Our visible imgui space lies from draw_data->DisplayPos (top left) to draw_data->DisplayPos+data_data->DisplaySize (bottom right).
298 {
299 float L = draw_data->DisplayPos.x;
300 float R = draw_data->DisplayPos.x + draw_data->DisplaySize.x;
301 float T = draw_data->DisplayPos.y;
302 float B = draw_data->DisplayPos.y + draw_data->DisplaySize.y;
303 float mvp[4][4] =
304 {
305 { 2.0f/(R-L), 0.0f, 0.0f, 0.0f },
306 { 0.0f, 2.0f/(T-B), 0.0f, 0.0f },
307 { 0.0f, 0.0f, 0.5f, 0.0f },
308 { (R+L)/(L-R), (T+B)/(B-T), 0.5f, 1.0f },
309 };
310 wgpuQueueWriteBuffer(g_defaultQueue, g_resources.Uniforms, 0, mvp, sizeof(mvp));
311 }
312
313 // Setup viewport
314 wgpuRenderPassEncoderSetViewport(ctx, 0, 0, draw_data->FramebufferScale.x * draw_data->DisplaySize.x, draw_data->FramebufferScale.y * draw_data->DisplaySize.y, 0, 1);
315
316 // Bind shader and vertex buffers
317 wgpuRenderPassEncoderSetVertexBuffer(ctx, 0, fr->VertexBuffer, 0, fr->VertexBufferSize * sizeof(ImDrawVert));
318 wgpuRenderPassEncoderSetIndexBuffer(ctx, fr->IndexBuffer, sizeof(ImDrawIdx) == 2 ? WGPUIndexFormat_Uint16 : WGPUIndexFormat_Uint32, 0, fr->IndexBufferSize * sizeof(ImDrawIdx));
319 wgpuRenderPassEncoderSetPipeline(ctx, g_pipelineState);
320 wgpuRenderPassEncoderSetBindGroup(ctx, 0, g_resources.CommonBindGroup, 0, NULL);
321
322 // Setup blend factor
323 WGPUColor blend_color = { 0.f, 0.f, 0.f, 0.f };
324 wgpuRenderPassEncoderSetBlendConstant(ctx, &blend_color);
325}

327// Render function
328// (this used to be set in io.RenderDrawListsFn and called by ImGui::Render(), but you can now call this directly from your main loop)
329void ImGui_ImplWGPU_RenderDrawData(ImDrawData* draw_data, WGPURenderPassEncoder pass_encoder)
330{
331 // Avoid rendering when minimized
332 if (draw_data->DisplaySize.x <= 0.0f || draw_data->DisplaySize.y <= 0.0f)
333 return;
334
335 // FIXME: Assuming that this only gets called once per frame!
336 // If not, we can't just re-allocate the IB or VB, we'll have to do a proper allocator.
337 g_frameIndex = g_frameIndex + 1;
338 FrameResources* fr = &g_pFrameResources[g_frameIndex % g_numFramesInFlight];
339
340 // Create and grow vertex/index buffers if needed
341 if (fr->VertexBuffer == NULL || fr->VertexBufferSize < draw_data->TotalVtxCount)
342 {
343 if (fr->VertexBuffer)
344 {
345 wgpuBufferDestroy(fr->VertexBuffer);
346 wgpuBufferRelease(fr->VertexBuffer);
347 }
348 SafeRelease(fr->VertexBufferHost);
349 fr->VertexBufferSize = draw_data->TotalVtxCount + 5000;
350
351 WGPUBufferDescriptor vb_desc =
352 {
353 NULL,
354 "Dear ImGui Vertex buffer",
355 WGPUBufferUsage_CopyDst | WGPUBufferUsage_Vertex,
356 fr->VertexBufferSize * sizeof(ImDrawVert),
357 false
358 };
359 fr->VertexBuffer = wgpuDeviceCreateBuffer(g_wgpuDevice, &vb_desc);
360 if (!fr->VertexBuffer)
361 return;
362
363 fr->VertexBufferHost = new ImDrawVert[fr->VertexBufferSize];
364 }
365 if (fr->IndexBuffer == NULL || fr->IndexBufferSize < draw_data->TotalIdxCount)
366 {
367 if (fr->IndexBuffer)
368 {
369 wgpuBufferDestroy(fr->IndexBuffer);
370 wgpuBufferRelease(fr->IndexBuffer);
371 }
372 SafeRelease(fr->IndexBufferHost);
373 fr->IndexBufferSize = draw_data->TotalIdxCount + 10000;
374
375 WGPUBufferDescriptor ib_desc =
376 {
377 NULL,
378 "Dear ImGui Index buffer",
379 WGPUBufferUsage_CopyDst | WGPUBufferUsage_Index,
380 fr->IndexBufferSize * sizeof(ImDrawIdx),
381 false
382 };
383 fr->IndexBuffer = wgpuDeviceCreateBuffer(g_wgpuDevice, &ib_desc);
384 if (!fr->IndexBuffer)
385 return;
386
387 fr->IndexBufferHost = new ImDrawIdx[fr->IndexBufferSize];
388 }
389
390 // Upload vertex/index data into a single contiguous GPU buffer
391 ImDrawVert* vtx_dst = (ImDrawVert*)fr->VertexBufferHost;
392 ImDrawIdx* idx_dst = (ImDrawIdx*)fr->IndexBufferHost;
393 for (int n = 0; n < draw_data->CmdListsCount; n++)
394 {
395 const ImDrawList* cmd_list = draw_data->CmdLists[n];
396 memcpy(vtx_dst, cmd_list->VtxBuffer.Data, cmd_list->VtxBuffer.Size * sizeof(ImDrawVert));
397 memcpy(idx_dst, cmd_list->IdxBuffer.Data, cmd_list->IdxBuffer.Size * sizeof(ImDrawIdx));
398 vtx_dst += cmd_list->VtxBuffer.Size;
399 idx_dst += cmd_list->IdxBuffer.Size;
400 }
401 int64_t vb_write_size = ((char*)vtx_dst - (char*)fr->VertexBufferHost + 3) & ~3;
402 int64_t ib_write_size = ((char*)idx_dst - (char*)fr->IndexBufferHost + 3) & ~3;
403 wgpuQueueWriteBuffer(g_defaultQueue, fr->VertexBuffer, 0, fr->VertexBufferHost, vb_write_size);
404 wgpuQueueWriteBuffer(g_defaultQueue, fr->IndexBuffer, 0, fr->IndexBufferHost, ib_write_size);
405
406 // Setup desired render state
407 ImGui_ImplWGPU_SetupRenderState(draw_data, pass_encoder, fr);
408
409 // Render command lists
410 // (Because we merged all buffers into a single one, we maintain our own offset into them)
411 int global_vtx_offset = 0;
412 int global_idx_offset = 0;
413 ImVec2 clip_scale = draw_data->FramebufferScale;
414 ImVec2 clip_off = draw_data->DisplayPos;
415 for (int n = 0; n < draw_data->CmdListsCount; n++)
416 {
417 const ImDrawList* cmd_list = draw_data->CmdLists[n];
418 for (int cmd_i = 0; cmd_i < cmd_list->CmdBuffer.Size; cmd_i++)
419 {
420 const ImDrawCmd* pcmd = &cmd_list->CmdBuffer[cmd_i];
421 if (pcmd->UserCallback != NULL)
422 {
423 // User callback, registered via ImDrawList::AddCallback()
424 // (ImDrawCallback_ResetRenderState is a special callback value used by the user to request the renderer to reset render state.)
425 if (pcmd->UserCallback == ImDrawCallback_ResetRenderState)
426 ImGui_ImplWGPU_SetupRenderState(draw_data, pass_encoder, fr);
427 else
428 pcmd->UserCallback(cmd_list, pcmd);
429 }
430 else
431 {
432 // Bind custom texture
433 ImTextureID tex_id = pcmd->GetTexID();
434 ImGuiID tex_id_hash = ImHashData(&tex_id, sizeof(tex_id));
435 auto bind_group = g_resources.ImageBindGroups.GetVoidPtr(tex_id_hash);
436 if (bind_group)
437 {
438 wgpuRenderPassEncoderSetBindGroup(pass_encoder, 1, (WGPUBindGroup)bind_group, 0, NULL);
439 }
440 else
441 {
442 WGPUBindGroup image_bind_group = ImGui_ImplWGPU_CreateImageBindGroup(g_resources.ImageBindGroupLayout, (WGPUTextureView)tex_id);
443 g_resources.ImageBindGroups.SetVoidPtr(tex_id_hash, image_bind_group);
444 wgpuRenderPassEncoderSetBindGroup(pass_encoder, 1, image_bind_group, 0, NULL);
445 }

447 // Project scissor/clipping rectangles into framebuffer space
448 ImVec2 clip_min((pcmd->ClipRect.x - clip_off.x) * clip_scale.x, (pcmd->ClipRect.y - clip_off.y) * clip_scale.y);
449 ImVec2 clip_max((pcmd->ClipRect.z - clip_off.x) * clip_scale.x, (pcmd->ClipRect.w - clip_off.y) * clip_scale.y);
450 if (clip_max.x <= clip_min.x || clip_max.y <= clip_min.y)
451 continue;
452
453 // Apply scissor/clipping rectangle, Draw
454 wgpuRenderPassEncoderSetScissorRect(pass_encoder, (uint32_t)clip_min.x, (uint32_t)clip_min.y, (uint32_t)(clip_max.x - clip_min.x), (uint32_t)(clip_max.y - clip_min.y));
455 wgpuRenderPassEncoderDrawIndexed(pass_encoder, pcmd->ElemCount, 1, pcmd->IdxOffset + global_idx_offset, pcmd->VtxOffset + global_vtx_offset, 0);
456 }
457 }
458 global_idx_offset += cmd_list->IdxBuffer.Size;
459 global_vtx_offset += cmd_list->VtxBuffer.Size;
460 }
