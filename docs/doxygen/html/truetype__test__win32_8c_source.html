32void print(float x, float y, char *text)
33{
34 // assume orthographic projection with units = screen pixels, origin at top left
35 glBindTexture(GL_TEXTURE_2D, tex);
36 glBegin(GL_QUADS);
37 while (*text) {
38 if (*text >= 32 && *text < 128) {
39 stbtt_aligned_quad q;
40 stbtt_GetBakedQuad(cdata, 1024,1024, *text-32, &x,&y,&q,1);//1=opengl & d3d10+,0=d3d9
41 glTexCoord2f(q.s0,q.t0); glVertex2f(q.x0,q.y0);
42 glTexCoord2f(q.s1,q.t0); glVertex2f(q.x1,q.y0);
43 glTexCoord2f(q.s1,q.t1); glVertex2f(q.x1,q.y1);
44 glTexCoord2f(q.s0,q.t1); glVertex2f(q.x0,q.y1);
45 }
46 ++text;
47 }
48 glEnd();
49}

98int mySetPixelFormat(HWND win)
99{
100 PIXELFORMATDESCRIPTOR pfd = { sizeof(pfd), 1, PFD_SUPPORT_OPENGL | PFD_DRAW_TO_WINDOW | PFD_DOUBLEBUFFER, PFD_TYPE_RGBA };
101 int pixel_format;
102 pfd.dwLayerMask = PFD_MAIN_PLANE;
103 pfd.cColorBits = 24;
104 pfd.cAlphaBits = 8;
105 pfd.cDepthBits = 24;
106 pfd.cStencilBits = 8;
107 pixel_format = ChoosePixelFormat(dc, &pfd);
108 if (!pixel_format) return FALSE;
109 if (!DescribePixelFormat(dc, pixel_format, sizeof(PIXELFORMATDESCRIPTOR), &pfd))
110 return FALSE;
111 SetPixelFormat(dc, pixel_format, &pfd);
112 return TRUE;
113}

115static int WINAPI WinProc(HWND wnd, UINT msg, WPARAM wparam, LPARAM lparam)
116{
117 switch (msg) {
