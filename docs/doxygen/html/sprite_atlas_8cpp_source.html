33std::shared_ptr<SpriteAtlas> SpriteAtlas::create(std::string_view jsonFile, const std::shared_ptr<Texture>& texture, bool flipAnchorY)
34{
35 std::string err;
36 const auto json = json11::Json::parse(getFileContents(jsonFile), err);
37 if (!err.empty())
38 {
39 LOG_ERROR("{}", err.c_str());
40 return {};
41 }
42 std::map<std::string, Sprite> sprites;
43 auto& list = json["frames"].array_items();
44 for (auto& spriteElement : list)
45 {
46 std::string name = spriteElement["filename"].string_value();
47 // "trimmed": true,
48 // "rotated": false, // rotated sprites not supported
49 // "spriteSourceSize": {"x":121,"y":94,"w":117,"h":131},
50 // "sourceSize": {"w":256,"h":257},
51 glm::ivec2 pos;
52 glm::ivec2 size;
53 glm::ivec2 sourcePos;
54 glm::ivec2 sourceSize;
55 glm::vec2 pivot;
56 pos.x = (int)spriteElement["frame"]["x"].number_value();
57 pos.y = (int)spriteElement["frame"]["y"].number_value();
58
59 size.x = (int)spriteElement["frame"]["w"].number_value();
60 size.y = (int)spriteElement["frame"]["h"].number_value();
61
62 if (spriteElement["rotated"].bool_value())
63 {
64 LOG_ERROR("Rotated sprites not supported: {}", jsonFile.data());
65 }
66
67 if (spriteElement["trimmed"].bool_value())
68 {
69 sourcePos.x = (int)spriteElement["spriteSourceSize"]["x"].number_value();
70 sourcePos.y = (int)spriteElement["spriteSourceSize"]["y"].number_value();
71 sourceSize.x = (int)spriteElement["sourceSize"]["w"].number_value();
72 sourceSize.y = (int)spriteElement["sourceSize"]["h"].number_value();
73 float spriteHeight = (float)spriteElement["spriteSourceSize"]["h"].number_value();
74 sourcePos.y = sourceSize.y - sourcePos.y - spriteHeight;
75 }
76 else
77 {
78 sourcePos = { 0, 0 };
79 sourceSize = size;
80 }
81 pos.y = texture->height() - pos.y - size.y;
82 if (spriteElement["pivot"].is_object())
83 {
84 pivot.x = (float)spriteElement["pivot"]["x"].number_value();
85 pivot.y = (float)spriteElement["pivot"]["y"].number_value();
86 }
87 else
88 {
89 pivot.x = 0.5f;
90 pivot.y = 0.5f;
91 }
92 if (flipAnchorY)
93 {
94 pivot.y = 1.0f - pivot.y;
95 }
96 Sprite sprite(pos, size, sourcePos, sourceSize, pivot, texture.get());
97 sprites.emplace(name, std::move(sprite));
98 }
99 return std::shared_ptr<SpriteAtlas>(new SpriteAtlas(std::move(sprites), texture, jsonFile));
100}

102Sprite SpriteAtlas::get(std::string_view name)
103{
104 if (m_sprites.find(name.data()) == m_sprites.end())
105 {
106 LOG_WARN("Cannot find sprite {} in spriteatlas", name.data());
107 return {};
108 }
109 return m_sprites[name.data()];
110}

134std::shared_ptr<SpriteAtlas> SpriteAtlas::createSingleSprite(const std::shared_ptr<Texture>& texture, std::string_view name, glm::vec2 pivot, glm::ivec2 pos, glm::ivec2 size)
135{
136 std::map<std::string, Sprite> sprites;
137 if (size == glm::ivec2{ 0, 0 })
138 {
139 size.x = texture->width();
140 size.y = texture->height();
141 }
142 Sprite sprite(pos, size, { 0, 0 }, size, pivot, texture.get());
143 sprites.emplace(std::pair<std::string, Sprite>(name, std::move(sprite)));
144 return std::shared_ptr<SpriteAtlas>(new SpriteAtlas(std::move(sprites), texture, std::string(name) + "_atlas"));
145}
