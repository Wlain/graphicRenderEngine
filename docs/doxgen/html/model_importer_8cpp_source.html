21// trim from start (in place)
22static inline void ltrim(std::string& s)
23{
24 s.erase(s.begin(), std::find_if(s.begin(), s.end(), [](int ch) {
25 return !std::isspace(ch);
26 }));
27}
28
29// trim from end (in place)
30static inline void rtrim(std::string& s)
31{
32 s.erase(std::find_if(s.rbegin(), s.rend(), [](int ch) {
33 return !std::isspace(ch);
34 }).base(),
35 s.end());
36}

38// trim from both ends (in place)
39static inline void trim(std::string& s)
40{
41 ltrim(s);
42 rtrim(s);
43}

45std::string concat(std::vector<std::string> v, int from)
46{
47 std::string res = v[from];
48 for (int i = from + 1; i < v.size(); i++)
49 {
50 res += " ";
51 res += v[i];
52 }
53 trim(res);
54 return res;

57std::string replaceAll(std::string& s, const std::string& search, const std::string& replace)
58{
59 for (size_t pos = 0;; pos += replace.length())
60 {
61 // Locate the substring to replace
62 pos = s.find(search, pos);
63 if (pos == std::string::npos) break;
64 // Replace by erasing and inserting
65 s.erase(pos, search.length());
66 s.insert(pos, replace);
67 }
68 return s;
69}

71std::string fixPath(std::string path)
72{
73 auto p = std::string(1, g_pathSeparator);
74 replaceAll(path, std::string(1, g_pathSeparator), p);
75 replaceAll(path, p + "." + p, p);
76 replaceAll(path, "\r", "");
77 return path;
78}

129// replace first occurrence of from in str with to
130bool replace(std::string& str, const std::string& from, const std::string& to)
131{
132 size_t startPos = str.find(from);
133 if (startPos == std::string::npos)
134 return false;
135 str.replace(startPos, from.length(), to);
136 return true;

241void parseMaterialLib(std::string& materialLib, std::vector<ObjMaterial>& materials)
242{
243 const int bufferSize = 256;
244 char buffer[bufferSize];
245 char buffer2[bufferSize];
246 std::stringstream ss{ materialLib };
247
248 while (ss.good())
249 {
250 ss.getline(buffer, bufferSize);
251 std::stringstream likeTokensizer{ buffer };
252 std::vector<std::string> tokens;
253 while (likeTokensizer.good())
254 {
255 likeTokensizer.getline(buffer2, bufferSize, ' ');
256 tokens.emplace_back(buffer2);
257 }
258 if (tokens.size() == 0)
259 {
260 continue;
261 }
262 if (tokens[0] == "newmtl")
263 {
264 Color zero{ 0, 0, 0, 0 };
265 auto name = concat(tokens, 1);
266 ObjMaterial material{ replaceAll(name, "\r", ""), zero, zero, zero, 50, 1 };
267 materials.push_back(material);
268 }
269 else

271 if (materials.size() == 0)

273 continue;
274 }
275 ObjMaterial& currentMat = materials.back();
276 if (tokens[0] == "Ka")

278 currentMat.ambientColor = toColorRGB(tokens);
279 }
280 else if (tokens[0] == "Kd")
281 {
282 currentMat.diffuseColor = toColorRGB(tokens);
283 }
284 else if (tokens[0] == "Ks")
285 {
286 currentMat.specularColor = toColorRGB(tokens);
287 }
288 else if (tokens[0] == "d")
289 {
290 currentMat.transparent = (float)atof(tokens[1].c_str());
291 }
292 else if (tokens[0] == "illum")
293 {
294 int illumMode = atoi(tokens[1].c_str());
295 currentMat.illuminationModes.push_back(static_cast<ObjIlluminationMode>(illumMode));
296 }
297 else if (tokens[0] == "map_Ka")
298 {
299 currentMat.textureMaps.push_back({ tokens[1], ObjTextureMapType::Ambient });
300 }
301 else if (tokens[0] == "map_Kd")
302 {
303 currentMat.textureMaps.push_back({ tokens[1], ObjTextureMapType::Diffuse });
304 }
305 else if (tokens[0] == "map_Ks")

307 currentMat.textureMaps.push_back({ tokens[1], ObjTextureMapType::Specular });
308 }
309 else if (tokens[0] == "map_Ns")

311 currentMat.textureMaps.push_back({ tokens[1], ObjTextureMapType::SpecularCoeficient });
312 }
313 else if (tokens[0] == "map_d")

315 }
316 else if (tokens[0] == "map_bump" || tokens[0] == "bump")
317 {
318 currentMat.textureMaps.push_back({ tokens[1], ObjTextureMapType::Bump });

331struct ObjVertexHash
332{
333 std::size_t operator()(const ObjVertex& k) const
334 {
335 return ((k.vertexPositionIdx << 16) ^ (k.textureIdx << 8)) ^ k.normalIdx;
336 }
337};

339struct ObjVertexEqual
340{
341 bool operator()(const ObjVertex& lhs, const ObjVertex& rhs) const
342 {
343 return lhs.vertexPositionIdx == rhs.vertexPositionIdx &&
344 lhs.textureIdx == rhs.textureIdx &&
345 lhs.normalIdx == rhs.normalIdx;
346 }

351int getCreateIndex(ObjVertex& vertexIndex, int nextIndex, ObjVertexHashTable& usedVertices)
352{
353 auto pos = usedVertices.find(vertexIndex);
354 if (pos != usedVertices.end())
355 {
356 return pos->second;
357 }
358 usedVertices.emplace(vertexIndex, nextIndex);
359 return nextIndex;

368std::shared_ptr<Material> createMaterial(std::string materialName, const std::vector<ObjMaterial>& matVector, std::string path)
369{
370 if (matVector.empty())
371 {
372 auto material = Shader::getStandardBlinnPhong()->createMaterial();
373 return material;
374 }
375 const ObjMaterial* foundMat = nullptr;
376 for (auto& v : matVector)
377 {
378 if (v.name == materialName || materialName.empty())
379 {
380 foundMat = &v;
381 break;
382 }
383 }
384 if (foundMat == nullptr)
385 {
386 LOG_ERROR("Could not find material {}", materialName);
387 foundMat = matVector.data();
388 }
389 auto shader = Shader::getStandardBlinnPhong();
390 auto material = shader->createMaterial();
391
392 material->setColor(foundMat->diffuseColor);
393 material->setSpecularity({ foundMat->specularColor.r, foundMat->specularColor.g, foundMat->specularColor.b, foundMat->specularCoefficient });
394 auto name = materialName;
395 for (auto& map : foundMat->textureMaps)
396 {
397 if (map.type == ObjTextureMapType::Diffuse)
398 {
399 material->setTexture(Texture::create().withFile(fixPath(path + map.filename)).build());
400 name += " " + map.filename;
401 }
402 }
403 material->setName(name);
404 return material;
405}

408std::shared_ptr<Mesh> ModelImporter::importObj(const std::filesystem::path& path, std::vector<std::shared_ptr<Material>>& outModelMaterials)
409{
410 std::string file = getFileContents(path);
411 std::vector<glm::vec3> vertexPositions;
412 std::vector<glm::vec4> textureCoords;
413 std::vector<glm::vec3> normals;
414 std::vector<ObjFace> faces;
415 std::vector<ObjGroup> namedObjects;
416 std::vector<ObjGroup> polygonGroups;
417 std::vector<SmoothGroup> smoothGroups;
418 std::vector<ObjMaterialChange> materialChanges;
419 std::vector<ObjMaterial> materials;
420 std::stringstream ss{ file };
421 const int bufferSize = 256;
422 char buffer[bufferSize];
423 char buffer2[bufferSize];
424 while (ss.good())
425 {
426 ss.getline(buffer, bufferSize);
427 std::stringstream likeTokensizer{ buffer };
428 std::vector<std::string> tokens;
429 while (likeTokensizer.good())
430 {
431 likeTokensizer.getline(buffer2, bufferSize, ' ');
432 for (char& i : buffer2)
433 {
434 if (std::isspace(i))
435 {
436 i = '\0';
437 }
438 }
439 tokens.emplace_back(buffer2);
440 }
441 if (tokens.empty())
442 {
443 continue;
444 }
445 int currentIndex = static_cast<int>(faces.size()) + 1;
446 if (tokens[0] == "#")
447 { // comment
448 // ignore
449 }
450 else if (tokens[0] == "v")
451 { // vertex position
452 glm::vec3 vertexPosition = toVec3(tokens);
453 vertexPositions.push_back(vertexPosition);
454 }
455 else if (tokens[0] == "vt")
456 { // vertex texture coordinates
457 glm::vec4 textureCoord = toVec4(tokens);
458 textureCoords.push_back(textureCoord);
459 }
460 else if (tokens[0] == "vn")
461 { // vertex normal
462 glm::vec3 normal = toVec3(tokens);
463 normals.push_back(normal);
464 }
465 else if (tokens[0] == "f")
466 { // face
467 faces.push_back(toObjFace(tokens));
468 }
469 else if (tokens[0] == "mtllib")
470 { // material library
471 std::string parentPath = path.parent_path().c_str();
472 std::string materialLib = getFileContents(parentPath + '/' + concat(tokens, 1));
473 parseMaterialLib(materialLib, materials);
474 }
475 else if (tokens[0] == "usemtl")
476 { // use material
477 materialChanges.push_back({ currentIndex, concat(tokens, 1) });
478 }
479 else if (tokens[0] == "o")
480 { // named object
481 namedObjects.push_back({ currentIndex, concat(tokens, 1) });
482 }
483 else if (tokens[0] == "g")
484 { // polygon group
485 polygonGroups.push_back({ currentIndex, concat(tokens, 1) });
486 }
487 else if (tokens[0] == "s")
488 { // smoothing groups
489 int smoothingGroup = 0; // 0 = no smoothing
490 if (tokens[1] != "off")
491 {
492 smoothingGroup = atoi(tokens[1].c_str());
493 }
494 smoothGroups.push_back({ currentIndex, smoothingGroup });
495 }
496 }
497 int vertexCount{};
498 int faceCount{};
499 std::vector<ObjInterleavedIndex> indices;
500 ObjInterleavedIndex* currentIndex = nullptr;
501 auto materialChange = materialChanges.cbegin();
502 bool includeTextureCoordinates = !textureCoords.empty();
503 bool includeNormals = !normals.empty();
504 ObjVertexHashTable usedVertices{ 42, ObjVertexHash{}, ObjVertexEqual{} };
505 std::vector<glm::vec3> finalPositions;
506 std::vector<glm::vec4> finalTextureCoordinates;
507 std::vector<glm::vec3> finalNormals;
508 bool firstFace = true;
509 for (auto& face : faces)
510 {
511 faceCount++;
512 if (materialChange != materialChanges.end() || firstFace)
513 {
514 if (firstFace && materialChange == materialChanges.end())
515 {
516 indices.push_back({ "", {} });
517 currentIndex = &indices.back();
518 }
519 else if (materialChange->faceIndex == faceCount)
520 {
521 bool foundMaterial = false;
522 for (auto& idx : indices)
523 {
524 if (idx.materialName == materialChange->name)
525 {
526 foundMaterial = true;
527 currentIndex = &idx;
528 break;
529 }
530 }
531 if (!foundMaterial)
532 {
533 indices.push_back({ materialChange->name, {} });
534 currentIndex = &indices.back();
535 }
536 materialChange++;
537 }
538 firstFace = false;
539 }
540
541 for (int i = 2; i < face.size(); i++)
542 {
543 int triangle[] = { 0, i - 1, i };
544 for (int triangleIndex : triangle)
545 {
546 auto& vertexIndexObject = face[triangleIndex];
547 int vertexIndex = getCreateIndex(vertexIndexObject, vertexCount, usedVertices);
548 bool existInInterleavedData = vertexIndex != vertexCount;
549 if (!existInInterleavedData)
550 {
551 // read data
552 glm::vec3 vertexPosition = vertexPositions[vertexIndexObject.vertexPositionIdx - 1];
553 glm::vec4 textureCoord{ 0, 0, 0, 0 };
554 if (vertexIndexObject.textureIdx > 0 && includeTextureCoordinates)
555 {
556 textureCoord = textureCoords[vertexIndexObject.textureIdx - 1];
557 }
558 glm::vec3 normal;
559 if (vertexIndexObject.normalIdx > 0 && includeNormals)
560 {
561 normal = normals[vertexIndexObject.normalIdx - 1];
562 }
563 // write interleaved data
564 finalPositions.push_back(vertexPosition);
565 if (includeTextureCoordinates)
566 {
567 finalTextureCoordinates.push_back(textureCoord);
568 }
569 if (includeNormals)
570 {
571 finalNormals.push_back(normal);
572 }
573 vertexCount++;
574 }
575 currentIndex->vertexIndices.emplace_back(vertexIndex);
576 }
577 }
578 }
579 // remove unused materials
580 indices.erase(std::remove_if(indices.begin(), indices.end(), [](const ObjInterleavedIndex& a) { return a.vertexIndices.empty(); }),
581 indices.end());
582
583 auto meshBuilder = Mesh::create();
584 meshBuilder.withPositions(finalPositions);
585 if (includeTextureCoordinates)
586 {
587 meshBuilder.withUvs(finalTextureCoordinates);
588 }
589 if (includeNormals)
590 {
591 meshBuilder.withNormals(finalNormals);
592 }
593
594 for (int i = 0; i < indices.size(); i++)
595 {
596 std::string parentPath = path.parent_path().c_str();
597 outModelMaterials.push_back(createMaterial(indices[i].materialName, materials, parentPath + '/'));
598 meshBuilder.withIndices(indices[i].vertexIndices, Mesh::Topology::Triangles, i);
599 }
600
601 return meshBuilder.build();
602}
